    # 一：上下翻转二叉树

    给你一个二叉树的根节点 root ，请你将此二叉树上下翻转，并返回新的根节点。

你可以按下面的步骤翻转一棵二叉树：

原来的左子节点变成新的根节点
原来的根节点变成新的右子节点
原来的右子节点变成新的左子节点

示例 1：
输入：root = [1,2,3,4,5]
输出：[4,5,2,null,null,3,1]

示例 2：
输入：root = []
输出：[]

示例 3：
输入：root = [1]
输出：[1]

    # (一)：迭代法则

    • 翻转二叉树类似于翻转链表，翻转链表主要让头节点为尾节点，尾节点为头节点。那么这个题目中，要不断迭代，让右节点为左节点，让根节点为右节点，左节点为根节点，然后再按照同样的规则迭代左节点，直到root为nil。这个题目难度较大
    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}

func upsideDownBinaryTree(root *TreeNode) *TreeNode {

	// 看做两个指针
	var  left, right, father *TreeNode

	for root != nil {

		// 前四步都是让root的左右节点为nil
		left = root.Left

		root.Left = right

		right = root.Right

		root.Right = father

		// 保存根节点，作为下次迭代的右节点
		father = root

		// 保存左节点，作为下次迭代判断条件
		root = left

	}

	return father

}



func main() {

	two := &TreeNode{Val: 2, Left: &TreeNode{Val: 4}, Right: &TreeNode{Val: 5}}

	root := &TreeNode{
		Val: 1,
		Left: two,
		Right: &TreeNode{Val: 3},
	}

	ret := upsideDownBinaryTree(root)
	fmt.Println(ret)


}

    # 二：二叉树的直径

    给定一棵二叉树，你需要计算它的直径长度。一棵二叉树的直径长度是任意两个结点路径长度中的最大值。这条路径可能穿过也可能不穿过根结点。

 

示例 :
给定二叉树

          1
         / \
        2   3
       / \     
      4   5    
返回 3, 它的长度是路径 [4,2,1,3] 或者 [5,2,1,3]。

 

注意：两结点之间的路径长度是以它们之间边的数目表示

    # (一)：递归法

    • 首先需要明确一个概念：树的深度指的是 以根节点为起点，左节点的深度与右节点的深度最大值+1。例如下面的三个节点的二叉树，根节点①的深度等于节点②的深度和节点③的深度最大值+1，那么要计算节点①的深度就需要先计算出节点②的深度，节点②的深度也由左节点和右节点深度的最大值+1得到。如果一个节点是nil，那么深度为0，因此节点②的左右两个节点都是nil，那么节点②的深度是1，节点③的深度也是1，因此节点①的深度就是max(1,1)+1=2，这个计算看过就是一个递归过程
      1
 / \
2   3

    • 因此需要首先写出计算节点深度的递归算法，递归的终止条件是节点是nil，返回深度为0
    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}

func max(a, b int) int {
	if a>b {
		return a
	} else {
		return b
	}
}

func depth(node *TreeNode) int {
	// 递归结束条件
	if node == nil {
		return 0
	}

	leftDepth := depth(node.Left)
	rightDepth := depth(node.Right)
	currentNodeDepth := max(leftDepth, rightDepth) + 1
	return currentNodeDepth
}


func main() {

	node := &TreeNode{Val: 1, Left: &TreeNode{Val: 2}, Right: &TreeNode{Val: 3}}

	ret := depth(node)
	fmt.Println(ret)

}

    • 可以获取到所有节点深度后，就可以让计算当前节点的路径，路径是：左节点深度 + 右节点深度 + 1。代码实现如下
    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}

func max(a, b int) int {
	if a>b {
		return a
	} else {
		return b
	}
}


// 在深度递归函数depth内部统计每个节点为根节点经过的最多节点数
func depth(node *TreeNode, targetAns *int) int {

	// 递归结束条件
	if node == nil {
		return 0
	}

	// 获取左节点深度
	leftDepth := depth(node.Left, targetAns)
	// 获取右节点深度
	rightDepth := depth(node.Right, targetAns)
	// 获取当前节点的深度
	currentNodeDepth := max(leftDepth, rightDepth) + 1
	// 获取所有节点，以其为根节点，经过的节点数
	nodes := leftDepth + rightDepth + 1
	// 比较nodes与targetAns的大小
	if nodes > *targetAns {
		// 边数对于节点数减一
		*targetAns = nodes - 1
	}

	// 返回当前节点的深度
	return currentNodeDepth
}

// diameterOfBinaryTree 返回最大的边数，也就是二叉树的直径
func diameterOfBinaryTree(root *TreeNode) int {

	// 用于统计最大节点数
	targetAns := 0

	depth(root, &targetAns)
	return targetAns
}

func main() {

	//node := &TreeNode{Val: 1, Left: &TreeNode{Val: 2, Left: &TreeNode{Val: 4}, Right: &TreeNode{Val: 5}}, Right: &TreeNode{Val: 3}}
	node := &TreeNode{Val: 1, Left: &TreeNode{Val: 2}}
	ret := diameterOfBinaryTree(node)

	fmt.Println(ret)

}

    • 这里再总结一下递归的思想：
    任何问题：如果要知道A的情况，那么必须知道B，如果要知道B的情况，必须先知道C。这样类型的问题就可以使用递归解决，递归的终止条件一般会是节点为nil，某个为为0等

写递归函数的时候，可以先思考最后一个情况，不能先思考第一个情况。例如在递归链表或者树结构的时候，可以看看如果当前node是叶子节点需要如何计算，如何判断终止条件。然后再通过几个情况推算一下就可以写出递归函数了

    # (一)：二分搜索树中序遍历：递归法

    • 基于二分搜索树的特性：根节点大于左节点，小于右节点。因此，二分搜索树的中序遍历就是从小到大的排序过程，因此只需要对二分搜索树进行k次中序遍历，就可以找到第k小的元素
    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}

func NewNode(val int) *TreeNode {
	return &TreeNode{
		Val: val,
		Left: nil,
		Right: nil,
	}
}

// 返回根节点
func (t *TreeNode) add (node *TreeNode, val int) *TreeNode {

	// 判断叶子节点是否为空
	if node == nil {
		return NewNode(val)
	}

	// 如果小于，添加到左边
	if val < node.Val {
		node.Left = t.add(node.Left, val)

	// 如果大于，添加到右边
	}else if val > node.Val {
		node.Right = t.add(node.Right, val)
	}

	return node

}

func (t *TreeNode) Add (val int)  {
	// 此时t就是根节点
	 t.add(t, val)
}

func inOrder(node *TreeNode, k int, arr *[]int)  {
	if node == nil {
		return
	}

	inOrder(node.Left, k, arr)
	*arr = append(*arr, node.Val)
	if len(*arr) == k {
		return
	}
	inOrder(node.Right, k, arr)

}

func kthSmallest(root *TreeNode, k int) int {
	arr := make([]int, 0)
	inOrder(root, k, &arr)
	return arr[k-1]
}

func main() {

	root := NewNode(5)
	root.Add(3)
	root.Add(6)
	root.Add(2)
	root.Add(4)
	root.Add(1)


	ret := kthSmallest(root, 3)
	fmt.Println(ret)

}

    func kthSmallest(root *TreeNode, k int) int {

			return root.Val
		}
		
		// 让右节点也入栈
		root = root.Right
	}

}

    # (一)：递归法

    • 一个节点的深度是： 左节点深度与右节点深度的最大值 + 1 ，因此要知道根节点的深度，就需要知道左节点的深度和右节点的深度，如果一个节点为nil，那么深度为0
    func max(n1, n2 int) int {
	if n1 > n2 {
		return n1
	}
	return n2
}

func depth(node *TreeNode, maxDepth *int) int {

	// 终止条件
	if node == nil {
		return 0
	} 
	
	curDepth := max(depth(node.Left, maxDepth), depth(node.Right, maxDepth)) + 1
	if curDepth > *maxDepth {
		*maxDepth = curDepth
	}
	
	return curDepth

}

func maxDepth(root *TreeNode) int {

	maxDepth := 0
	depth(root, &maxDepth)
	return maxDepth
}

    # 五：二叉树中的最大路径和

    路径 被定义为一条从树中任意节点出发，沿父节点-子节点连接，达到任意节点的序列。同一个节点在一条路径序列中 至多出现一次 。该路径 至少包含一个 节点，且不一定经过根节点。

路径和 是路径中各节点值的总和。

给你一个二叉树的根节点 root ，返回其 最大路径和 。

    # (一)：递归法

    • 本题与求解二叉树的最大直径类似。在最大直径求解过程中，需要计算一个节点的最大深度，一个节点的最大深度为 左节点深度 与 右节点深度 的最大值 + 1。这里计算的是一个节点贡献值，一个节点的贡献值为 左节点贡献值 与 右节点贡献值 的最大值 + 当前节点的值。那么一个节点的路径和就是 左节点的贡献值 + 右节点的贡献值 + 当前节点的值。只需要从二叉树的根节点递归遍历整棵树，终止条件是node==nil，贡献值为0。叶子节点 例如 节点⑨，最大贡献值是9，最大路径和也是9。非叶子节点 例如 节点 20，最大贡献值是15+20=35，路径和是15+7+20=42。递归算法复杂度为O(n)，n是节点数
    func max(x, y int) int {
	if x > y {
		return x
	}
	return y
}


func maxPathSum(root *TreeNode) int {

	maxPathNum := math.MinInt32

	var getMaxGain func(node *TreeNode) int

	// 计算节点的最大贡献值
	getMaxGain = func(node *TreeNode) int {

		// 终止条件，nil节点，贡献值为0
		if node == nil {
			return 0
		}

		// 计算左节点的贡献值, 如果贡献值小于0 ，那么就取0
		leftGain := max(getMaxGain(node.Left), 0)
		// 计算右节点的贡献值
		rightGain := max(getMaxGain(node.Right), 0)

		// 计算以当前节点为根节点的路径和
		curPathNum := leftGain + rightGain + node.Val
		if curPathNum > maxPathNum {
			maxPathNum = curPathNum
		}

		// 返回当前节点的贡献值
		return max(leftGain, rightGain) + node.Val

	}

	// 函数的返回值是root节点的贡献值
	getMaxGain(root)

	return maxPathNum

}

    # (一)：找到路径的公共节点

    • 以节点③和节点⑤为例，最近的公共节点是节点④，节点⑥也是公共节点，但是不是最近的，是最远的。主要思想是：基于二分搜索树的性质，搜索到给定的节点，将搜索过程中的所有节点添加到数组中。然后遍历两个节点的路径节点数组，如果值相等，那么就是公共节点，最近的公共节点是最后一个值相等的节点。算法实现如下
    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}


func getPathNodes(root, node *TreeNode) []*TreeNode {

	pathNodes := make([]*TreeNode, 0)

	for root.Val != node.Val {

		pathNodes = append(pathNodes, root)

		if node.Val < root.Val {
			root = root.Left
		}else {
			root = root.Right
		}

	}
	// 自己也要添加到数组中
	pathNodes = append(pathNodes, root)

	return pathNodes

}


func lowestCommonAncestor(root, p, q *TreeNode)  *TreeNode {

	pPathNodes := getPathNodes(root, p)
	qPathNodes := getPathNodes(root, q)

	// 路径中的公共节点
	publicNodes := []*TreeNode{}

	// 两个数组同时遍历
	i:=0
	for {
		// 循环退出条件
		if i>=len(pPathNodes) || i>=len(qPathNodes) {
			break
		}

		if pPathNodes[i] == qPathNodes[i] {
			publicNodes = append(publicNodes, pPathNodes[i])
		}

		i++
	}

	// 返回最后一个公共节点
	return publicNodes[len(publicNodes)-1]
}

	node9 := &TreeNode{Val: 9}
	node8 := &TreeNode{Val: 8, Left: node7, Right: node9}
	node6 := &TreeNode{Val: 6, Left: node2, Right: node8}

	ret := lowestCommonAncestor(node6, node2, node4)
	fmt.Println(ret)

}

    # (一)：借助队列实现

    • 层序遍历需要借助队列来实现，每次迭代队列的时候，只取队列最后一个节点，并且更新队列长度，只要节点左孩子不为nil，那么将左孩子添加到队列头部，只要节点右孩子不为nil，那么将右孩子添加到队列头部
    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}

func levelOrder(root *TreeNode) {


	// 创建一个队列，队列中的元素是*TreeNode类型
	queue := make([]*TreeNode, 0)

	// 在队列中添加头部元素
	queue = append(queue, root)

	// 如果队列中有元素，每次迭代的时候，队列中保存的是一层从右到左的节点
	for len(queue) != 0 {

		// 获取队列中最后一个节点，并且更新队列长度
		cur := queue[len(queue)-1]
		queue = queue[:len(queue)-1]

		fmt.Println(cur.Val)

		// 如果左孩子不为空，将左孩子加入到队列头部(合并两个数组)
		if cur.Left != nil {
			queue = append([]*TreeNode{cur.Left}, queue...)
		}


		// 如果右孩子不为空，将右孩子接入到队列头部
		if cur.Right != nil {
			queue = append([]*TreeNode{cur.Right}, queue...)
		}

	}

}

    type TreeNode struct {
	Val int
	Left *TreeNode
	Right *TreeNode
}

func levelOrder(root *TreeNode) [][]int {

	targetNums := [][]int{}

	// 根节点为nil的时候，直接返回
	if root ==  nil {
		return targetNums
	}

	// 创建一个队列，队列中的元素是*TreeNode类型
	queue := make([]*TreeNode, 0)
	
	// 头部元素作为第一层
	// 在队列中添加头部元素
	queue = append(queue, root)
	
	// 这个queue是一整层的节点
	for len(queue) != 0 {

		// 创建临时队列，用于保存一层的节点
		tmpQueue := make([]*TreeNode, 0)
		levelNums := make([]int, 0)

		// 再次迭代queue的目的是：记录每一层节点的值，并且将每一层节点的左右孩子添加到临时队列中
		for len(queue) != 0 {
			node := queue[len(queue)-1] // 获取队列中最后一个节点
			queue = queue[:len(queue)-1] // 更新队列长度

			levelNums = append(levelNums, node.Val) // 记录当前层每一个节点的值

			// 如果左孩子不为空，添加到临时队列中
			if node.Left != nil {
				tmpQueue = append([]*TreeNode{node.Left}, tmpQueue...)
			}
			if node.Right != nil {
				tmpQueue = append([]*TreeNode{node.Right}, tmpQueue...)
			}
		}
		
		// 添加一层节点值
		targetNums = append(targetNums, levelNums)
		
		// 将临时队列中的一层节点添加到队列中
		queue = append(queue, tmpQueue...)


	}

	return targetNums

}

func main() {

	root := &TreeNode{Val: 3, Left: &TreeNode{Val: 9}, Right: &TreeNode{Val: 20, Left: &TreeNode{Val: 15}, Right: &TreeNode{Val: 7}}}
	ret := levelOrder(root)
	fmt.Println(ret)
}

func (Codec) serialize(root *TreeNode) string {
	// 使用 &strings.Builder{}实例对字符串进行拼接操作
	sb := &strings.Builder{}

	// 创建递归函数
	var dfs func(*TreeNode)
	dfs = func(node *TreeNode) {
		// 如果node为kong，直接return
		if node == nil {
			// 这里写入nil的时候加了 逗号
			sb.WriteString("nil,")
			return
		}

		sb.WriteByte(',')

		dfs(node.Left)
		dfs(node.Right)
	}

	dfs(root)
	return sb.String()
}

func (Codec) deserialize(data string) *TreeNode {
	// 通过逗号分割字符串
	nums := strings.Split(data, ",")

	var build func() *TreeNode

	build = func() *TreeNode {
		// 如果头节点是nil，那么从nums剔除头节点，返回nil
		if nums[0] == "nil" {

		// 这里会掉调用左边的build()，然后再调用右边的build()
		return &TreeNode{val, build(), build()}
	}

	return build()
