    # 一：二叉树的最近公共祖先

    给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。

百度百科中最近公共祖先的定义为：“对于有根树 T 的两个节点 p、q，最近公共祖先表示为一个节点 x，满足 x 是 p、q 的祖先且 x 的深度尽可能大（一个节点也可以是它自己的祖先）。”

    # (一)：回溯法

    • 回溯算法是以递归的形式体现的，但是又不同于递归的思想。回溯算法中，如果求解不到目标结果，会回退到上一步，尝试新的求解路径。回溯算法的核心思想是：在一棵 隐式的树（看不见的树） 上进行一次深度优先遍历

    • 本题使用了回溯算法的思想，找两个节点。算法的终止条件有两个，一个是root为nil，返回nil，一个是root就是需要找的节点，返回root。回溯算法迭代过程中对左节点和右节点分别进行递归，然后判断递归的两个返回值left和right。两个返回值组合有四种情况：1：如果left和right都为nil，说明p和q都不在root的左右子树中，返回nil，结束递归；2：如果left和right都不为nil，说明p和q在root的左右子树中，返回root，继续递归root的左节点和右节点；3：如果left不为nil，right为nil，说明p和q都在left中，返回left，继续递归left；4：如果left为nil，right不为nil，返回right，继续递归right

    func lowestCommonAncestor(root, p, q *TreeNode) *TreeNode {

	// 终止条件：如果节点为空，返回nil
	if root == nil {
		return nil
	}

	// 终止条件：如果root为q节点或者p节点，那么直接返回root，
	if root == p || root == q {
		return root
	}

	// 先坐后右，再处理左右的输出

	// 左节点递归
	left := lowestCommonAncestor(root.Left, p, q)
	// 右节点递归
	right := lowestCommonAncestor(root.Right, p, q)


	if left == nil && right == nil { // 如果left和right同时为空，说明 root的左节点和右节点都不包含p和q节点，返回nil
		return nil
	}else if left != nil && right != nil { // 如果left和right同时不为空，说明 p 和 q 分别在root节点的两侧，那么最近公共祖先就是root
		return root
	}else if left != nil && right == nil { // 如果left不为空，right为空，说明 p 和 q都在left中，那么返回left即可，说明left是公共祖先
		return left
	}else {   // 如果right不为空，left为空，说明 p 和 q都在right中，那么返回right即可，说明right是公共祖先
		return right
	}

}

    # (一)：回溯

    • 由于这个体可能有点像全排列问题，想着基于全排列来做。在 "()()" 选择4个字符串做全排列，一共有24中可能性。但是里面有重复的，因为 在"()()"字符串中，索引0和索引2是重复的字符，索引1和索引3是重复字符。下图这种全排列的方式，虽然使用了bool数组，标记了是否使用过的，但是也会有重复，因此不能使用全排列的方式实现

    • 再仔细分析后发现：要使得一个字符串有效，必须满足在添加')'的时候，必须保证剩余可选的')'数量比'('的数量多，那么才能选择')'添加进path。并且只有还有左括号剩余就应该添加'('。算法实现如下
    func generateParenthesis(n int) []string {

	targetStr := []string{}

	var dfs func(leftRemain, rightRemain int, path []byte)

	dfs = func(leftRemain, rightRemain int, path []byte) {
		if len(path) == 2 * n {
			tmp := make([]byte, len(path))
			copy(tmp, path)
			targetStr = append(targetStr, string(tmp))
		}



		// 如果剩余的左边括号数量大于0，可以一直添加
		if leftRemain > 0 {
			path = append(path, '(')
			leftRemain--
			dfs(leftRemain, rightRemain, path)
			path = path[:len(path)-1]
			leftRemain++
		}

		// 如果剩余的右边括号数量大于剩余的左边括号数量
		if rightRemain > leftRemain {
			path = append(path,  ')')
			rightRemain--
			dfs(leftRemain, rightRemain, path)
			path = path[:len(path)-1]
			rightRemain++
		}

	}

	dfs(n, n, []byte{})

	return targetStr

}


func main() {

	ret := generateParenthesis( 3)
	fmt.Println(ret)

}

    # (一)：回溯法

    • 求数组的子集其实就是求数组的组合，只是在这个组合过程中并没有终止条件，每个节点的path都会添加到targetNums中。
    func subsets(nums []int) [][]int {

	targetNums := [][]int{}

	var dfs func(nums, path []int, startIndex int, )

	dfs = func(nums, path []int, startIndex int) {

		// 每个节点都添加
		tmp := make([]int, len(path))
		copy(tmp, path)
		targetNums = append(targetNums, tmp)

		for i:=startIndex; i<len(nums); i++ {
			path = append(path, nums[i])
			dfs(nums, path, i+1) // 组合问题在这是i+1
			path = path[:len(path)-1]
		}

	}

	dfs(nums, []int{}, 0)

	return targetNums

}


func main() {

	ret := subsets( []int{1,2,3})
	fmt.Println(ret)

}

    # (一)：回溯

    • 算法实现如下，排列问题需要使用一个bool数组标记已经选择过的元素，并且在递归的时候，传递的是startIndex参数
    func permute(nums []int) [][]int {

	targetNums := [][]int{}
	used := make([]bool, len(nums))

	var dfs func(nums, path []int, startIndex int, used []bool)
	dfs = func(nums, path []int, startIndex int, used []bool) {

		if len(path) == len(nums) {
			tmp := make([]int, len(path))
			copy(tmp, path)
			targetNums = append(targetNums, tmp)
		}

		for i:=startIndex; i<len(nums); i++ {
			if !used[i] {
				path = append(path, nums[i])
				used[i] = true
				dfs(nums, path, startIndex, used)
				path = path[:len(path)-1]
				used[i] = false
			}

		}
	}

	dfs(nums, []int{}, 0, used)

	return targetNums

}


func main() {

	ret := permute( []int{1,2,3})
	fmt.Println(ret)

}

    # (一)：回溯

    • 由题目的意思是一个组合题目：Cn=4k=2 = A42 / A22 = 6 。其实就是从【1,2,3,4】中挑选2个数，有多少组合的情况。当前问题的回溯树如下

    • 算法实现如下
    // 给一个n，给一个k
func combine(n int, k int) [][]int {

	targetNums := [][]int{}

	var dfs func(n, k, startIndex int, path []int)

	dfs = func(n, k, startIndex int, path []int) {

		// 判断path路径是否满足k
		if len(path) == k {
			tmp := make([]int, len(path))
			copy(tmp, path)
			targetNums = append(targetNums, tmp)
			return
		}

		// 如果path路径不满足条件，一直向path中添加元素
		for i:=startIndex; i<=n; i++ { // 第二轮循环的时候：i=2，path=[]
			path = append(path, i)     // 添加元素
			dfs(n, k, i+1, path)       // 继续递归向下添加元素, 此时path=[1]
			path = path[:len(path)-1]  // 剔除最后一个元素
		}

	}

	dfs(n, k, 1, []int{})

	return targetNums

}

func main() {

	ret := combine(4, 2)
	fmt.Println(ret)

}

    # (二)：回溯的减枝优化

    • 如果 给定的 n = 6， k = 4。那么如果path中有1个元素了，还需要选择3个元素，那么应该是【4,5,6】，因此如果startIndex > 4，应该结束函数。由此说明startIndex是有上界的，这里的startIndex的上界是4

    • 下面假设 n = 6 ，k = 4，推算startIndex最大值的公式

    len(path) == 1 的时候，接下来要选择 3 个数，搜索起点最大是 4，被选的组合是 [4, 5, 6]
len(path) == 2 的时候，接下来要选择 2 个数，搜索起点最大是 5，被选的组合是 [5, 6]
len(path) == 3 的时候，接下来要选择 1 个数，搜索起点最大是 6，被选的组合是 [6]

    // 给一个n，给一个k
func combine(n int, k int) [][]int {

	targetNums := [][]int{}

	var dfs func(n, k, startIndex int, path []int)

	dfs = func(n, k, startIndex int, path []int) {

		// 判断path路径是否满足k
		if len(path) == k {
			tmp := make([]int, len(path))
			copy(tmp, path)
			targetNums = append(targetNums, tmp)
			return
		}

		// 剪枝
		if startIndex > n + 1 - k + len(path) {
			return
		}

		// 如果path路径不满足条件，一直向path中添加元素
		for i:=startIndex; i<=n; i++ {
			path = append(path, i)
			dfs(n, k, i+1, path)
			path = path[:len(path)-1]
		}

	}

	dfs(n, k, 1, []int{})

	return targetNums

}

func main() {

	ret := combine(4, 2)
	fmt.Println(ret)

}

    # (一)：回溯

    • 先画出回溯图

    • 排列在组合的基础上，需要引入一个bool数组，用于记录哪些字符已经使用了，每次添加元素的时候，必须保证此元素没有被添加过，添加过的元素必须在bool数组中标记为true，删除path中的元素必须在bool数组中标记为false
    func allSubChars(s string, k int) []string {

	targetStr := []string{}

	// 初始化used
	used := make([]bool, len(s))

	var dfs func(s string, k int, path []byte, startIndex int, used []bool)

	dfs = func(s string, k int, path []byte, startIndex int, used []bool) {

		// 终止条件：如果path有两个元素那么就终止，添加到targetStr
		if len(path) == k {
			tmp := make([]byte, len(path))
			copy(tmp, path)
			targetStr = append(targetStr, string(tmp))
			return
		}

		for i:=startIndex; i<len(s); i++ {
			if !used[i] {
				path = append(path, s[i])
				used[i] = true
				dfs(s, k, path, startIndex, used)
				used[i] = false
				path = path[:len(path)-1]
			}

		}

	}

	dfs(s, k, []byte{}, 0, used)

	return targetStr

}


func main() {

	ret := allSubChars("abc", 2)
	fmt.Println(ret)

}
