    # 一：两数之和

    给定一个整数数组 nums 和一个整数目标值 target，请你在该数组中找出两个数的和与为目标值 target  的值相等那两个整数，并返回它们的数组下标。

你可以假设每种输入只会对应一个答案。但是，数组中同一个元素在答案里不能重复出现。


输入：nums = [2,7,11,15], target = 9
输出：[0,1]
解释：因为 nums[0] + nums[1] == 9 ，返回 [0, 1]

    # (一)：哈希表

    • 这里使用哈希表，只需要遍历一次数组即可。算法思想是：在遍历数组的时候，从哈希表中找对应的目标数据(目标数据=target-遍历数据)，如果没有找到，将遍历数据添加到哈希表中，如果找到了那么取出目标数据的索引，目标数据的索引和遍历数据的索引就是返回值。Map的key是数组的元素值，value是数组元素的索引
    func twoSum(nums []int, target int) []int {

	hashTable := make(map[int]int, 0)

	for i, v := range nums {
		p, ok := hashTable[target-v]
		if ok {
			return []int{p, i}
		}else {
			hashTable[v]=i
		}
	}
	return make([]int, 0)

}

func main() {

	nums := []int{3,2,4}
	target := 6

	ret := twoSum(nums, target)
	fmt.Println(ret)

}

    # 二：差的绝对值为 K 的数对数目

    给你一个整数数组 nums 和一个整数 k ，请你返回数对 (i, j) 的数目，满足 i < j 且 |nums[i] - nums[j]| == k 。

|x| 的值定义为：

如果 x >= 0 ，那么值为 x 。
如果 x < 0 ，那么值为 -x 。


示例 1：
输入：nums = [1,2,2,1], k = 1
输出：4
解释：差的绝对值为 1 的数对为：
- [1,2]
- [2,1]

示例 2：
输入：nums = [1,3], k = 3
输出：0
解释：没有任何数对差的绝对值为 3 。

    # (一)：哈希表

    • (1,2), (2,1)算两个数对，有顺序要求。这里需要分情况讨论，先处理 a > b 的情况，再处理 a < b 的情况。在hashTable中，key是元素的值，value是元素的出现的个数
    func countKDifference(nums []int, k int) int {

	// 哈希表获取数对
	hashTable := map[int]int{}
	targetNum := 0

	for i:=0; i<len(nums); i++ {

		// 先处理 a > b 的情况
		value, ok := hashTable[nums[i] - k]
		if ok { // 如果有值，那么需要看有几个
			targetNum += value

		}
		
		// 再处理 a < b 的情况
		value, ok = hashTable[nums[i] + k]
		if ok {
			targetNum += value
		}

		// 每次都需要将元素加入到哈希表中
		hashTable[nums[i]]++

	}

	return targetNum
}

func main() {


	ret := countKDifference([]int{9,3,1,1,4,5,4,9,5,10}, 1)
	fmt.Println(ret)

}

    # 三：环形链表

    给你一个链表的头节点 head ，判断链表中是否有环。

如果链表中有某个节点，可以通过连续跟踪 next 指针再次到达，则链表中存在环。 为了表示给定链表中的环，评测系统内部使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。注意：pos 不作为参数进行传递 。仅仅是为了标识链表的实际情况。

如果链表中存在环 ，则返回 true 。 否则，返回 false 。

    # (一)：哈希表 存储链表节点

    • 可以遍历链表，将链表的节点添加到哈希表中，注意不是添加链表节点的val，因为相同的value，不能说明是同一个节点
    type ListNode struct {
	Val int
	Next *ListNode
}

func hasCycle(head *ListNode) bool {

	hashTable := map[*ListNode]struct{}{}

	// 如果节点为nil退出，那么说明没有环
	for head != nil {
		// 如果可以从哈希表中获取到，说明有环
		if _, ok := hashTable[head]; ok {
			return true
		}else { // 如果取不到，将节点添加到哈希表中
			hashTable[head] = struct{}{}
		}


		head = head.Next

	}

	return false
}

func main() {
	node4 := &ListNode{Val: -4, Next: nil}
	node2 := &ListNode{Val: 2, Next: &ListNode{Val: 0, Next: node4}}
	node4.Next = node2
	node := &ListNode{Val: 3, Next: node2}

	ret := hasCycle(node)
	fmt.Println(ret)

}

你可以假设数组是非空的，并且给定的数组总是存在多数元素。

 
示例 1：
输入：[3,2,3]
输出：3

示例 2：

    # (一)：哈希表

    • 使用哈希表的过程中，可以一般添加数值到哈希表中，一般统计最大的次数。数据复杂度O(n)，空间复杂度O(n)
    func majorityElement(nums []int) int {

	// 第一个元素用于存放数字，第二个元素用于存放次数
	bigNums := make([]int, 2)
	hashTable := map[int]int{}

	if len(nums) == 1 {
		return nums[0]
	}

	for i:=0; i<len(nums); i++ {
		if numses, ok := hashTable[nums[i]]; ok {
			if numses + 1 > bigNums[1] {
				bigNums[0] = nums[i]
				bigNums[1] = numses + 1
			}
			hashTable[nums[i]]++
		}else {
			hashTable[nums[i]]++
		}
	}

	return bigNums[0]
}

func main() {

	ret := majorityElement([]int{8,8,7,7,7})
	fmt.Println(ret)

}

    # 五：除自身以外数组的乘积

    给你一个整数数组 nums，返回 数组 answer ，其中 answer[i] 等于 nums 中除 nums[i] 之外其余各元素的乘积 。

题目数据 保证 数组 nums之中任意元素的全部前缀元素和后缀的乘积都在  32 位 整数范围内。

请不要使用除法，且在 O(n) 时间复杂度内完成此题。


示例 1:
输入: nums = [1,2,3,4]
输出: [24,12,8,6]

示例 2:
输入: nums = [-1,1,0,-3,3]
输出: [0,0,9,0,0]

    # (一)：事先算好左右乘积

    • 在枚举算法中，是临时计算的左右乘积，那么如果要将时间复杂度降低到O(n)，那么需要事先将左右乘积算好，然后存储在列表中，需要的时候直接获取就好了，也就是通过：``answer[i] = L[i] * R[i]`给新数组赋值
    func productExceptSelf(nums []int) []int {
	length := len(nums)

	// L 和 R 分别表示左右两侧的乘积列表
	L, R, answer := make([]int, length), make([]int, length), make([]int, length)

	// L[i] 为索引 i 左侧所有元素的乘积
	// 对于索引为 '0' 的元素，因为左侧没有元素，所以 L[0] = 1
	L[0] = 1
	for i := 1; i < length; i++ {
		L[i] = nums[i-1] * L[i-1]
	}

	// R[i] 为索引 i 右侧所有元素的乘积
	// 对于索引为 'length-1' 的元素，因为右侧没有元素，所以 R[length-1] = 1
	R[length-1] = 1
	for i := length - 2; i >= 0; i-- {
		R[i] = nums[i+1] * R[i+1]
	}

	// 这里是核心
	for i := 0; i < length; i++ {
		answer[i] = L[i] * R[i]
	}

	return answer
}


func main() {
	ret := productExceptSelf([]int{1,2,3,4})
	fmt.Println(ret)
}
