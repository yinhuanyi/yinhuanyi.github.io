    # (一)：常见的平衡二叉树

    • 满二叉树是平衡二叉树

    • 完全二叉树不一定是平衡二叉树，因为平衡二叉树需要有序。平衡二叉树也不一定是完全二叉树。下图是二叉堆，二叉堆有序，因此是平衡二叉树

    • 线段树树是平衡二叉树

    # (二)：AVL树是平衡二叉树

    • 平衡二叉树定义
    对于任意一个节点，左子树和右子树的高度差不能超过1

    • 平衡二叉树的高度h和节点数量的关系也是O(logn)

    # (四)：二叉树节点的平衡因子计算

    • 平衡二叉树的判定标准：每个节点的左右子树高度差的绝对值小于等于1，每个节点的左右子树高度差的绝对值为节点的平衡因子

    • 各个节点的平衡因子计算方式为：左节点高度-右节点高度的绝对值。例如节点【2】的平衡因子为0，节点【4】的平衡因子为1。节点的平衡因子在图中使用蓝色数字表示

	Add(k string, v int)
	Remove (k string) interface{}
	Contains(k string) bool
	Get(k string) interface{}
	Set(k string, v int)
	GetSize() int
	IsEmpty() bool
}

type Node struct {
	k string
	v int
	height int // 每个节点添加一个高度
	left *Node
	right *Node
}

func NewNode(k string, v int) *Node {
	return &Node{
		k: k,
		v: v,
		height: 1,
		left: nil,
		right: nil,
	}
}

type AVLTree struct {
	root *Node
	size int
}

func NewAVLTree() IAVLTree {
	return &AVLTree{
		root: nil,
		size: 0,
	}
}

func (b *AVLTree) GetSize() int {
	return b.size
}

func (b *AVLTree) IsEmpty() bool {
	return b.size == 0
}

func (b *AVLTree) max(item1, item2 int) int {
	if item1 > item2 {
		return item1
	}
	return item2
}

// add 添加元素返回根节点
func (b *AVLTree) add (node *Node, k string, v int) *Node {

	if node == nil {
		b.size++
		return NewNode(k, v)
	}

	if k < node.k {
		node.left = b.add(node.left, k, v)

	}else if k > node.k {
		node.right = b.add(node.right, k, v)

	}else {
		node.v = v
	}
	
	// 添加了元素，需要重新计算一下节点的高度
	node.height = 1 + b.max(b.getHeight(node.left), b.getHeight(node.right))

	// 添加元素，需要重新计算一下节点的平衡因子
	balanceFactor := b.getBalanceFactor(node)
	if balanceFactor > 1 {
		fmt.Printf("当前树不是平衡树，节点的平衡因子为：%d\n", balanceFactor)
	}

	return node

}

func (b *AVLTree) Add(k string, v int) {

	b.root = b.add(b.root, k, v)

}

// GetNode 辅助方法
func (b *AVLTree) getNode(node *Node, k string) *Node {

	if node == nil {
		return nil
	}

	if node.k == k {
		return node

	}else if k < node.k {

		return b.getNode(node.left, k)

	}else {
		return b.getNode(node.right, k)

	}

}

func (b *AVLTree) contains (node *Node, k string) bool {
	return b.getNode(node, k) != nil

}

func (b *AVLTree) Contains(k string) bool {
	return b.contains(b.root, k)
}

func (b *AVLTree) Get(k string) interface{} {

	node := b.getNode(b.root, k)
	if node == nil {
		return nil
	}else {
		return node.v
	}

}

func (b *AVLTree) Set(k string, v int) {
	node := b.getNode(b.root, k)
	if node == nil {
		panic(fmt.Sprintf("%s 不存在", k))
	}else {
		node.v = v
	}
}

// minimum 获取右子树的的最小节点
func (b *AVLTree) minimum(node *Node) *Node {
	if node.left == nil {
		return node
	}
	return b.minimum(node.left)
}

// removeMin 删除右子树的最小节点，也就是删除后继节点
func (b *AVLTree) removeMin(node *Node) *Node {
	if node.left == nil {
		rightNode := node.right
		node.right = nil
		b.size--
		return rightNode
	}

	node.left = b.removeMin(node.left)
	return node
}

// remove 递归删除
func (b *AVLTree) remove(node *Node, k string) *Node {
	if node == nil {
		return nil
	}

	if k < node.k {
		node.left = b.remove(node.left, k)
		return node

	}else if k > node.k {
		node.right = b.remove(node.right, k)
		return node

	}else {

		if node.left == nil {
			rightNode := node.right
			node.right = nil
			b.size--
			return rightNode
		}

		if node.right == nil {
			leftNode := node.left
			node.left = nil
			b.size--
			return leftNode
		}

		successor := b.minimum(node.right)
		successor.right = b.removeMin(node.right)
		successor.left = node.left
		b.size++
		node.left, node.right = nil, nil
		b.size--

		return successor

	}
}

func (b *AVLTree) Remove(k string) interface{} {

	node := b.getNode(b.root, k)

	if node != nil {
		b.root = b.remove(b.root, k)
		return node.v
	}

	return nil
}

// getHeight 获取节点的高度
func (b *AVLTree) getHeight(node *Node) int {
	if node == nil { return 0}
	return node.height
}

func (b *AVLTree) intAbs (n int) int {
	if n < 0 {
		return -n
	}
	return n
}

// getBalanceFactor 计算每个节点的平衡因子
func (b *AVLTree) getBalanceFactor (node *Node) int {
	if node == nil { return 0 }

	return b.intAbs(b.getHeight(node.left) - b.getHeight(node.right))
}

// add 添加元素返回根节点
func (b *MapTree) add (node *Node, k string, v int) *Node {

	if node == nil {
		b.size++
		return NewNode(k, v)
	}

	if k < node.k {
		node.left = b.add(node.left, k, v)

	}else if k > node.k {
		node.right = b.add(node.right, k, v)

	}else {
		node.v = v
	}
	
	// 添加了元素，需要重新计算一下节点的高度
	node.height = 1 + b.max(b.getHeight(node.left), b.getHeight(node.right))

	// 添加元素，需要重新计算一下节点的平衡因子
	balanceFactor := b.getBalanceFactor(node)
	if b.intAbs(balanceFactor) > 1 {
		fmt.Printf("当前树不是平衡树，节点的平衡因子为：%d\n", balanceFactor)
	}

	// 添加了元素，需要对二叉树进行平衡维护
	if balanceFactor > 1 && b.getBalanceFactor(node.left) >= 0 {  // 不平衡出现在节点node左边，使用右旋转, LL类型
		return b.rightRotate(node)

	}else if balanceFactor < -1 && b.getBalanceFactor(node.right) <= 0 { // 不平衡出现在节点node右边，使用左旋转 RR类型
		return b.leftRotate(node)

	}else if balanceFactor > 1 &&  b.getBalanceFactor(node.left) <= 0 { // LR类型
		node.left = b.leftRotate(node.left)
		return b.rightRotate(node)

	} else if balanceFactor < -1 && b.getBalanceFactor(node.right) > 0 { // RL类型
		node.right = b.rightRotate(node.right)
		return b.leftRotate(node)

	}

	return node

}

	node.left = b.removeMin(node.left)

// remove 递归删除
func (b *MapTree) remove(node *Node, k string) *Node {
	if node == nil {
		return nil
	}

	if k < node.k {
		node.left = b.remove(node.left, k)
		return node

	}else if k > node.k {
		node.right = b.remove(node.right, k)
		return node

	}else {

		if node.left == nil {
			rightNode := node.right
			node.right = nil
			b.size--
			return rightNode
		}

		if node.right == nil {
			leftNode := node.left
			node.left = nil
			b.size--
			return leftNode
		}

		successor := b.minimum(node.right)
		successor.right = b.removeMin(node.right)
		successor.left = node.left
		b.size++
		node.left, node.right = nil, nil
		b.size--

		return successor

	}
}

	return x

}

// add 添加元素返回根节点
func (b *MapTree) add (node *Node, k string, v int) *Node {

	if node == nil {
		b.size++
		return NewNode(k, v)
	}

	if k < node.k {
		node.left = b.add(node.left, k, v)

	}else if k > node.k {
		node.right = b.add(node.right, k, v)

	}else {
		node.v = v
	}
	
	// 添加了元素，需要重新计算一下节点的高度
	node.height = 1 + b.max(b.getHeight(node.left), b.getHeight(node.right))

	// 添加元素，需要重新计算一下节点的平衡因子
	balanceFactor := b.getBalanceFactor(node)
	if b.intAbs(balanceFactor) > 1 {
		fmt.Printf("当前树不是平衡树，节点的平衡因子为：%d\n", balanceFactor)
	}

	// 添加了元素，需要对二叉树进行平衡维护
	if balanceFactor > 1 && b.getBalanceFactor(node.left) >= 0 {  // 不平衡出现在节点node左边，使用右旋转, LL类型
		return b.rightRotate(node)

	}else if balanceFactor < -1 && b.getBalanceFactor(node.right) <= 0 { // 不平衡出现在节点node右边，使用左旋转 RR类型
		return b.leftRotate(node)

	}else if balanceFactor > 1 &&  b.getBalanceFactor(node.left) <= 0 { // LR类型
		node.left = b.leftRotate(node.left)
		return b.rightRotate(node)

	} else if balanceFactor < -1 && b.getBalanceFactor(node.right) > 0 { // RL类型
		node.right = b.rightRotate(node.right)
		return b.leftRotate(node)

	}

	return node

}

// IsBST 判断当前二叉树是否是二分搜索树，对二叉树进行中序遍历，二分搜索树的中序遍历节点是顺序排列的
func (b *MapTree) IsBST() bool {
	keys := make([]string, 0)

	// 在中序遍历过程中，将key加入到keys数组中
	b.inOrder(b.root, keys)

	for i:=1; i<len(keys); i++ {
		if keys[i] < keys[i-1] {
			return false
		}
	}

	return true
}

func (b *MapTree) isBalancer(node *Node) bool {
	if node == nil {
		return true
	}

	balanceFactor := b.intAbs(b.getBalanceFactor(node))
	if balanceFactor > 1 {
		return false
	}
	return b.isBalancer(node.left) && b.isBalancer(node.right)
}

// IsBalancer 判断当前二叉树是否是平衡二叉树，只需要判断左右子树的高度差不能超过1

	return b.isBalancer(b.root)
}

// rightRotate 对节点y进行右旋转
func (b *MapTree) rightRotate (y *Node) *Node {

	// y节点右旋转处理
	x := y.left
	T3 := x.right

	x.right = y
	y.left = T3

	// 更新x,y的高度
	y.height = b.max(b.getHeight(y.left), b.getHeight(y.right)) + 1
	x.height = b.max(b.getHeight(x.left), b.getHeight(x.right)) + 1

	// 返回节点x
	return x

}

// leftRotate 对节点y进行左旋转
func (b *MapTree) leftRotate(y *Node) *Node {
	x := y.right

	x.left = y
	y.right = T3

	y.height = b.max(b.getHeight(y.left), b.getHeight(y.right)) + 1
	x.height = b.max(b.getHeight(x.left), b.getHeight(x.right)) + 1

	return x
}

    # (三)：为什么2-3树是一棵绝对平衡的树？

    • 如果有一个节点【42】，当添加一个元素【37】的时候，由于2-3树添加节点无法添加到空的位置，那么会添加到最后的叶子节点上，【37】节点和【42】节点就融合为一个节点

    • 如果在添加一个【12】节点，此时【12】节点小于【37】，但是【37】节点的叶子节点为空，因此【12】节点、【37】节点、【42】节点融合在一起

    • 对于2-3树来说，不能有三个节点，那么【12】【37】【42】这三个节点就必须分裂成一颗子树

    • 如果再添加一个【18】节点，按照二分搜索树的性质，应该添加到【12】节点的右节点上，但是此时【12】节点的右子树为空，那么就与【12】节点融合

    • 如果再添加一个【6】节点，按照二分搜索树的性质，应该添加到【12】节点的左节点，但是【12】节点的左节点为空，因此与【12】节点融合

    • 按照2-3树的性质，最多只能有两个节点融合在一起，那么【6】【12】【18】三个节点就必须分裂

    • 按照2-3树的完全平衡的性质，【12】节点需要与父节点【37】节点融合

    • 如果再添加一个【11】节点，那么需要与【6】节点融合

    • 如果再添加一个【5】节点，那么需要与【6】节点融合，按照2-3树的性质，最多只能有两个节点融合在一起，那么【6】【12】【18】三个节点就必须分裂

    • 为了保证2-3树的绝对平衡性质，【6】节点需要与父节点【12】【37】融合。【6】【12】【37】是三个节点，因此需要分裂

type IBSTTree interface {
	IsBST() bool
}

type IMapTree interface {
	IBSTTree
	GetSize() int
	IsEmpty() bool
	IsRed(node *Node) bool
	Add(k string, v int)
	Contains(k string) bool
	Get(k string) interface{}
	Set(k string, v int)

}

type Node struct {
	k string
	v int
	//height int // 每个节点添加一个高度
	color bool   // 用于标识节点是红色还是黑色
	left *Node
	right *Node
}

func NewNode(k string, v int) *Node {
	return &Node{
		k: k,
		v: v,
		//height: 1,
		color: RED,  // 默认新节点为红色
		left: nil,
		right: nil,
	}
}

type BRMapTree struct {
	root *Node
	size int
}

func NewBRMapTree() IMapTree {
	return &BRMapTree{
		root: nil,
		size: 0,
	}
}

func (b *BRMapTree) GetSize() int {
	return b.size
}

func (b *BRMapTree) IsEmpty() bool {
	return b.size == 0
}

func (b *BRMapTree) IsRed (node *Node) bool {
	if node == nil {
		return BLACK
	}
	return node.color
}

// GetNode 辅助方法
func (b *BRMapTree) getNode(node *Node, k string) *Node {

	if node == nil {
		return nil
	}

	if node.k == k {
		return node

	}else if k < node.k {

		return b.getNode(node.left, k)

	}else {
		return b.getNode(node.right, k)

	}

}

func (b *BRMapTree) contains (node *Node, k string) bool {
	return b.getNode(node, k) != nil
}

func (b *BRMapTree) Contains(k string) bool {
	return b.contains(b.root, k)
}

func (b *BRMapTree) Get(k string) interface{} {

	node := b.getNode(b.root, k)
	if node == nil {
		return nil
	}else {
		return node.v
	}

}

func (b *BRMapTree) Set(k string, v int) {
	node := b.getNode(b.root, k)
	if node == nil {
		panic(fmt.Sprintf("%s 不存在", k))
	}else {
		node.v = v
	}
}

// leftRotate 左旋转
func (b *BRMapTree) leftRotate (node *Node) *Node {
	// 获取元素
	x := node.right

	// 旋转
	node.right = x.left
	x.left = node

	// 设置颜色
	x.color = node.color
	node.color = RED

	return x
}

// rightRotate 右旋转
func (b *BRMapTree) rightRotate(node *Node) *Node {
	// 获取元素
	x := node.left

	// 右旋转
	node.left = x.right
	x.right = node

	// 设置颜色
	x.color = node.color
	node.color = RED

	return x

}

// flipColors 颜色的翻转
func (b *BRMapTree) flipColors (node *Node) {
	node.color = RED
	node.left.color, node.right.color = BLACK, BLACK
}

func (b *BRMapTree) add (node *Node, k string, v int) *Node {

	if node == nil {
		b.size++
		return NewNode(k, v)   // 默认创建的是红色节点
	}

	if k < node.k {
		node.left = b.add(node.left, k, v)

	}else if k > node.k {
		node.right = b.add(node.right, k, v)

	}else {
		node.v = v
	}

	// 这三个条件是并列的
	if b.IsRed(node.right) && !b.IsRed(node.left) { // 右节点是红色，左节点不是红色
		node = b.leftRotate(node)
	}

	if b.IsRed(node.left) && b.IsRed(node.left.left) { // 左节点是红色，左节点的左节点也是红色
		node = b.rightRotate(node)
	}

	if b.IsRed(node.left) && b.IsRed(node.right) {  // 左节点是红色，右节点也是红色
		b.flipColors(node)
	}

	return node

}

func (b *BRMapTree) Add(k string, v int) {
	b.root = b.add(b.root, k, v)
	b.root.color = BLACK // 保证根节点是黑色的
}

// inOrder 对二叉树进行中序遍历
func (b *BRMapTree) inOrder (node *Node, arr []string) {
	if node == nil { return }
	b.inOrder(node.left, arr)
	arr = append(arr, node.k)
	b.inOrder(node.right, arr)
}

// IsBST 判断当前二叉树是否是二分搜索树，对二叉树进行中序遍历，二分搜索树的中序遍历节点是顺序排列的
func (b *BRMapTree) IsBST() bool {
	keys := make([]string, 0)

	// 在中序遍历过程中，将key加入到keys数组中
	b.inOrder(b.root, keys)

	for i:=1; i<len(keys); i++ {
		if keys[i] < keys[i-1] {
			return false
		}
	}

	return true
}
