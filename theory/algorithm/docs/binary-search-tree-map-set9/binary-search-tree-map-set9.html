    // Node 节点
type Node struct {
	e int
	left *Node
	right *Node
}

// NewNode 节点构造函数
func NewNode(e int) *Node {
	return &Node{
		e: e,
		left: nil,
		right: nil,
	}
}

// BST 二分搜索树
type BST struct {
	root *Node  // 根节点
	size int    // 树的大小
}

// NewBST 二分搜索树构造函数
func NewBST() *BST {
	return &BST{
		root: nil,
		size: 0,
	}
}

// size 获取树的大小
func (b *BST) bstSize () int {
	return b.size
}

// isEmpty 是否为空
func (b *BST) isEmpty() bool {
	return b.size == 0
}

// add 递归调用，插入新节点后，返回子树的根节点
// 这里的目的是创建一个子树
func (b *BST) add (node *Node, e int) *Node {

	// 判断节点是否为空
	if node == nil {
		b.size++
		return NewNode(e)
	}

	// 如果小于，添加到左边
	if e < node.e {
		node.left = b.add(node.left, e)

		// 如果大于，添加到右边
	}else if e > node.e {
		node.right = b.add(node.right, e)
	}

	return node

}
func (b *BST) Add(e int) {

	// 将新的子树根节点挂载到树上
	b.root = b.add(b.root, e)

}

// contains 在某个子树中递归
func (b *BST) contains (node *Node, e int) bool {

	if node == nil {
		return false
	}

	if e == node.e {
		return true

	}else if e < node.e {
		return b.contains(node.left, e)

	}else {
		return b.contains(node.right, e)

	}

}
func (b *BST) Contains (e int) bool {
	return b.contains(b.root, e)

}


func (b *BST) preOrder (node *Node) {

	if node == nil {
		return
	}

	fmt.Printf("%d  ", node.e)

	// 迭代
	b.preOrder(node.left)
	b.preOrder(node.right)

}
// PreOrder 前序遍历
func (b *BST) PreOrder () {
	b.preOrder(b.root)
}

func (b *BST) inOrder (node *Node) {

	if node == nil {
		return
	}

	b.inOrder(node.left)
	fmt.Printf("%d  ", node.e)
	b.inOrder(node.right)

}
// InOrder 中序遍历
func (b *BST) InOrder () {
	b.inOrder(b.root)
}

func (b *BST) postOrder (node *Node) {
	if node == nil {
		return
	}

	b.postOrder(node.left)
	b.postOrder(node.right)
	fmt.Printf("%d  ", node.e)

}
// PostOrder 后续遍历
func (b *BST) PostOrder () {
	b.postOrder(b.root)
}

    • 分别对比Set的add、contains、remove操作消耗的时间

	链表的CRUD操作时间复杂度是O(n)

    • 一个h层的满二叉树一共有多少节点
    2⁰ + 2¹ + 2² + 2³ + ... + 2^(h-1) = 2^h - 1 = n

==>  h = log₂(n+1) ==> O(h) ~ O(logn)

    type LinkListMap interface {
	Add(k string, v int)
	Remove (k string) interface{}
	Contains(k string) bool
	Get(k string) interface{}
	Set(k string, v int)
	GetSize() int
	IsEmpty() bool
}

// Node 节点
type Node struct {
	k    string
	v    int
	next *Node
}

// NodeToString 打印节点k/v值
func (n *Node) NodeToString() string {
	return n.k + ":" + strconv.Itoa(n.v)
}

// LinkedList 链表
type LinkedList struct {
	dummyHead *Node
	size int
}

// NewLinkedList 构造函数
func NewLinkedList() LinkListMap {
	return &LinkedList{
		dummyHead: new(Node),
		size: 0,
	}
}

// getNode 辅助函数，根据key的值，遍历链表，找到与key相等的节点，然后返回
func (l *LinkedList) getNode (k string) *Node {

	cur := l.dummyHead.next

	for cur != nil {
		if cur.k == k {
			return cur
		}
		cur = cur.next
	}

	return  nil

}

func (l *LinkedList) GetSize() int {
	return l.size
}

func (l *LinkedList) IsEmpty() bool {
	return l.size == 0
}

func (l *LinkedList) Contains(k string) bool {
	return l.getNode(k) != nil
}

func (l *LinkedList) Get(k string) interface{} {
	node := l.getNode(k)
	if node == nil {
		return nil
	}else {
		return node.v
	}
}

func (l *LinkedList) Add(k string, v int) {
	node := l.getNode(k)
	if node == nil { // 如果k不存在，说明可用添加
		 l.dummyHead.next = &Node{k, v, l.dummyHead.next}
		 l.size++
	}else {
		node.v = v  // 如果k存在，直接更新节点的value
	}
}

func (l *LinkedList) Set(k string, v int) {

	node := l.getNode(k)
	if node == nil {
		panic(k + "不存在")
	}

	node.v = v
}

func (l *LinkedList) Remove(k string) interface{} {

	prev := l.dummyHead

	for prev.next != nil {
		if prev.next.k == k {
			break
		}
		prev = prev.next
	}

	if prev.next != nil {
		delNode := prev.next
		prev.next = delNode.next
		delNode.next = nil
		l.size--
		return delNode.v
	}

	return nil

}

    type BSTMap interface {
	Add(k string, v int)
	Remove (k string) interface{}
	Contains(k string) bool
	Get(k string) interface{}
	Set(k string, v int)
	GetSize() int
	IsEmpty() bool
}

type Node struct {
	k string
	v int
	left *Node
	right *Node
}

func NewNode(k string, v int) *Node {
	return &Node{
		k: k,
		v: v,
		left: nil,
		right: nil,
	}
}

type BST struct {
	root *Node
	size int
}

func NewBSTMap() BSTMap {
	return &BST{
		root: nil,
		size: 0,
	}
}

func (b *BST) GetSize() int {
	return b.size
}

func (b *BST) IsEmpty() bool {
	return b.size == 0
}

// add 添加元素返回根节点
func (b *BST) add (node *Node, k string, v int) *Node {

	// 判断节点是否为空
	if node == nil {
		b.size++
		return NewNode(k, v)
	}

	// 如果小于，添加到左边
	if k < node.k {
		node.left = b.add(node.left, k, v)

		// 如果大于，添加到右边
	}else if k > node.k {
		node.right = b.add(node.right, k, v)

	}else {
		node.v = v

	}

	return node

}

func (b *BST) Add(k string, v int) {

	// 将新的子树根节点挂载到树上
	b.root = b.add(b.root, k, v)

}

// GetNode 辅助方法
func (b *BST) getNode(node *Node, k string) *Node {

	if node == nil {
		return nil
	}

	// 递归的终止条件
	if node.k == k {
		return node

	}else if k < node.k {
		// 将问题转为为更小的问题
		return b.getNode(node.left, k)

	}else {
		return b.getNode(node.right, k)

	}

}

func (b *BST) contains (node *Node, k string) bool {
	return b.getNode(node, k) != nil

}

func (b *BST) Contains(k string) bool {
	return b.contains(b.root, k)
}

func (b *BST) Get(k string) interface{} {

	node := b.getNode(b.root, k)
	if node == nil {
		return nil
	}else {
		return node.v
	}

}

func (b *BST) Set(k string, v int) {
	node := b.getNode(b.root, k)
	if node == nil {
		panic(fmt.Sprintf("%s 不存在", k))
	}else {
		node.v = v
	}
}

// minimum 获取右子树的的最小节点
func (b *BST) minimum(node *Node) *Node {
	if node.left == nil {
		return node
	}
	return b.minimum(node.left)
}

// removeMin 删除右子树的最小节点，也就是删除后继节点
func (b *BST) removeMin(node *Node) *Node {
	if node.left == nil {
		rightNode := node.right
		node.right = nil
		b.size--
		return rightNode
	}

	node.left = b.removeMin(node.left)
	return node
}

// remove 递归删除
func (b *BST) remove(node *Node, k string) *Node {
	if node == nil {
		return nil
	}

	if k < node.k {
		node.left = b.remove(node.left, k)
		return node

	}else if k > node.k {
		node.right = b.remove(node.right, k)
		return node

	}else {

		if node.left == nil {
			rightNode := node.right
			node.right = nil
			b.size--
			return rightNode
		}

		if node.right == nil {
			leftNode := node.left
			node.left = nil
			b.size--
			return leftNode
		}

		successor := b.minimum(node.right)  // 获取后继节点
		successor.right = b.removeMin(node.right) // 删除二分搜索树中的后继节点，将右节点赋值给后继节点的右节点
		successor.left = node.left // 将当前节点的左节点赋值给【s】的左节点
		b.size++
		node.left, node.right = nil, nil
		b.size--

		// 这里也是迭代终止条件，返回的是后继节点
		return successor

	}
}

// Remove 删除节点，返回节点的key对应的value值
func (b *BST) Remove(k string) interface{} {

	// 先获取一下k对应的节点
	node := b.getNode(b.root, k)

	// 带删除的节点不为空，那么从根节点开始，递归二分搜索树，删除k所在的节点，最后将新的root根节点返回
	if node != nil {
		b.root = b.remove(b.root, k)
		return node.v
	}

	// 如果带删除的节点不存在，返回nil
	return nil
}

func main() {

	m := map[string]int{"a": 1, "t": 23, "c": 3, "d": 4, "e": 5, "f": 6, "g": 7}
	bstMap := NewBSTMap()

	for k, v := range m {
		bstMap.Add(k, v)
	}

	isContains := bstMap.Contains("d")
	fmt.Printf("map中是否存在key为f的元素：%t\n", isContains)

	value := bstMap.Remove("d")
	fmt.Printf("删除的d节点的value为：%d\n", value)

	isContains = bstMap.Contains("d")
	fmt.Printf("map中是否存在key为f的元素：%t\n", isContains)

	bstMap.Set("t", 2)
	tValue := bstMap.Get("t")
	fmt.Printf("map中t元素的值：%d\n", tValue)
}
