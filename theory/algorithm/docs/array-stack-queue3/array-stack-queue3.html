    # (二)：实现动态数组(由于golang中数组无法使用变量指定长度，因此使用slice替代)

    • 在golang中，slice是对底层数组的抽象和控制，它是一个结构体
    type slice struct {
    array unsafe.Pointer  // 指向底层数组的指针，golang中没有操作原始内层的指针，unsafe包提供相关的对内存指针的操作
    len   int // 切片的长度
    cap   int // 切片的容量，底层数组的长度
}
    1
    2
    3
    4
    5

    • golang的slice在append的过程中，如果需要扩容，那么会返回一个新的引用地址，然后将该引用重新赋值给之前的切片变量。当slice的长度大于1024后，会进行1.25倍的扩容，缩容也是一样。在runtime包的slice.go文件中有对应的slice实现。由于golang无法申明var arr [n]int可变容量数组，只能var arr [4]int常量申明，因此下面会使用slice替代array，插入和删除操作使用了slice的append方法

    • 动态数组实现：代码中值得注意的是，删除和插入都会改变slice的len，因此代码中为了不改变len，删除的时候，使用了copy方法对切片进行拷贝，让slice的底层数组重新开辟内存空间，其实slice.go实现也是如此，在扩缩容的时候需要重新创建新的底层array数组

    /*
需要实现的功能
	①：初始化：
	②：append：插入单个元素
	③：appendMany：批量插入元素
	④：getElement：基于索引获取元素
	⑤：getLen：获取长度
	⑥：getCap：获取容量
	⑦：toString：打印
*/

package main

import (
	"fmt"
	"sync"
)

/*
需要实现的功能
	①：初始化：
	②：append：插入单个元素
	③：appendMany：批量插入元素
	④：getElement：基于索引获取元素
	⑤：getLen：获取长度
	⑥：getCap：获取容量
	⑦：toString：打印
*/

type Slice struct {
	array []int
	len int
	cap int
	lock sync.Mutex // 并发下，保证线程安全
}

// NewSlice 构造函数
func NewSlice(len, cap int) *Slice {
	if len > cap {
		panic("len must larger than cap")
	}

	return &Slice{
		array: make([]int, cap, cap),  // 这里使用golang的slice作为数组，仅仅只是使用到了使用变量指定数组容量的功能
		len: 0,
		cap: cap,
	}
}

// Append 添加一个元素
func (s *Slice) Append (ele int) {

	s.lock.Lock()
	defer s.lock.Unlock()

	// 判断是否需要扩容
	if s.len == s.cap {

		newCap := 2 * s.len

		if s.cap == 0 {
			newCap = 1
		}

		newArray := make([]int, newCap, newCap)

		// 把老数据移动到新数组中
		for k,v := range s.array{
			newArray[k] = v
		}

		// 替换以前的数组变量
		s.array = newArray
		s.cap = newCap

	}

	// 向新数组中添加元素
	fmt.Printf("len=%d, cap=%d\n", len(s.array), cap(s.array))
	s.array[s.len] = ele
	s.len = s.len + 1
}

// AppendMany 添加多个元素
func (s *Slice) AppendMany (eles ...int)  {
	for _, v := range eles {
		s.Append(v)
	}
}

// Insert 插入元素, 这里需要使用到slice的功能了, 向中间插入一个元素
func (s *Slice) Insert (index, ele int) {

	s.lock.Lock()
	defer s.lock.Unlock()

	if index<0 || index>s.Len() {
		panic(" index invalid ")
	}

	// 扩容
	if s.len == s.cap {

		newCap := 2 * s.len

		if s.cap == 0 {
			newCap = 1
		}

		newArray := make([]int, newCap, newCap)

		// 把老数据移动到新数组中
		for k,v := range s.array{
			newArray[k] = v
		}

		// 替换以前的数组变量
		s.array = newArray
		s.cap = newCap

	}

	// 插入元素的时候，需要确保len不变
	// 插入元素
	if index == 0 { // 头部插入
		var header []int
		header = append(header, ele)
		s.array = append(header, s.array...)
	}else { // 中间插入
		front := s.array[:index]
		// 这里重新创建了一个切片，确保底层数组的数据不变
		newFront := make([]int, len(front)+1)
		copy(newFront, front)
		//newFront = append(newFront, ele)
		newFront[index] = ele
		s.array = append(newFront, s.array[index:]...)
	}

	s.len++
}

// Get 获取某个索引的元素
func (s *Slice) Get (index int) int {
	if s.len == 0 || index >= s.len {
		panic("index over len")
	}
	return s.array[index]
}

func (s *Slice) Contains(ele int) bool {
	for i:=0;i<s.len;i++{
		if s.Get(i) == ele {
			return true
		}
	}
	return false
}

// Set 基于索引修改元素
func (s *Slice) Set (index, ele int) {
	if s.len == 0 || index >= s.len {
		panic("index over len")
	}

	s.array[index] = ele
}

// Remove 删除元素
func (s *Slice) Remove(index int) {
	s.lock.Lock()
	defer s.lock.Unlock()

	if index<0 || index>s.Len() {
		panic(" index invalid ")
	}

	fmt.Printf("len=%d, cap=%d\n", len(s.array), cap(s.array))

	// 先删除, 删除元素应该保证的是len不变，如果使用切片操作，len会变, len变化了那么添加元素可能会出现【index out of range [4] with length 4】
	//front := s.array[:index]
	//s.array = append(front, s.array[index+1:]...)
	front := s.array[:index]
	tail := s.array[index:]
	newTail := make([]int, len(tail), len(tail))
	copy(newTail, tail[1:])
	s.array = append(front, newTail...)

	// 再考虑缩容, 如果到了4倍就缩容到2倍
	if float32(s.cap) / float32(s.len) > 4 {
		newCap := 2 * s.len
		newArray := make([]int, newCap, newCap)

		for k, v := range s.array {
			newArray[k] = v
		}

		s.array = newArray
		s.cap = newCap
	}

	s.len--
	fmt.Printf("len=%d, cap=%d\n", len(s.array), cap(s.array))

}

// Len 获取长度
func (s *Slice) Len() int {
	return s.len
}

// Cap 获取容量
func (s *Slice) Cap() int {
	return s.cap
}

// ToString 打印
func (s *Slice) ToString() (ret string) {

	for i:=0;i<s.Len();i++ {
		if i == 0 {
			ret = fmt.Sprintf("%s%d", "[", s.Get(i))
			continue
		}

		ret = fmt.Sprintf("%s %d", ret, s.Get(i))

	}

	ret = ret + "]"
	return

}



func main() {
	slice := NewSlice(2, 4)
	slice.Append(1)
	slice.AppendMany(2,3,4,5)
	ret := slice.ToString()
	fmt.Println(ret)
	slice.Insert(1, 100)
	ret = slice.ToString()
	fmt.Println(ret)
	fmt.Println(slice.len)
	slice.Set(4, 100)
	ret = slice.ToString()
	fmt.Println(ret)
	fmt.Println(slice.Contains(30))
	slice.Remove(1)
	ret = slice.ToString()
	fmt.Println(ret)
	slice.Remove(slice.len-2)
	slice.Remove(3)
	ret = slice.ToString()
	fmt.Println(ret)

	slice.AppendMany(1,2,3,4,5,6,7,8,9)
	ret = slice.ToString()
	fmt.Println(ret)
}

    # 二：栈和栈的应用

    # (一)：概念

    • 栈是一种先进后出的数据结构，是数组的子集

    package selfstack

// IStack 栈的接口
type IStack interface {
	GetSize() int
	IsEmpty() bool
	Push(e interface{})
	Pop() interface{}
	Peek() interface{}
}

type Stack struct {
	// 声明一个接口类型的切片
	Slice []interface{}
}

/*
实现接口定义的方法
*/

func (s *Stack) GetSize() int {
	return len(s.Slice)
}

func (s *Stack) IsEmpty() bool {
	return len(s.Slice) == 0
}

// Push 从最后一个元素添加
func (s *Stack) Push(e interface{}) {
	s.Slice = append(s.Slice, e)
}

// Pop 删除最后一个元素，且返回最后一个元素
func (s *Stack) Pop() interface{} {
	lastEle := s.Slice[len(s.Slice)-1]
	s.Slice = s.Slice[:len(s.Slice)-1]
	return lastEle
}

// Peek 获取最后一个元素
func (s *Stack) Peek() interface{} {
	return s.Slice[len(s.Slice)-1]
}

// NewStack 构造方法
func (s *Stack) NewStack(capacity int) Stack {
	stack := Stack{
		Slice: make([]interface{}, 0, capacity),
	}
	return stack
}

// NewEmptyStack 空stack
func NewEmptyStack() Stack {
	return Stack{
		Slice: make([]interface{}, 0, 0),
	}
}

// GetCapacity 获取stack的容量
func (s *Stack) GetCapacity() int {
	return cap(s.Slice)
}

// String 打印为字符串
func (s *Stack) String() interface{} {
	return s.Slice
}

    # (二)：做个题目：判断一个字符串的括号是否匹配，例如{()}是匹配的， {{{()]不是匹配的

    • 实现方法(这里使用到了自己实现的stack)
    import (
	"Algorithm/selfstack"
	"fmt"
)

func IsOk(str string) bool {

	// 这里很好的解释了，接口是验证结构体是否实现接口的方法，主要如果方法是指针接受者，那么结构体赋值给接口必须是地址
	stack := selfstack.NewEmptyStack()
	var _ selfstack.IStack = &stack

	for _, item := range str {
		char := fmt.Sprintf("%c", item)
		if char == "(" || char == "{" || char == "[" {
			stack.Push(char)

		} else {
			if stack.IsEmpty() {
				return false
			}

			topChar := stack.Pop()
			if char == ")" && topChar != "(" {
				return false
			}
			if char == "]" && topChar != "[" {
				return false
			}
			if char == "}" && topChar != "{" {
				return false
			}
		}
	}

	// 如果此时栈中还有元素，说明左括号多了，那么不匹配
	return stack.IsEmpty()

}

func main() {

	str := "{()}"
	ret := IsOk(str)
	fmt.Println(ret)

}

    package selfqueue

type IQueue interface {
	Enqueue(e interface{})
	Dequeue() interface{}
	GetFront() interface{}

}

type Queue struct {
	Slice []interface{}
}

// NewEmptyQueue 构造函数
func NewEmptyQueue() Queue {
	return Queue{
		Slice: make([]interface{}, 0, 0),
	}
}

// NewEmptyQueueWithCap 创建指定cap的队列
func NewEmptyQueueWithCap(cap int) Queue {
	return Queue{
		Slice: make([]interface{}, 0, cap),
	}
}

// Enqueue 添加元素
func (q *Queue) Enqueue(e interface{}) {
	q.Slice = append(q.Slice, e)
}

// Dequeue 删除元素, 从头部删除
func (q *Queue) Dequeue() interface{} {
	firstEle := q.Slice[0]
	q.Slice = q.Slice[1:]
	return firstEle
}

// GetFront 获取队列第一个元素
func (q *Queue) GetFront() interface{} {
	return q.Slice[0]
}

func (q *Queue) GetSize() int {
	return len(q.Slice)
}

func (q *Queue) IsEmpty() bool {
	return len(q.Slice) == 0
}

func (q *Queue) String() interface{} {
	return q.Slice
}

    # (二)：循环队列实现

    • 为什么需要循环队列
    之所以需要循环队列，是因为队列的底层是动态数组，删除头部元素是O(n)的时间复杂度，因为在动态数组头部删除一个元素，那么其他元素都将会向前移动，因此时间复杂度是O(n)，那么循环队列正好解决了这个问题。在循环队列中，删除头部元素，只需要front指针向后移动即可，时间复杂度为O(1)
    1

    • 循环队列的思想

    • 当循环列表是空时，front == tail

    • 向循环队列中添加元素，tail向后移动

    • 当删除一个元素，那么front向后移动

    • 当添元素，tail向后移动，如果元素超出了队列的容量，那么tail再从循环队列头部开始，此时循环队列没有满

    • 当再添加一个元素，此时循环队列就满了，满足 (tail + 1) / capacity == front

    • 循环队列实现代码
    package selflqueue

type Lqueue struct {
	Slice []interface{} // 这里认为slice是一个数组
	Front int           // 对首指针
	Tail  int           // 队尾指针
	Size  int           // 队列中元素个数
}

// NewLQueueWithLength 创建一个带容量的循环队列, 因为循环队列满的时候，会浪费一个元素空间，所以这里多申请一个内存空间
func NewLQueueWithLength(length int) *Lqueue {
	return &Lqueue{
		Slice: make([]interface{}, length+1),
		Front: 0,
		Tail:  0,
		Size:  0,
	}
}

// GetLength 获取队列容量
func (l *Lqueue) GetLength() int {
	return len(l.Slice)
}

// IsEmpty 判断循环队列是否为空
func (l *Lqueue) IsEmpty() bool {
	return l.Front == l.Tail
}

// GetSize 获取循环队列元素个数
func (l *Lqueue) GetSize() int {
	return l.Size
}

// resize 循环队列扩容，也就是扩容slice的len
func (l *Lqueue) resize(newLength int) {

	// 创建一个新的slice
	newSlice := make([]interface{}, newLength+1)
	// todo：将之前slice中的元素放进新slice中，但是这里有个注意点，现在是循环队列，新slice遍历可以从0开始，对于之前slice的遍历不是从i=0开始的
	for i := 0; i < l.Size; i++ { // l.Size是元素个数
		newSlice[i] = l.Slice[(l.Front+i)%len(l.Slice)]
	}

	l.Slice = newSlice
	l.Front = 0
	l.Tail = l.Size

}

// Enqueue 加入元素
func (l *Lqueue) Enqueue(e interface{}) {

	// 先判断是否循环队列已经满了，如果满了，先将slice扩容一倍
	if (l.Tail+1)%l.GetLength() == l.Front {
		l.resize(l.GetLength() * 2)
	}

	l.Slice[l.Tail] = e
	l.Tail = (l.Tail + 1) % l.GetLength()
	l.Size++

}

// Dequeue 删除元素, 并且将元素返回
func (l *Lqueue) Dequeue() interface{} {

	// 先判断是否为空
	if l.IsEmpty() {
		panic("Lqueue is empty now")
	}

	firstEle := l.Slice[l.Front]
	// 赋值为nil
	l.Slice[l.Front] = nil
	l.Front = (l.Front + 1) % l.GetLength()
	l.Size--

	// 判断是否需要缩容(元素个数小于长度的四分之一，且不能缩容至长度为0)
	if l.Size <= l.GetLength()/4 && l.GetLength()/2 != 0 {
		l.resize(l.GetLength() / 2)
	}

	return firstEle
}

// GetFront 获取第一个元素
func (l *Lqueue) GetFront() interface{} {
	// 先判断是否为空
	if l.IsEmpty() {
		panic("队列为空")
	}
	return l.Slice[l.Front]
}

func (l *Lqueue) String() interface{} {
	return l.Slice
}

    • 小结
    循环队列比普通队列的优势在于：
	当删除一个元素的时候，在动态数组中，后面的元素需要向前移动，时间复杂度为O(n)
	而循环队列设置了front和tail指针只需要将其赋值为nil即可，循环队列删除头部元素的时间复杂度为O(1)
