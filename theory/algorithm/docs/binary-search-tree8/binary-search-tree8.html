	return b.size == 0

// add 递归调用
func (b *BST) add (node *Node, e int) {

	// 基本问题
	if node.e == e {
		return

	}else if e < node.e && node.left == nil {
		node.left = NewNode(e)
		b.size++
		return

	}else if e > node.e && node.right == nil {
		node.right = NewNode(e)
		b.size++
		return

	}

	// 如果小于，添加到左边
	if e < node.e {
		b.add(node.left, e)

	// 如果大于，添加到右边
	}else {
		b.add(node.right, e)
	}

}

func (b *BST) Add(e int) {

	// 如果root为空
	if b.root == nil {
		b.root = NewNode(e)
		b.size ++
		return

	}else {
		// 将root节点作为第一个节点进行递归调用
		b.add(b.root, e)
	}

}

    • 最小值：在二分搜索树中，向左走，一直走到最左边走不动了就是最小值

    • 上面的这一颗二分搜索树向左走，走到了【16】就没有左叶子节点了，那么此时就走不动，那么最小值就是【16】。同理，向右走，【30】这个元素没有右子树了，走不动了，因此【30】就是最大值

    # (四)：删除任意节点

    • 删除情况分类
    1：删除最小值节点：一直向左走，当左节点为nil的时候，当前节点就是最小值节点，那么需要保存右边节点，让右边节点返回，覆盖当前的根节点即可

// 删除二分搜索树中值为e的节点，返回root根节点
// remove 方法中调用了removeMin方法，用来删除后继节点
func (b *BST) remove (node *Node, e int) *Node {

	// 递归终止条件
	if node == nil {
		return nil
	}

	// 如果e小于节点的值，说明待删除节点在node的左子树中，那么迭代左子树，删除节点，返回根节点
	if e < node.e {
		node.left = b.remove(node.left, e)
		return node

	// 如果e大于节点的值，说明待删除节点在node的右子树中，那么迭代右子树，删除节点，返回根节点
	}else if e > node.e {
		node.right = b.remove(node.right, e) // 第一次迭代
		return node

	// 如果e就是当前节点，说明需要删除当前节点
	}else {

		// 如果节点的左节点为空，返回右节点，让node脱离二分搜索树
		if node.left == nil {
			rightNode := node.right
			node.right = nil
			b.size--

			// 这里也是迭代终止条件，返回右节点
			return rightNode
		}

		// 如果节点的右节点为空，返回左节点，让node脱离二分搜索树
		if node.right == nil {
			leftNode := node.left
			node.left = nil
			b.size--

			// 这里也是迭代终止条件，返回左节点
			return leftNode
		}

		// 如果节点的左右节点都不为空，找到比待删除节点大的最小节点，也就是待删除节点右子树最小节点【s】,用【s】节点替代当前节点即可
		successor := b.minimum(node.right)        // 先找到右子树最小节点【s】
		successor.right = b.removeMin(node.right) // 删除掉右子树最小的节点，返回根节点赋值给【s】的右节点
		successor.left = node.left                // 将当前节点的左节点赋值给【s】的左节点
		b.size++								  // 这里相当于在二分搜索树中添加了一个【s】节点
		node.left, node.right = nil, nil          // 删除node节点的左右节点，让node节点与二分搜索树脱离关系，让node节点被GC回收
		b.size--								  // node节点脱离了，树的节点个数就需要减一

		// 这里也是迭代终止条件，返回的是后继节点
		return successor

	}

}

// Remove 删除值为e的节点
func (b *BST) Remove (e int) {
	b.root = b.remove(b.root, e)
}
