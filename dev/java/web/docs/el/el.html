    # 1: EL表达式

    • 基本语法
    基本语法：${[作用域对象].属性对象.[子属性]}


示例：<h1>学生姓名：${student.name}</h1>

    • 基于jsp表达式写一个请求
    1：创建Student类
public class Student {
    private String name;
    private String phone;

    public void setName(String name) {
        this.name = name;
    }

    public void setPhone(String phone) {
        this.phone = phone;
    }

    public String getName() {
        return name;
    }

    public String getPhone() {
        return phone;
    }
}

2：创建一个StudentServlet，接收用户的/info请求
public class StudentServlet extends HttpServlet {
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        // 创建对象
        Student student = new Student();
        student.setName("Robby");
        student.setPhone("18670236750");
        String grade = "A";

        // 给req添加属性对象
        req.setAttribute("student", student);
        req.setAttribute("grade", grade);

        // 将请求转发给jsp页面， 用于展现学生的基本信息
        req.getRequestDispatcher("/student_info.jsp").forward(req, resp);
    }
}

3：创建student_info.jsp文件，用户展示用户信息

        out.println("<h2>手机：" + student.getPhone() + "</h2>");

    # 5: JSTL标签库安装

    • JSTL也是用于简化JSP代码开发的，首先需要下载Tomcat的JSTL相关的jar包

    • 相关jar包介绍

    • 首先导入两个jar包
    1: 在WEB-INF目录中创建lib目录，从tomcat官网下载`taglibs-standard-impl-1.2.5.jar , taglibs-standard-spec-1.2.5.jar`，且拷贝在lib目录中

2：点击jar包右键，Add as Library

3: 打开Project Structure, 选择Modules，选择tomcatapp项目，选择依赖，将这两个jar包的Export栏打钩，选择应用后点击OK

4：在pom文件中添加依赖，放置reimport的时候，丢失第三方依赖包
    <dependency>
      <groupId>taglibs-standard-impl</groupId>
      <artifactId>taglibs-standard-impl</artifactId>
      <version>1.0</version>
      <scope>system</scope>
      <systemPath>${project.basedir}/src/main/webapp/WEB-INF/lib/taglibs-standard-impl-1.2.5.jar</systemPath>
    </dependency>

    <dependency>
      <groupId>taglibs-standard-spec</groupId>
      <artifactId>taglibs-standard-spec</artifactId>
      <version>1.0</version>
      <scope>system</scope>
      <systemPath>${project.basedir}/src/main/webapp/WEB-INF/lib/taglibs-standard-spec-1.2.5.jar</systemPath>
    </dependency>

5：再次打开Project Structure，选择Libraries，看看是否jar包已经导入项目

    • 创建core.jsp作为jsp页面

<%--忽略web.xml中设置的ELIgnore--%>
<%@page isELIgnored="false"%>
<%--引入jstl标签库--%>
<%@ taglib  uri ="http://java.sun.com/jsp/jstl/core" prefix="c" %>

<html>
<head>
    <title>Title</title>
</head>
<body>
    <h1>
        成绩： ${score}
    </h1>

    <%--基于c标签进行判断--%>
    <c:if test="${score >=60}">
        <h2 style="color: green;">成绩通过测试</h2>
    </c:if>

    <c:if test="${score <60}">
        <h2 style="color: red;">成绩不合格</h2>
    </c:if>

    <%--多分支判断，类似于switch--%>
    <h1>
        等级：${grade}
    </h1>
    <c:choose>
        <c:when test="${grade == 'A'}">
            <h2>优秀</h2>
        </c:when>
        <c:when test="${grade == 'B'}">
            <h2>良好</h2>
        </c:when>
        <c:when test="${grade == 'C'}">
            <h2>及格</h2>
        </c:when>
        <c:when test="${grade == 'D'}">
            <h2>不及格</h2>
        </c:when>
        <c:otherwise>
            <h2>继续加油</h2>
        </c:otherwise>
    </c:choose>

</body>
</html>

    <hr>

</body>
</html>

        this.department = department;
        this.job = job;
        this.salary = salary;

    public ListServlet() {
    }

        // 获取context
        ServletContext servletContext = req.getServletContext();

        // 获取数据, 例如进行CRUD操作
        if (servletContext.getAttribute("employees") == null){

    package employee;

import javax.servlet.ServletContext;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.util.List;

/**
 * Description:
 * Author:         Robby
 * CreateDate:     2019-11-04 23:34
 * Version:        1.0.0
 */

public class CreateServlet extends HttpServlet {
    public CreateServlet() {
    }

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        req.setCharacterEncoding("UTF-8");
        String num = req.getParameter("empno");
        String name = req.getParameter("ename");
        String department = req.getParameter("department");
        String job = req.getParameter("job");
        String salary = req.getParameter("salary");

        /***
         * 类型转换函数
         * (1)：其他数据类型转String类型
         * String s = String.valueOf( value);
         * (2)：String类型转其他类型
         * String s = "169";
         * byte b = Byte.parseByte( s );
         * short t = Short.parseShort( s );
         * int i = Integer.parseInt( s );
         * long l = Long.parseLong( s );
         * Float f = Float.parseFloat( s );
         * Double d = Double.parseDouble( s );
         */


        // 注意进行类型转换
        Employee employee = new Employee(Integer.parseInt(num), name, department, job, Float.parseFloat(salary));
        // 获取上下文
        ServletContext servletContext = req.getServletContext();
        // 从上下文中获取employees对象
        List employees = (List) servletContext.getAttribute("employees");
        // 添加到列表
        employees.add(employee);
        // 再将新列表写入到上下文
        servletContext.setAttribute("employees", employees);

        // 让请求跳转到jsp页面， 这是post方法返回的响应报文
        req.getRequestDispatcher("/employee.jsp").forward(req, resp);
    }
}
