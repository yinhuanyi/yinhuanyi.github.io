""" 
    1：用户注册
    2：用户找回密码
    3：用户修改邮箱
在发送邮箱验证码之前，需要先将随机字符串和用户的邮箱保存到数据库，之后再根据不同的需求发送邮件，
用户在接收到邮件之后，
    如果是注册、重置密码，那么需要点击邮箱中的链接，这个链接中有之前生成的随机字符串，且保存在数据库中，用户点击这个链接，会发起一个get请求，服务器端接收这个get请求之后，获取链接中的随机字符串，与数据库中的随机字符串比对，如果数据库中有，那么通过用户的邮箱拿到Userprofile的记录，最后修改Userprofile记录的字段

    如果是修改邮箱，那么需要复制这个随机字符串，粘贴到对应的input框中，当用户点击提交，那么后台的处理逻辑和上面类似

"""
import random

# 导入邮箱验证码model
from users.models import EmailVerifyRecord   # 验证码model
from django.core.mail import send_mail       # 导入Django自带邮箱发送函数
from MxOnline.settings import EMAIL_FROM     # 这里的EMAIL_FROM就是163邮箱地址

# 像这种以后直接打包上传到pypi服务器
def generate_ramdom_str(random_length=8):
    '''
    生成随机字符串
    :param random_length: 表示随机字符串的长度
    :return: 返回一个随机字符串
    '''
    string = ""
    chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    length = len(chars) - 1  # 因为字符串后面需在这个chars字符串中取字符，因此需要减去1
    for _ in range(random_length):  # 默认循环8次
        string += chars[random.randint(0, length)]

    return string


def send_register_email(email, send_type="register"):
    '''
    将发送给用户的邮件信息保存到数据库中，方便密码找回
    :param email: 发送邮件的目标地址
    :param send_type: 发送邮件的类型
    :return: None,
    '''
    # 发送邮件之前保存到数据库
    email_record = EmailVerifyRecord() # 实例化这个邮箱认证的model， 将发送认证邮件的信息保存到数据库
    code = generate_ramdom_str(4) if send_type == "update_email" else generate_ramdom_str(16) # 生成一个16位的字符串
    email_record.code = code # 发送的随机验证码
    email_record.email = email # 发送的邮箱地址
    email_record.send_type = send_type # 发送的类型有两种，一种是注册，一种是找回密码
    email_record.save() # 保存到数据库

    # 发送邮件的内容
    email_title = ""
    email_body = ""

    # 发送邮件的类型是注册类型， 由于用户注册发送邮件
    if send_type == "register":
        email_title = "尹欢一教育在线网注册链接"
        # 这个处理连接激活的逻辑需要写在urls.py中
        email_body = "请点击下边的连接，激活您的账号: http://127.0.0.1:8000/active/{}".format(code) # 测试
        # email_body = "请点击下边的连接，激活您的账号: http://edu.yhyblog.cn:8000/active/{}".format(code) # 部署
        # 调用Django的邮件函数发送邮件， 需要在setting中配置发送邮件的人
        send_status = send_mail(email_title, email_body, EMAIL_FROM, [email])
        if send_status: # 如果邮件发送成功
            return

    elif send_type == 'forget':
        email_title = "尹欢一教育在线网密码重置链接"
        email_body = "请点击下边的连接，重置您的密码: http://127.0.0.1:8000/reset/{}".format(code)
        send_status = send_mail(email_title, email_body, EMAIL_FROM, [email])
        if send_status: # 如果邮件发送成功
            return


    elif send_type == "update_email":
        email_title = "尹欢一教育在线网 修改邮箱验证码"
        email_body = "您的邮箱验证码是：{}".format(code)
        send_status = send_mail(email_title, email_body, EMAIL_FROM, [email])
        if send_status:  # 如果邮件发送成功
            return

class ActiveUserView(View):
    def get(self, request, active_code): # 这里的active_code传递进来的就是URL中的正则表达式匹配到的code
        all_code_set = EmailVerifyRecord.objects.filter(code=active_code) # 这里不用get过滤的原因是可能在EmailVerifyRecord表中生成的随机字符串重复

            # return render(request, 'login.html')
            return HttpResponseRedirect(reverse("login"))
        else:  # 如果这个验证码在后台没有
            return render(request, 'active_fail.html')

class OrgHomeView(View):

    def get(self,request, org_id):
        current_page = 'home'
        # 拿到这条机构的记录
        course_org = CourseOrg.objects.get(id=int(org_id))
        # TODO：这里是Django的一个特点，如果一个 model—A 使用了外键关联另外一个 model-B，那么 model-B 就可以通过这个model的course_set.all() 拿到所有model—A属于model-B的记录
        all_courses = course_org.course_set.all()  # 取到在org_id机构中的课程
        all_teachers = course_org.teacher_set.all() # 取到在org_id机构中的老师

        course_org.click_nums += 1

        fav_status = False

        return render(request, 'org-detail-homepage.html', {
