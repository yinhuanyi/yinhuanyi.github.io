    # 一：asyncio基础

    # (一)：基本概念

    • asyncio是异步I/O编程的一套解决方案，具备的功能如下
    1：包含各种特定系统实现的模块化事件循环

2：传输和协议抽象

3：对TPC/UDP/SSL/子进程/延时调用的支持

4：模仿futures模块，使用事件循环机制处理Future类

5：基于yield from的协议和任务，可以让用户顺序式编写异步并发代码

6：当线程中产生阻塞IO调用时，可以将这个事件转移到线程池中

7：模仿threading模块中的同步原语，可以用在单线程的协程之间

    # (二)：获取协程对象的返回值

    • 获取协程对象的返回结果与多线程获取返回结果api类似，代码如下：可以通过asyncio.ensure_future()方法，先获取协程返回的future对象，然后从future对象中获取到返回值
    import asyncio
import time


async def get_html(url):
    print("start get url")
    await asyncio.sleep(2)
    return "200"


if __name__ == '__main__':
    start_time = time.time()
    # 启动事件循环
    loop = asyncio.get_event_loop()
    # ensure_future方法可以获取协程对象的返回值，其实是一个future对象
    future = asyncio.ensure_future(get_html("https://www.yhyblog.cn"))
    loop.run_until_complete(future)
    print(time.time() - start_time)
    print(future.result())

// 输出结果

    # (三)：asyncio.wait和asyncio.gather使用区别

    • asyncio.wait是一个协程函数，等待所有协程执行完毕。多线程的wait函数是一个生成器，等待所有的线程执行完毕

    • asyncio.gather用法

        start_time = time.time()
    # 启动事件循环
    loop = asyncio.get_event_loop()
    # 创建协程对象列表
    tasks = [get_html("https://www.yhyblog.cn") for i in range(10)]
    # asyncio.gather接收一个指针
    loop.run_until_complete(asyncio.gather(*tasks))
    print(time.time() - start_time)

import time

    await asyncio.sleep(2)

if __name__ == '__main__':

    # 启动事件循环

    tasks2 = [get_html("https://www.baidu.com") for i in range(10)]

    group2 = asyncio.gather(*tasks2)

    loop.run_until_complete(asyncio.gather(group1, group2))

    print(time.time() - start_time)

    # 二：asyncio的task取消和子协程调用原理

    # (一)：asyncio中task的取消原理

    • loop实例启动事件循环会调用两个函数，run_forever()或者run_util_complete()，它们的区别如下
    1：run_util_complete()事件循环会停止，当所有的协程都执行完毕了，事件循环停止运行。因为其内部当future执行完毕，会调用一个_run_until_complete_cb方法，在这个回调方法里面，会执行loop.stop()，停止事件循环

2：run_forever()事件循环不会停止，即使所有的协程都执行完毕了，asyncio的事件循环依然会继续运行
    1
    2
    3
    • 由于loop可以从future对象或者task对象中获取，当future执行完毕后，就可以获取loop实例，停止事件循环。或者当某个future执行失败的时候，可以调用loop.stop()停止其他协程继续工作。下面代码中：演示了当执行代码时候，ctr+c，捕获到KeyboardInterrupt异常，然后基于asyncio.Task.all_tasks()获取到所有的task，调用task.cancel()终止所有的task，再重启loop（注意：下面代码只能在Python3.6版本执行，如果是python3.7+版本，会抛出异常，因为asyncio.Task.all_tasks()被弃用了，无法获取到task）
    import asyncio

async def get_html(sleep_time):
    print("waiting")
    await asyncio.sleep(sleep_time)
    print("done after {}s".format(sleep_time))


if __name__ == '__main__':
    task1 = get_html(3)
    task2 = get_html(2)
    task3 = get_html(1)

    tasks = [task1, task2, task3]

    # 创建task组，这里必须传递指针
    task_group = asyncio.gather(*tasks)

    # 获取事件循环实例
    loop = asyncio.get_event_loop()

    # 运行事件循环，模拟一种异常机制，捕获到异常后，获取所有task取消
    try:
        loop.run_until_complete(asyncio.gather(task_group))

    except KeyboardInterrupt as e:
        # 获取所有的tasks任务。内部通过events.get_event_loop()获取loop对象，然后基于loop对象获取到所有的tasks
        all_tasks = asyncio.Task.all_tasks()
        # all_tasks = asyncio.all_tasks()
        for task in all_tasks:
            is_ok = task.cancel()
            if is_ok:
                print("task取消成功")
        loop.stop() # 停止loop
        loop.run_forever() # 再次重新启动
    finally:
        loop.close() # 关闭事件循环

    # (二)：asyncio中子协程调用原理

    • 上图是下面代码的运行过程：loop.run_until_complete()函数将print_sum(1, 2)协程注册到loop中后，会执行print_sum()协程，执行到await compute(x, y)时，print_sum()会pending，执行子协程compute(x,y)，执行到await asyncio.sleep(1.0)时，compute(x,y)pending，此时程序返回到Task调用方，如果此时调用方还有协程需要执行，那么就会继续执行，程序不会等待。compute(x, y)协程等待1秒后，调用方直接驱动return x + y，此时compute(x,y)协程执行完毕，并且抛出StopIteration异常。异常会被await compute(x, y)捕获到，并且获取返回值，赋值给result，再打印print("%s + %s = %s" % (x, y, result))，此时print_sum(x, y)协程执行完毕，程序继续会到调用方Task
    import asyncio

async def compute(x, y):
    print("Compute %s + %s ..." % (x, y))
    await asyncio.sleep(1.0)
    return x + y

async def print_sum(x, y):
    result = await compute(x, y)
    print("%s + %s = %s" % (x, y, result))

loop = asyncio.get_event_loop()
loop.run_until_complete(print_sum(1, 2))
loop.close()

// 输出结果：
Compute 1 + 2 ...
1 + 2 = 3

    # (二)： asyncio在线程池中的使用

    • 一般阻塞代码不要写在协程代码中，但是如果有一些任务是I/O阻塞的，就可以放在线程池中执行

    • 下面代码中：将get_url()这个阻塞I/O函数，放在线程中执行，每个线程封装为asyncio的Task任务，然后将任务注册到loop事件循环中

    import asyncio
from concurrent.futures import ThreadPoolExecutor
import socket
from urllib.parse import urlparse

def get_url(url):
    #通过socket请求html
    url = urlparse(url)
    host = url.netloc
    path = url.path
    if path == "":
        path = "/"

    #建立socket连接
    client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    # client.setblocking(False)
    client.connect((host, 80)) #阻塞不会消耗cpu

    client.send("GET {} HTTP/1.1\r\nHost:{}\r\nConnection:close\r\n\r\n".format(path, host).encode("utf8"))

    data = b""
    while True:
        d = client.recv(1024)
        if d:
            data += d
        else:
            break

    data = data.decode("utf8")
    html_data = data.split("\r\n\r\n")[1]
    print(html_data)
    client.close()


if __name__ == "__main__":
    import time
    start_time = time.time()
    loop = asyncio.get_event_loop()

    # 创建线程池
    executor = ThreadPoolExecutor(3)

    # 创建多线程任务列表
    tasks = []
    for url in range(20):
        url = "https://www.yhyblog.cn"
        # 让任务在线程中执行，get_url是一个阻塞函数
        task = loop.run_in_executor(executor, get_url, url)
        # 将线程任务添加到任务列表
        tasks.append(task)

    # 迭代组成协程任务
    loop.run_until_complete(asyncio.wait(tasks))
    print("last time:{}".format(time.time()-start_time))

    # (三)： asyncio模拟HTTP请求

    • 由于asyncio中没有提供HTTP请求的方法，也不能在函数中封装requests同步请求库，那么只能TCP模拟HTTP请求。下面代码中：通过封装协程的HTTP请求为一个future对象，再添加到tasks列表中，遍历tasks列表，获取已经执行完毕的协程，通过await关键字获取协程的返回结果，最后打印出来
    import asyncio
import socket
from urllib.parse import urlparse

# 这里类似于一个协程的requests请求
async def get_url(url):
    #通过socket请求html
    url = urlparse(url)
    host = url.netloc
    path = url.path
    if path == "":
        path = "/"

    #建立socket连接，异步操作
    reader, writer = await asyncio.open_connection(host,80)
    writer.write("GET {} HTTP/1.1\r\nHost:{}\r\nConnection:close\r\n\r\n".format(path, host).encode("utf8"))
    all_lines = []
    async for raw_line in reader:
        data = raw_line.decode("utf8")
        all_lines.append(data)
    html = "\n".join(all_lines)
    return html

async def main():
    # 创建future列表，目的是获取协程的返回值，如果没有返回值，创建协程列表就好了
    tasks = []
    for url in range(20):
        url = "https://www.yhyblog.cn"
        # tasks元素是future对象
        tasks.append(asyncio.ensure_future(get_url(url)))

    # asyncio.as_completed方法，类似于多线程的completed方法，可以获取到当前已经执行完毕的future对象的结果
    for task in asyncio.as_completed(tasks):
        # 这里的task是一个协程，不是future对象了，通过await获取协程的返回值
        result = await task
        print(result)

if __name__ == "__main__":
    import time
    start_time = time.time()
    loop = asyncio.get_event_loop()

    # 这里直接注册了一个函数，这是一个协程函数，所以可以直接组成到loop中
    loop.run_until_complete(main())
    print('last time:{}'.format(time.time()-start_time))

    # (四)：asyncio中Future和Task对象的区别

    • Future是协程的结果容器，可以从Future中获取到协程的返回值

    • Task是Future的子类，Task是协程和Future通信的桥梁，在Task对象中需要激活协程，如果协程执行完毕，会捕获到协程的StopIteration异常，将结果存储在Future中。那么基于Task就完成了协程与Future的通信

    # (五)：asyncio中使用lock互斥锁和queue的使用

    • asyncio.Lock是协程锁，因此在加锁的时候，需要使用async with语法，或者使用with await语法
    import asyncio
import aiohttp
from asyncio import Lock, Queue

lock = Lock()

cache = {}

async def get_stuff(url):
    async with lock: # 这里加一把锁，避免发送两次请求
        if url in cache:
            return cache[url]
        stuff = await aiohttp.request('GET', url)
        cache[url] = stuff
        return stuff

async def parse_stuff():
    stuff = await get_stuff("https://www.yhyblog.cn")

async def use_stuff():
    stuff = await get_stuff("https://www.yhyblog.cn")

if __name__ == '__main__':
    tasks = [parse_stuff(), use_stuff()]

    loop = asyncio.get_event_loop()
    loop.run_until_complete(asyncio.wait(tasks))
