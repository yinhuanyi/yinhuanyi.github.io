    import numpy as np 
from sklearn import datasets

digits = datasets.load_digits()
X = digits.data
# 因为是引用，所以拷贝
y = digits.target.copy()

# 将10个类别，分为了2个类别，将9这个类别改为1，其他类别改为0
y[digits.target==9] = 1
y[digits.target!=9] = 0

# 数据拆分
from sklearn.model_selection import train_test_split
X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=666)

# 使用逻辑回归预测
from sklearn.linear_model import LogisticRegression
log_reg = LogisticRegression()
log_reg.fit(X_train, y_train)
log_reg.score(X_test, y_test)
输出结果：
0.9755555555555555

# 拿到预测值
y_log_predict = log_reg.predict(X_test)

# 计算混淆矩阵
def TN(y_true, y_predict):
    assert len(y_true) == len(y_predict)
    return np.sum((y_true == 0) & (y_predict == 0))

def FP(y_true, y_predict):
    assert len(y_true) == len(y_predict)
    return np.sum((y_true == 0) & (y_predict == 1))

def FN(y_true, y_predict):
    assert len(y_true) == len(y_predict)
    return np.sum((y_true == 1) & (y_predict == 0))

def TP(y_true, y_predict):
    assert len(y_true) == len(y_predict)
    return np.sum((y_true == 1) & (y_predict == 1))



def confusion_matrix(y_true, y_predict):
    return np.array([
        [TN(y_true, y_predict), FP(y_true, y_predict)],
        [FN(y_true, y_predict), TP(y_true, y_predict)],
    ])

# 混淆矩阵
confusion_matrix(y_test, y_log_predict)
输出结果：
array([[403,   2],
       [  9,  36]])


# 计算精准率和
def precision_score(y_true, y_predict):
    tp = TP(y_true, y_predict)
    fp = FP(y_true, y_predict)
    try:
        return tp / (tp + fp)
    except:
        return 0.0
    


# 召回率
def recall_score(y_true, y_predict):
    tp = TP(y_true, y_predict)
    fn = FN(y_true, y_predict)
    try:
        return tp / (tp + fn)
    except:
        return 0.0
    
print(precision_score(y_test, y_log_predict))
print(recall_score(y_test, y_log_predict))
输出结果：
0.9473684210526315
0.8

    # (三)精准率和召回率的重要性对比以及F1 Score

    • 什么时候精准率比较重要
    	当预测股票的时候，精准率可以明显判断出，股票是赚钱还是亏损了，而召回率对于股票来说，没有买的股票，跌了还是涨了不会有任何亏损
    1
    • 什么时候召回率比较重要
    	当预测医疗事故的时候，本来病人已经得病了，但是没有预测出来，导致病人病情恶化，我们希望将所有有病的病人都预测出来
    1
    • F1 Score
    	F1 Score：可以同时兼顾精准率和召回率，使用的是precision和recall的调和平均值
    1

    • F1 Score实现
    def f1_score(precision, recall):
    try:
        return 2*precision*recall / (precision + recall)
    except:
        return 0.0


precision = 0.5
recall = 0.5
f1_score(precision, recall)
输出结果：
0.5

    # (五)准确率和召回率可视化曲线

    • 让threshold取不同的值，绘制Precision-Recall曲线
    from sklearn.linear_model import LogisticRegression
import matplotlib.pyplot as plt
log_reg = LogisticRegression()
log_reg.fit(X_train, y_train)
# 获取决策分数值
decision_scores = log_reg.decision_function(X_test)

# 通过决策分数值的最小值和最大值得到一个区间，步长为0.1
thresholds = np.arange(np.min(decision_scores), np.max(decision_scores), 0.1)
precisions = []
recalls = []
for threshold in thresholds:
    # 通过threshold的取不同的值，直接对decision_scores的值进行样本分类, 获取分类结果
    y_predict = np.array(decision_scores >= threshold, dtype='int')
    #计算准确率和召回率, 保存到列表中
    precisions.append(precision_score(y_test, y_predict))
    recalls.append(recall_score(y_test, y_predict))

plt.plot(thresholds, precisions, label='precision')
plt.plot(thresholds, recalls, label='recall')
plt.legend()
plt.show()

    def TPR(y_true, y_predict):
	tp = TP(y_true, y_predict)
	fn = FN(y_true, y_predict)
	try:
		return tp/(tp+fn)
	except:
		return 0.0
	
def FPR(y_true, y_predict):
	fp = FP(y_true, y_predict)
	tn = TN(y_true, y_predict)
	try:
		return fp / (fp + tn)
