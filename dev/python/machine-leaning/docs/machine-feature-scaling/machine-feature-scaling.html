    # (一)数据归一化 Feature Scaling 思想

    解决方法：将所有的数据都映射到同一尺度
    1
    • 1:最值归一化（normalization）：将所有的数据都映射到0-1之间, 适用于分布有明显边界的情况， 受到outlier影响较大

    • 2:均值方差归一化 standardization: 将所有数据归一到均值为0，方差为1的分布中

    • 自己实现StandardScaler归一化处理类
    import numpy as np
from sklearn import datasets
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier


class MyStandardScaler:
    def __init__(self):
        self.mean = None
        self.scale = None

    def fit(self, X):
        # 保证矩阵的维度一定是2
        assert X.ndim == 2, 'The dimension of X must be 2'
        # 获取矩阵每一列的均值
        self.mean_ = np.array([np.mean(X[:, i]) for i in range(X.shape[1])])
        # 获取矩阵每一列的方差
        self.scale_ = np.array([np.std(X[:, j]) for j in range(X.shape[1])])

        return self

    # transform 方法需要返回一个归一化之后的新特征矩阵
    def transform(self, X):
        '''将X进行举止方差归一化'''
        assert X.ndim == 2, '特征矩阵必须是二维矩阵'
        assert self.mean_ is not None and self.scale_ is not None, \
            '必须先调用fit()方法，再调用transform()方法'
        assert X.shape[1] == len(self.mean_), \
            '矩阵的列数必须等于矩阵均值向量和方差向量的长度'

        # 创建一个新的特征矩阵用于返回结果
        restX = np.empty(shape=X.shape, dtype=float)

        # 对新矩阵的每一列进行归一化
        # 归一化计算过程使用了universal function
        for col in range(X.shape[1]):
            restX[:, col] = (X[:, col] - self.mean_[col]) / self.scale_[col]
        return restX

    myStandardScaler = MyStandardScaler()

    X_train = myStandardScaler.transform(X_train)
    X_test = myStandardScaler.transform(X_test)

    # 4：使用knn算法进行测试
    knn_clf = KNeighborsClassifier()
    knn_clf.fit(X_train, y_train)
