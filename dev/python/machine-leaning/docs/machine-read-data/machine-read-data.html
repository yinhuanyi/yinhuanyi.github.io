    # (一)iris数据集说明

    import numpy as np
import matplotlib as mpl
import matplotlib.pyplot as plt
from sklearn import datasets

# 获取iris数据集
iris = datasets.load_iris()

# 查看iris数据集的元数据
iris.keys()
dict_keys(['data', 'target', 'target_names', 'DESCR', 'feature_names', 'filename'])

# 看看数据集的描述
print(iris.DESCR)


# 查看iris数据集的矩阵维度，150行，4列
iris.data.shape
输出结果：
(150, 4)

# 查看每一列表示什么意思
iris.feature_names
输出结果：
['sepal length (cm)',

 'petal width (cm)']

# 这个属性是一维向量，150个元素，表示每种花的类型

    # (二)iris数据集的使用

    # 查看iris数据集矩阵
iris.data
输出结果：
array([[5.1, 3.5, 1.4, 0.2],
       [4.9, 3. , 1.4, 0.2],
       [4.7, 3.2, 1.3, 0.2],
       [4.6, 3.1, 1.5, 0.2],
       ...
       [5.9, 3. , 5.1, 1.8]])

# 获取数据集中的前2列， 描绘为散点图
X = iris.data[:, :2]

# 将矩阵的第1列和第2列绘制散点图
plt.scatter(X[:, 0], X[:, 1])
plt.show()

# 这里的y==1,表示取出versicolor这种花的行
plt.scatter(X[y==1, 0], X[y==1, 1], color='blue', marker='+')
# 这里的y==2,表示取出virginica这种花的行
plt.scatter(X[y==2, 0], X[y==2, 1], color='green', marker='x')
