    # 一：etcd功能与原理

    # (一) etcd作为服务发现系统的特点

    • 特点
    1：简单：安装配置简单，而且提供了HTTP API进行交互，使用也很简单
2：安全：支持SSL证书验证
3：快速：根据官方提供的benchmark数据，单实例支持每秒2k+读操作
4：允许外部应用实时的监听K-V变化
5：可靠：采用raft算法，实现分布式系统数据的可用性和一致性
    1
    2
    3
    4
    5
    • etcd应用场景
    Etcd要解决服务发现的问题，需要下面三大支柱，缺一不可
	1：一个强一致性、高可用的服务存储目录：raft协议
	2：一种注册服务和健康服务健康状况的机制：对注册的服务配置key生存时间值，保持服务的心跳以达到监控健康状态的效果
	3：一种查找和连接服务的机制：在主题下注册的服务，能在对应的主题下查找到
    1
    2
    3
    4
    • Etcd与Raft的关系
    1：Raft是强一致性集群日志同步算法
2：Etcd基于Raft算法在集群中同步K/V
    1
    2
    • Etcd高性能写入的原理（集群需要2N+1个节点）

    1：客户端请求leader写入数据
2：Leader基于raft协议，将数据同步到Follow
3：只要满足N+1个Follow同步到了数据，那么Leader执行本地提交事务，且响应客户端请求
4：最后Leader将本地的提交事务同步到所有Follow

    # (三) 设置key过期以及自动续约

    • 代码如下
    func main() {

	// 生成配置
	config := clientv3.Config{
		Endpoints: []string{"127.0.0.1:2379"},
		DialTimeout: 5 * time.Second,
	}

	// 创建连接
	client, err := clientv3.New(config)
	if err != nil {
		fmt.Println(err)
		return
	}
	defer client.Close()

	// 申请租约， 申请一个10秒的租约
	lease := clientv3.NewLease(client)
	leaseGrantResp, err := lease.Grant(context.Background(), 10)
	if err !=nil {
		fmt.Println(err)
	}
	
	// 获取租约ID
	leaseId := leaseGrantResp.ID

	// 让这个leaseId自动续租，到了9秒就续租
	keepRespChan, err := lease.KeepAlive(context.Background(), leaseId)
	if err !=nil {
		fmt.Println(err)
	}

	// 验证自动续约
	go func() {
		for {
			select {
			case keepResp := <- keepRespChan:
				if keepRespChan == nil {
					fmt.Println("租约已经失效了")
					// 跳出for循环
					goto END
				}else { // 每秒续租一次，所以会收到一次应答
					fmt.Printf("收到自动续租的应答: %#v\n", keepResp)
				}
			}
		}
		END:
	}()

	// 写入数据
	kv := clientv3.NewKV(client)
	putResp, err := kv.Put(context.Background(), "/cron/lock/job1", "", clientv3.WithLease(leaseId))
	if err !=nil {
		fmt.Println(err)
	}
	fmt.Println("写入成功：", putResp.Header.Revision)


	// 定时看下key是否过期, 使用for循环一直查看
	var getResp *clientv3.GetResponse
	for {
		getResp, err = kv.Get(context.Background(), "/cron/lock/job1")
		if err !=nil {
			fmt.Println(err)
		}
		if getResp.Count == 0 {
			fmt.Println("kv过期了")
			break
		}
		fmt.Println("还没有过期：", getResp.Kvs)
		time.Sleep(2 * time.Second)
	}

}

    # (四) watch监听Etcd的kv数据变化

    • 监听kv的PUT和Delete变化
    func main() {

	// 生成配置
	config := clientv3.Config{
		Endpoints: []string{"127.0.0.1:2379"},
		DialTimeout: 5 * time.Second,
	}

	// 创建连接
	client, err := clientv3.New(config)
	if err != nil {
		fmt.Println(err)
		return
	}
	defer client.Close()

	// 模拟kv变化, 先创建kv，然后删除
	kv := clientv3.NewKV(client)
	go func() {
		for {
			kv.Put(context.Background(), "/cron/jobs/job3", "i am job3")
			kv.Delete(context.Background(), "/cron/jobs/job3")
			time.Sleep(1 * time.Second)
		}
	}()

	// 先查询到值，然后再watch变化
	getResp, err := kv.Get(context.Background(), "/cron/jobs/job3")
	if err != nil {
		fmt.Println(err)
		return
	}

	// 如果可以获取到数据
	if len(getResp.Kvs) != 0 {
		fmt.Println("当前值：", string(getResp.Kvs[0].Value))
	}

	// 获取 "/cron/jobs/job3" 当前的revisionID后一个ID值，表示从这个ID开始监听
	watchStartRevision := getResp.Header.Revision + 1

	// 创建一个watch
	watcher := clientv3.NewWatcher(client)

	fmt.Printf("从当前%d开始监听\n", watchStartRevision)

	// 开始监控, 5秒后停止监控
	ctx, _ := context.WithTimeout(context.Background(), 5 * time.Second)
	watchRespChan := watcher.Watch(ctx, "/cron/jobs/job3", clientv3.WithRev(watchStartRevision))

	// 从通道中获取事件
	for watchResp := range watchRespChan {
		// 获取事件
		for _, event := range watchResp.Events {
			switch event.Type {
			case mvccpb.DELETE:
				fmt.Printf("这是一个Delete操作, ModRevision=%d\n", event.Kv.ModRevision)
			case mvccpb.PUT:
				fmt.Printf("这是一个Put操作, value=%s, CreateRevision=%d, ModRevision=%d\n", event.Kv.Value, event.Kv.CreateRevision, event.Kv.ModRevision)
			}
		}
	}
}

    • 在Etcd中，可以将所有的CRUD操作都抽象为OP操作
    func main() {

	// 生成配置
	config := clientv3.Config{
		Endpoints: []string{"127.0.0.1:2379"},
		DialTimeout: 5 * time.Second,
	}

	// 创建连接
	client, err := clientv3.New(config)
	if err != nil {
		fmt.Println(err)
		return
	}
	defer client.Close()

	// 模拟kv变化
	kv := clientv3.NewKV(client)

	// 创建putOp
	putOp := clientv3.OpPut("/cron/jobs/job8", "haha")
	// 执行opResp
	opResp, err := kv.Do(context.Background(), putOp)
	if err != nil {
		fmt.Println(err)
		return
	}

	fmt.Printf("写入Revision：%d\n", opResp.Put().Header.Revision)

	// 创建getOp
	getOp := clientv3.OpGet("/cron/jobs/job8" )
	// 执行getOp
	opResp, err = kv.Do(context.Background(), getOp)
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Printf("查询的value：%s\n", opResp.Get().Kvs[0].Value)
}


// 输出：
写入Revision：48
查询的value：haha

    # 四：etcd事务

    # (一) 基于Option操作、txn事务实现乐观锁

    • 主要思路：申请租约、创建key、执行业务、删除租约让key失效
    func main() {

	// 生成配置
	config := clientv3.Config{
		Endpoints: []string{"127.0.0.1:2379"},
		DialTimeout: 5 * time.Second,
	}

	// 创建连接
	client, err := clientv3.New(config)
	if err != nil {
		fmt.Println(err)
		return
	}
	defer client.Close()

	// 模拟kv变化
	kv := clientv3.NewKV(client)

	// lease实现锁自动过期
	// op操作
	// txn事务

	// 1：上锁
	lease := clientv3.NewLease(client)
	leaseGrantResp, err := lease.Grant(context.Background(), 5)
	if err !=nil {
		fmt.Println(err)
	}
	// 获取租约ID
	leaseId := leaseGrantResp.ID

	// 创建context，用于取消自动续租
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel() // 取消context
	defer lease.Revoke(context.TODO(), leaseId) // 取消lease，让key过期

	keepRespChan, err := lease.KeepAlive(ctx, leaseId)
	if err !=nil {
		fmt.Println(err)
	}

	go func() {
		for {
			select {
			case keepResp := <- keepRespChan:
				if keepRespChan == nil { // 如果租约被remove了,通道关闭了，就会获取到nil
					fmt.Println("租约已经失效了")
					// 跳出for循环
					goto END
				}else { // 每秒续租一次，所以会收到一次应答
					fmt.Printf("收到自动续租的应答: %d\n", keepResp.ID)
				}
			}
		}
	END:
	}()

	txn := kv.Txn(context.Background())  // 创建事务
	// 如果CreateRevision创建成功，表明key过期，创建key，也就是添加锁
	txn.If(clientv3.Compare(clientv3.CreateRevision("/cron/lock/job9"), "=", 0)). // 判断key是否已经存在，如果存在返回1，不存在返回0
		Then(clientv3.OpPut("/cron/lock/job9", "", clientv3.WithLease(leaseId))).  // 创建key
		Else(clientv3.OpGet("/cron/lock/job9")) // 抢锁失败

	txnResp, err := txn.Commit() // 提交事务
	if err !=nil {
		fmt.Println(err)
	}
	// 如果没有创建key，也就是没有抢到锁
	if !txnResp.Succeeded {
		fmt.Println("锁被占用", string(txnResp.Responses[0].GetResponseRange().Kvs[0].Value))
		return
	}

	// 2：处理业务
	fmt.Println("处理业务")
	time.Sleep(30 * time.Second)

	// 3：执行完业务后，必须释放锁(取消自动续租，并且释放租约)
	fmt.Println("执行defer操作")

}
