    golang中分布式任务执行架构分析与实现

    尹欢一的技术博客 2020/6/1

    # 一：架构分析

    # (一) 架构思想

    • 1：使用Etcd的watch功能，替代了消息队列发布订阅功能
    	当Etcd中写入一个key的时候，所有的worker都将监控到这个key，并且获取到key的value值，替代了rabbitmq的发布订阅功能
    1
    • 2：多个worker存储的任务是一样的，只要有一个worker监控，那么任务就可以执行
    	worker具备高可用性
    1
    • 3：使用Etcd的txn事务创建一个key的revision，如果创建成功，说明抢到了锁
    	抢到etcd锁通过创建key的revision实现
	释放锁通过取消etcd中key的自动续约和删除leaseID实现

    • 2：将下发的定时任务{"name": "job1", "command": "echo hello", "cronExpr": "* * * * *"}，写入到Etcd的/cron/jobs/目录下，例如：key为:/cron/jobs/果是更新，那么会返回之前的定时任务
    // SaveJob 写入任务到Etcd(创建新任务 或者 修改任务)
func (j *JobMgr) SaveJob (job *common.Job) (oldJob *common.Job, err error) {
	// 把任务写入到/cron/jobs/任务名称 -> json

	jobKey := common.JOB_SAVE_DIR + job.Name
	jobValue, err := json.Marshal(job)
	if err != nil {
		log.Panicf(err.Error())
	}

	// 这里的任务写入不会失效
	putResp, err := j.kv.Put(context.Background(), jobKey, string(jobValue), clientv3.WithPrevKV())
	if err != nil {
		log.Panicf(err.Error())
	}

	// 如果是更新，可以返回覆盖之前的值
	if putResp.PrevKv != nil {
		oldJobString := new(common.Job)
		err = json.Unmarshal(putResp.PrevKv.Value, oldJobString)
		if err != nil {
			log.Panicf(err.Error())
		}
		oldJob = oldJobString
	}

	return

}

    • 3：删除定时任务是将任务从/cron/jobs/目录下删除掉，删除任务的时候，只需要知道key即可，也就是/cron/jobs/job1, 删除也会返回之前的任务
    func (j *JobMgr) DeleteJob(name string) (oldJob *common.Job, err error) {

	// etcd中保存任务的key
	jobKey := common.JOB_SAVE_DIR + name

	// 从etcd中删除它
	delResp, err := j.kv.Delete(context.Background(), jobKey, clientv3.WithPrevKV())
	if err != nil {
		log.Panicf(err.Error())
	}

	// 如果删除之前有值，返回删除之前的数据
	if len(delResp.PrevKvs) != 0 {

		oldJobObj := new(common.Job)
		err = json.Unmarshal(delResp.PrevKvs[0].Value, oldJobObj)
		if err != nil {
			log.Panicf(err.Error())
		}
		oldJob = oldJobObj
	}
	return
}

    • 4：强杀任务需要将任务名称写入到/cron/killer/目录下，例如：/cron/killer/job1，写入的时候使用租约，租约有效时间是1秒，1秒后，key自动删除，目的只是通过etcd给worker节点发送任务强杀的指令，不需要存储
    // KillJob 杀死任务
func (j *JobMgr) KillJob(name string) (err error) {
	// 更新一下key=/cron/killer/任务名

	// 通知worker杀死对应任务
	killerKey := common.JOB_KILLER_DIR + name

	// 授权一个1秒的租约，租约让key稍后自动过期，如果key写入etcd，被worker监控到，那么杀死对应的任务，且key会1秒后自动过期
	leaseGrantResp, err := j.lease.Grant(context.Background(), 1)
	if err != nil {
		log.Panicf(err.Error())
	}

	// 获取租约ID
	leaseId := leaseGrantResp.ID


	// 写入一个带租约的key
	_, err = j.kv.Put(context.TODO(), killerKey, "", clientv3.WithLease(leaseId))
	if err != nil {
		log.Panicf(err.Error())
	}

	return
}

    # 三：worker 实现

    # (一) 业务设计流程

    • worker的业务设计流程如下

    1：在jobmgr中，监控etcd中/cron/jobs/ 和 /cron/killer/目录的变化, 基于变化的key获取到value，将value转换为事件对象写入到jobEventChan中。【启动协程】监控/cron/jobs/目录的时候，会先将已经存在的任务，推送到jobEventChan中，再监控/cron/jobs/目录，判断事件是PUT 还是DELETE，PUT和DELETE分别构建后，推送到jobEventChan中。启动协程监控/cron/killer/目录的时候，如果判断是PUT事件，构建kill时间对象，推送到jobEventChan中

2：scheduler会在初始化的时候，【启动协程】调用scheduleLoop，在scheduleLoop里面，for select 方式获取jobEventChan、jobResultChan、scheduleTimer.C这三个通道的数据，如果jobEventChan有数据，判断类型后，交由executor去执行，执行的结果写入到jobResultChan中。对于jobEventChan中的事件，如果是save类型就将任务写入到任务表，如果是delete类型就从任务表删除任务，如果是强杀类型就获取执行表任务，调用cancel终止任务执行。对于jobResultChan中的事件，接收从jobResultChan中获取的事件结果，将结果转换为log对象，交由sink模块处理，写入到MongoDB中。如果从scheduleTimer.C获取到数据，那么什么都不做，主要是让for select语句不阻塞，主要for select语句不阻塞，那么就可以遍历jobPlanTable任务表，计算所有任务中，最近一次需要执行的时间点，从而获取下定时器的时间间隔。如果是从jobEventChan、jobResultChan获取到数据，那么遍历列表的时间点(调度的时间点)会提前

3：在executor中比较特别，启动的时候只是初始化一个全局变量，当schedule有job任务需要执行的时候，才会调用执行器的ExecuteJob方法，在方法里面【启动协程】，每个任务都是在协程中执行的，再将任务的结果返回给scheduler的jobResultChan

4：在jobsink中，【启动协程】从logChan队列中获取log结构体对象写入到MongoDB中，这是scheduler将job对象写入到logChan队列中的

5：在joblock中，是解决多个worker工作的情况下，每个worker都有相同的jobPlan，那么当每个worker执行的时候，哪个worker可以抢到etcd中的锁，就哪个执行任务，执行完毕后，在处理结果日志方法中，将jobExecutingTable(任务执行表)中对应的job移除掉。那么下次当任务过期后，所有的worker又再次争抢etcd的锁了。因此jobExecutingTable的意义在于：有些任务需要执行很长时间，可能卡住了，在这个过程中，定时任务到了时间又会触发任务，jobExecutingTable避免前一个任务没有执行完，又执行任务的情况。分布式任务锁的意义在于：所有worker都有完整的jobPlanTable任务表，但是只能一个worker执行任务，也就是谁抢到锁，谁执行任务

6：在register中，主要是将worker的IP都注册到etcd中，让master知道有多少worker可以执行任务

    # (二) 相关代码

    • jobmgr.go
    /**
 * @Author: Robby
 * @File name: jobmgr.go
 * @Create date: 2021-08-17
 * @Function: 监控etcd中，/cron/jobs/ 和 /cron/killer/的变化, 将基于变化的key获取到value，将value转换为事件写入到jobEventChan中
 **/

package worker

import (
	"context"
	"go-task/common"
	"go.etcd.io/etcd/api/v3/mvccpb"
	clientv3 "go.etcd.io/etcd/client/v3"
	"time"
)

var G_jobMgr *JobMgr

type JobMgr struct {
	client *clientv3.Client
	kv clientv3.KV
	lease clientv3.Lease
	watcher clientv3.Watcher
}

// watchJobs 监听/cron/jobs/目录下key的变化，将变化的job转换为事件写入到jobEventChan
func (j *JobMgr) watchJobs() (err error) {

	var jobEvent *common.JobEvent

	// 获取/cron/jobs/目录下的所有任务，也就是所有的key，并且获知当前集群的revision
	getResp, err := j.kv.Get(context.TODO(), common.JOB_SAVE_DIR, clientv3.WithPrefix())
	if err != nil {
		return
	}

	// 遍历/cron/jobs/目录下所有的key， getResp.Kvs == [{"/cron/jobs/job1": "{"name": "job1", "command": "echo hello", "cronExpr": "* * * * *"}"}]
	// 每次程序启动的时候，会先从执行一下所有写入到/cron/jobs/目录下的key
	for _, kvpair := range getResp.Kvs {

		// 反序列化json得到Job
		job, err1 := common.UnpackJob(kvpair.Value)
		if err1 != nil {
			return
		}

		// 创建save事件
		jobEvent = common.BuildJobEvent(common.JOB_EVENT_SAVE, job)

		// 将事件写入到jobEventChan
		G_scheduler.PushJobEvent(jobEvent)
	}

	// 2, 从该revision向后监听变化事件， 监听协程
	go func() {

		// 从GET时刻的后续版本开始监听变化
		watchStartRevision := getResp.Header.Revision + 1
		// 监听/cron/jobs/目录的后续变化
		watchChan := j.watcher.Watch(context.TODO(), common.JOB_SAVE_DIR, clientv3.WithRev(watchStartRevision), clientv3.WithPrefix())

		// 如果没有key变化，那么这里会阻塞
		for watchResp := range watchChan {
			for _, watchEvent := range watchResp.Events {

				switch watchEvent.Type {

				case mvccpb.PUT: // 如果是保存事件
					job, err := common.UnpackJob(watchEvent.Kv.Value)
					if err != nil {
						continue
					}
					// 构建一个更新Event
					jobEvent = common.BuildJobEvent(common.JOB_EVENT_SAVE, job)

				case mvccpb.DELETE: // 如果是删除事件

					jobName := common.ExtractJobName(string(watchEvent.Kv.Key), common.JOB_SAVE_DIR)

					job := &common.Job{Name: jobName}

					// 构建一个删除Event
					jobEvent = common.BuildJobEvent(common.JOB_EVENT_DELETE, job)
				}

				// 将事件写入到jobEventChan
				G_scheduler.PushJobEvent(jobEvent)
			}
		}
	}()
	return
}

// watchKiller  监听/cron/killer/目录下key的变化，将变化的job转换为事件写入到jobEventChan
func (j *JobMgr) watchKiller() {

	// 监听/cron/killer目录
	go func() {

		// 监听/cron/killer/目录下key的变化
		watchChan := j.watcher.Watch(context.TODO(), common.JOB_KILLER_DIR, clientv3.WithPrefix())

		// 处理监听事件, 如果没有事件，这里会阻塞
		for watchResp := range watchChan {
			for _, watchEvent := range watchResp.Events {
				switch watchEvent.Type {
				case mvccpb.PUT: // 新建kill任务

					jobName := common.ExtractJobName(string(watchEvent.Kv.Key), common.JOB_KILLER_DIR)
					job := &common.Job{Name: jobName}
					jobEvent := common.BuildJobEvent(common.JOB_EVENT_KILL, job)

					// 将事件写入到jobEventChan
					G_scheduler.PushJobEvent(jobEvent)

				}
			}
		}
	}()
}

// InitJobMgr 初始化管理器
func InitJobMgr() (err error) {

	// 初始化配置
	config := clientv3.Config{
		Endpoints: G_config.EtcdEndpoints, // 集群地址
		DialTimeout: time.Duration(G_config.EtcdDialTimeout) * time.Millisecond, // 连接超时
	}

	// 建立连接
	client, err := clientv3.New(config)
	if err != nil {
		return
	}

	// 创建上下文
	timeoutCtx, cancel := context.WithTimeout(context.Background(), 2 * time.Second)
	defer cancel()
	_, err = client.Status(timeoutCtx, config.Endpoints[0])
	if err != nil {
		return
	}

	// 得到KV和Lease的API子集
	kv := clientv3.NewKV(client)
	lease := clientv3.NewLease(client)
	watcher := clientv3.NewWatcher(client)

	// 赋值单例
	G_jobMgr = &JobMgr{
		client: client,
		kv: kv,
		lease: lease,
		watcher: watcher,
	}

	// 基于协程启动监听/cron/jobs/目录下key的变化
	err = G_jobMgr.watchJobs()
	if err != nil {
		return
	}

	// 基于协程启动监听/cron/killer/目录下key的变化
	G_jobMgr.watchKiller()


	return

}

// Scheduler 任务调度
type Scheduler struct {

	/*
		etcd的事件
		jobEventChan = {
		    "EventType":"1",  表示put操作
		    "Job":{
		        "name":"job1",
		        "command":"ls -l",

	jobEventChan chan *common.JobEvent //  etcd任务事件队列

	/*
		任务计划表结构

		            "name":"job1",

		        },
		        "expr":"* * * * * *",
		        "nextTime":"1秒"
		    }
		}
	*/
	jobPlanTable map[string]*common.JobSchedulePlan // 任务调度计划表

	/*
			任务执行表结构
		jobExecutingTable = {
		    "job1":{
		        "job":{
		            "name":"job1",
		            "command":"ls -l",
		            "conExpr":"* * * * *"
		        },
		        "PlanTime":"理论调度时间点",
		        "RealTime":"时间调度时间点",
		        "CancelCtx":"ctx上下文",
		        "CancelFunc":"ctx cancel函数"
		    }
		}
	*/
	jobExecutingTable map[string]*common.JobExecuteInfo // 任务执行表

	/*
		jobResultChan == {
		    "output":"脚本执行结果",
		    "err":"错误原因",
		    "StartTime":"任务启动时间点",
		    "EndTime":"任务结束时间点",
		    "ExecuteInfo":{
		        "job":{
		            "name":"job1",
		            "command":"ls -l",
		            "conExpr":"* * * * *"
		        },
		        "PlanTime":"理论调度时间点",
		        "RealTime":"时间调度时间点",
		        "CancelCtx":"ctx上下文",
		        "CancelFunc":"ctx cancel函数"
		    }
		}
	*/
	jobResultChan chan *common.JobExecuteResult // 任务结果队列
}

// InitScheduler 初始化调度器, 也就是初始化任务执行
func InitScheduler() (err error) {
	G_scheduler = &Scheduler{
		jobEventChan:      make(chan *common.JobEvent, 1000),
		jobPlanTable:      make(map[string]*common.JobSchedulePlan),
		jobExecutingTable: make(map[string]*common.JobExecuteInfo),
		jobResultChan:     make(chan *common.JobExecuteResult, 1000),
	}
	// 启动调度协程
	go G_scheduler.scheduleLoop()
	return
}

// TryStartJob 为TrySchedule服务，这里是真正的执行任务，将jobPlan转换为jobExecuteInfo，调用 executor.go的接口执行任务  【看懂了】
// jobplay == {"expr": "* * * * * *", "nextTime": 1秒, "Job":{ "name":"job1", "command":"ls -l", "conExpr":"* * * * *" }}
func (s *Scheduler) TryStartJob(jobPlan *common.JobSchedulePlan) {
	// todo: 这里有个避免任务重复执行的核心思想
	// 在执行任务之前，先判断一下，任务是否在执行，如果任务执行表中有任务，那么说明已经有任务在执行了，就直接返回，避免在执行的过程中再来发现取不到etcd的锁，而放弃执行
	// 执行任务表解决的问题是：单个进程中，任务不会重复执行
	// etcd分布式任务所解决的问题是：多个进程中，任务不会重复执行
	jobExecuteInfo, jobExecuting := s.jobExecutingTable[jobPlan.Job.Name]
	if jobExecuting { // 如果可以取到值，说明任务正在执行
		fmt.Println("尚未退出,跳过执行:", jobPlan.Job.Name)
		return
	}

	// 创建执行信息
	jobExecuteInfo = common.BuildJobExecuteInfo(jobPlan)

	// 保存执行状态
	s.jobExecutingTable[jobPlan.Job.Name] = jobExecuteInfo

	// 执行任务
	fmt.Println("执行任务:", jobExecuteInfo.Job.Name, jobExecuteInfo.PlanTime, jobExecuteInfo.RealTime)

	// 基于执行信息执行任务 jobExecuteInfo == {"job":{"name":"job1","command":"ls -l","conExpr":"* * * * *"},"PlanTime":"理论调度时间点","RealTime":"时间调度时间点","CancelCtx":"ctx上下文","CancelFunc":"ctx cancel函数"}}
	G_executor.ExecuteJob(jobExecuteInfo)
}

// TrySchedule 为scheduleLoop服务，计算下次需要调度的时间间隔，同时这个方法里面也遍历了任务表执行调度任务  【看懂了】
func (s *Scheduler) TrySchedule() (scheduleAfter time.Duration) {

	// nearTime的意义在于计算出jobPlan中，最小需要执行任务的时间点
	var nearTime *time.Time

	// 当前时间
	now := time.Now()

	// 如果任务表不为空，遍历所有任务 jobPlanTable == {"job1": {"expr": "* * * * * *", "nextTime": 2021-08-08 13:13:13, "Job":{ "name":"job1", "command":"ls -l", "conExpr":"* * * * *" }}}
	for _, jobPlan := range s.jobPlanTable {
		if jobPlan.NextTime.Before(now) || jobPlan.NextTime.Equal(now) { // 满足定时表达式任务执行时间，执行任务
			s.TryStartJob(jobPlan)                    // 这是执行真正的业务。 jobplay == {"expr": "* * * * * *", "nextTime": 2021-08-08 13:13:13, "Job":{ "name":"job1", "command":"ls -l", "conExpr":"* * * * *" }}
			jobPlan.NextTime = jobPlan.Expr.Next(now) // 更新下次执行时间
		}

		// 找到所有任务中，下次需要执行时间点最近的那个时间点
		if nearTime == nil || jobPlan.NextTime.Before(*nearTime) {
			nearTime = &jobPlan.NextTime
		}
	}

	// 动态计算最近时间点到现在的时间间隔，如果没有任何任务，neartime可能是nil，那么直接设置下次调度时间为1秒即可
	if nearTime != nil {
		scheduleAfter = (*nearTime).Sub(now)
	}else {
		scheduleAfter = 1 * time.Second
	}


	return
}

// handleJobEvent ：判断事件类型，如果是save类型就将任务写入到任务表，如果是delete类型就从任务表删除任务，如果是强杀类型就获取执行表任务，调用cancel终止任务执行 【看懂了】
// jobEvent == ("EventType":"1", "Job":{"name":"job1", "command":"ls -l", "conExpr":"* * * * *"})
func (s *Scheduler) handleJobEvent(jobEvent *common.JobEvent) {

	// 判断事件类型
	switch jobEvent.EventType {

	case common.JOB_EVENT_SAVE:	// 保存任务事件, 也即是etcd中put了新的job任务
		jobPlan, err := common.BuildJobSchedulePlan(jobEvent.Job)
		if err != nil {
			return
		}
		// 将jobPlan添加到jobPlanTable中
		s.jobPlanTable[jobEvent.Job.Name] = jobPlan

	case common.JOB_EVENT_DELETE: // 删除任务事件
		_, jobExisted := s.jobPlanTable[jobEvent.Job.Name]
		if jobExisted {
			delete(s.jobPlanTable, jobEvent.Job.Name)
		}

	case common.JOB_EVENT_KILL: // 强杀任务事件
		// 先从执行表中获取任务，如果能够获取到，说明任务正在执行
		jobExecuteInfo, jobExecuting := s.jobExecutingTable[jobEvent.Job.Name]
		if jobExecuting {
			// 如果任务正在执行，调用context的cancel函数，中断任务执行
			jobExecuteInfo.CancelFunc()
		}
	}
}

// handleJobResult 接收从jobResultChan中获取的事件结果，将结果转换为log对象，交由sink模块处理，写入到MongoDB中【看懂了】
func (s *Scheduler) handleJobResult(result *common.JobExecuteResult) {

	// 只要获取到了任务的执行结果，说明任务执行完毕了，那么必须从任务执行表中删除掉
	delete(s.jobExecutingTable, result.ExecuteInfo.Job.Name)

	// 生成执行日志
	if result.Err != errors.New("锁已被占用") { // 这里只记录锁没有被占用的日志

		// 创建MongoDB存储的日志对象
		jobLog := &common.JobLog{
			JobName: result.ExecuteInfo.Job.Name,
			Command: result.ExecuteInfo.Job.Command,
			Output: string(result.Output),
			PlanTime: result.ExecuteInfo.PlanTime.UnixNano() / 1000 / 1000,  // 精确到毫秒
			ScheduleTime: result.ExecuteInfo.RealTime.UnixNano() / 1000 / 1000,
			StartTime: result.StartTime.UnixNano() / 1000 / 1000,
			EndTime: result.EndTime.UnixNano() / 1000 / 1000,
		}

		// 如果有错误日志，那么存储，如果没有就是空
		if result.Err != nil {
			jobLog.Err = result.Err.Error()

		} else {
			jobLog.Err = ""
		}

		// 将log日志对象交给sink模块处理
		G_logSink.Append(jobLog)
	}

}

// scheduleLoop 调度协程，死循环不断获取jobEventChan中的etcd事件，和jobResultChan执行结果对象
func (s *Scheduler) scheduleLoop() {

	// 初始化一次(1秒)
	scheduleAfter := 1 * time.Second


	// 基于设置的过期时间，创建timer定时器
	scheduleTimer := time.NewTimer(scheduleAfter)

	// 死循环不断获取jobEventChan中的etcd事件，和jobResultChan执行结果对象
	for {
		select {
		// 	jobEvent = ("EventType":"1", "Job":{"name":"job1", "command":"ls -l", "conExpr":"* * * * *"})
		case jobEvent := <- s.jobEventChan: // 监听channel中etcd事件
			s.handleJobEvent(jobEvent) // 处理etcd事件

		case jobResult := <- s.jobResultChan: // 监听channel中的执行结果
			s.handleJobResult(jobResult)

			// todo：这个是干啥的 ?
			// 这个是当jobEventChan和jobResultChan没有数据的时候，为了让select不阻塞，那么当定时器过期，就会获取到scheduleTimer.C时间点，让for循环可以向下走，不断执行s.TrySchedule()，遍历jobPlanTable中的任务，执行过期任务
		case <-scheduleTimer.C: // 最近的任务到期了, scheduleTimer.C 返回当前的时间对象
		}

		// 由于scheduler.C定时器通道的存在，如果即使获取不到事件channel和结果channel的数据，也会在1秒后，执行调度任务
		scheduleAfter = s.TrySchedule()

		// 重置timer，并且每次调度的时间是通过
		scheduleTimer.Reset(scheduleAfter)
	}

}

// PushJobResult 将executor任务的执行结果写入到channel中
func (s *Scheduler) PushJobResult(jobResult *common.JobExecuteResult) {
	s.jobResultChan <- jobResult
}

// PushJobEvent 为jobmgr服务，接收job事件，写入到jobEventChan中
func (s *Scheduler) PushJobEvent(jobEvent *common.JobEvent) {
	s.jobEventChan <- jobEvent
}

 * @Author: Robby
 * @File name: executor.go
 * @Create date: 2021-08-17
 * @Function: 执行scheduler中传递的job任务，将任务的执行结果返回给scheduler 【完毕】

package worker

import (
	"go-task/common"
	"math/rand"
	"os/exec"
	"time"
)

var G_executor *Executor

// Executor 任务执行器
type Executor struct {}

// InitExecutor 初始化执行器
func InitExecutor() (err error) {
	G_executor = &Executor{}
	return
}

// ExecuteJob 执行一个任务   info=={"job":{"name":"job1","command":"ls -l","conExpr":"* * * * *"},"PlanTime":"理论调度时间点","RealTime":"时间调度时间点","CancelCtx":"ctx上下文","CancelFunc":"ctx cancel函数"}}
func (executor *Executor) ExecuteJob(info *common.JobExecuteInfo) {
	go func() {

		// 初始化执行结果
		result := &common.JobExecuteResult{
			ExecuteInfo: info,
			Output: make([]byte, 0),
		}

		// 基于job名称，初始化锁
		jobLock := G_jobLock.CreateJobLock(info.Job.Name)

		// 记录任务开始时间
		result.StartTime = time.Now()

		// 上锁
		// 随机睡眠(0~1s)
		time.Sleep(time.Duration(rand.Intn(1000)) * time.Millisecond)

		// 给jobLock上锁
		err := jobLock.TryLock()

		// 任务执行完毕后，释放锁
		defer jobLock.Unlock()

		if err != nil { // 上锁失败
			result.Err = err
			result.EndTime = time.Now()

		} else {

			// 上锁成功后，重置任务启动时间
			result.StartTime = time.Now()

			// 执行shell命令
			cmd := exec.CommandContext(info.CancelCtx, "/bin/bash", "-c", info.Job.Command)

			// 执行并捕获输出
			output, err := cmd.CombinedOutput()

			// 记录任务结束时间
			result.EndTime = time.Now()
			result.Output = output
			result.Err = err

		}

		// 任务执行完成后，把执行的结果返回给Scheduler，Scheduler会从executingTable中删除掉执行记录，并且如果是没有成功获取到所的结果，在scheduler里面会过滤掉，只会写入获取到锁的执行结果
		G_scheduler.PushJobResult(result)

	}()
}

 * @File name: joblock.go

package worker

import (
	"context"
	"errors"
	"go-task/common"
	clientv3 "go.etcd.io/etcd/client/v3"
	"log"
)


var G_jobLock *JobLock

// JobLock 定义锁的结构
type JobLock struct {
	// etcd客户端
	kv      clientv3.KV // kv对象
	lease   clientv3.Lease // 租约对象
	jobName string // 任务名

	cancelFunc context.CancelFunc // 用于终止自动续租
	leaseId    clientv3.LeaseID   // 租约ID
	isLocked   bool               // 是否上锁成功
}

// InitJobLock 初始化一把锁
func InitJobLock(jobName string, kv clientv3.KV, lease clientv3.Lease) (jobLock *JobLock) {
	jobLock = &JobLock{
		kv:      kv,
		lease:   lease,
		jobName: jobName,
	}
	G_jobLock = jobLock
	return
}

// CreateJobLock 在executor中被调用，初始化一把锁
func (j *JobLock) CreateJobLock(jobName string) (jobLock *JobLock) {
	jobLock = InitJobLock(jobName, j.kv, j.lease)
	return
}

// RemoveLease 取消租约，让/cron/lock/目录下的key失效(相当于删除吧), 此函数为TryLock服务
func RemoveLease(cancelFunc context.CancelFunc, jobLock *JobLock, leaseId clientv3.LeaseID) error {
	cancelFunc()                                  // 取消自动续租
	_, err := jobLock.lease.Revoke(context.TODO(), leaseId) //  释放租约
	return err
}

// TryLock 尝试给任务上锁，上锁的标准是可以在/cron/lock/目录下创建key
func (j *JobLock) TryLock() (err error) {

	// 创建租约(5秒)
	leaseGrantResp, err := j.lease.Grant(context.TODO(), 5)
	if err != nil {
		return
	}

	// 创建上下文
	cancelCtx, cancelFunc := context.WithCancel(context.TODO())

	// 租约ID
	leaseId := leaseGrantResp.ID

	// 自动续租
	keepRespChan, err := j.lease.KeepAlive(cancelCtx, leaseId)
	if err != nil {
		// 释放租约
		errLease := RemoveLease(cancelFunc, j, leaseId)
		if errLease != nil {
			log.Printf(err.Error())
		}
	}

	// 处理续租应答的协程, 没有什么用，主要是起到一个提醒作用
	go func() {
		for {
			select {
			case keepResp := <-keepRespChan: // 自动续租的应答
				if keepResp == nil {
					log.Printf("租约被取消")
					goto END
				}
			}
		}
	END:
	}()

	// 创建事务txn
	txn := j.kv.Txn(context.TODO())

	// 锁路径
	lockKey := common.JOB_LOCK_DIR + j.jobName

	// 5, 事务抢锁
	txn.If(clientv3.Compare(clientv3.CreateRevision(lockKey), "=", 0)).
		Then(clientv3.OpPut(lockKey, "", clientv3.WithLease(leaseId))).
		Else(clientv3.OpGet(lockKey))

	// 提交事务
	txnResp, err := txn.Commit()
	if err != nil {
		errLease := RemoveLease(cancelFunc, j, leaseId)
		if errLease != nil {
			log.Printf(errLease.Error())
		}
	}

	// 成功返回, 失败释放租约
	if !txnResp.Succeeded { // 锁被占用
		err = errors.New("锁已被占用")
		errLease := RemoveLease(cancelFunc, j, leaseId) // 取消租约
		if errLease != nil {
			log.Printf(errLease.Error())
		}
	}

	// 抢锁成功，说明可以写入key
	j.leaseId = leaseId
	j.cancelFunc = cancelFunc
	j.isLocked = true
	return

}

// Unlock 释放锁
func (j *JobLock) Unlock() {
	if j.isLocked {
		errLease := RemoveLease(j.cancelFunc, j, j.leaseId)
		if errLease != nil {
			log.Printf(errLease.Error())
		}
	}
}

package worker

import (
	"context"
	"go-task/common"
	"go.mongodb.org/mongo-driver/mongo"
	"go.mongodb.org/mongo-driver/mongo/options"
	"time"
)

var G_logSink *LogSink

// LogBatch
type LogBatch struct {
	Logs []interface{}
}

// LogSink 将任务执行日志写入到MongoDB
type LogSink struct {
	client         *mongo.Client
	logCollection  *mongo.Collection
	logChan        chan *common.JobLog
	autoCommitChan chan *LogBatch // 日志批量自动提交
}

// saveLogs 批量写入log日志到MongoDB中，此方法为writeLoop服务
func (l *LogSink) saveLogs(batch *LogBatch) {
	l.logCollection.InsertMany(context.TODO(), batch.Logs)
}

// writeLoop 将日志写入到MongoDB协程
func (l *LogSink) writeLoop() {

	var (
		logBatch    *LogBatch
		commitTimer *time.Timer
	)

	// 不断的从logchan队列里面获取日志
	for {
		select {

		case log := <- l.logChan:

			// 过期的匹日志，写入到MongoDB后，会设置logBatch为nil
			if logBatch == nil {

				logBatch = &LogBatch{}

				// 让这个批次超时自动提交(给1秒的时间）
				// 这是一个闭包，内部函数innerFunc获取外部函数timeoutFunc的自由变量batch, 这个timeoutFunc就是innerFunc
				timeoutFunc := func(logBatch *LogBatch) func() {

					innerFunc := func() {
						// 将匹日志添加到匹日志的channel
						l.autoCommitChan <- logBatch
					}
					return innerFunc

				}(logBatch)

				// 创建定时器的目的是，避免在日志少的时候，日志数量很长时间才会达到100条，使得用户虽然发送了命令，但是无法获取到执行结果，因此创建一个定时器
				// 创建一个1秒执行一次的定时器, AfterFunc接收一个函数对象
				commitTimer = time.AfterFunc(time.Duration(G_config.JobLogCommitTimeout)*time.Millisecond, timeoutFunc)
			}

			// 将新的日志加入到批处理日志中，
			logBatch.Logs = append(logBatch.Logs, log)

			// 如果批日志满了, 就立即写入到MongoDB中，配置文件中定义100条就满了
			if len(logBatch.Logs) >= G_config.JobLogBatchSize {
				// 写入MongoDB
				l.saveLogs(logBatch)
				// 清空logBatch
				logBatch = nil
				// 取消定时器
				commitTimer.Stop()
			}

		// 这里是获取过期的批量日志
		case timeoutBatch := <- l.autoCommitChan:
			// 这个有一个关键的时间点，就是当logBatch的日志条数为100条的时候，那么写入MongoDB，但是此时再停止定时器，可能已经来不及了，可能这个批次已经提交到了autoCommitChan，如果这个匹日志正好是满足100条写入的匹日志，那么直接continue
			if timeoutBatch != logBatch {
				continue // 跳过已经写入到MongoDB的日志
			}
			// 如果是过期的，不管是不是满了，都写入到MongoDB
			l.saveLogs(timeoutBatch)
			// 清空logBatch
			logBatch = nil
		}
	}
}

// InitLogSink 初始化日志写入MongoDB
func InitLogSink() (err error) {

	// 建立mongodb连接
	clientOptions := options.Client().ApplyURI(G_config.MongodbUri).SetConnectTimeout(time.Duration(G_config.MongodbConnectTimeout) * time.Millisecond)
	client, err := mongo.Connect(context.TODO(), clientOptions)
	if err != nil {
		return
	}

	// 选择db和collection
	G_logSink = &LogSink{
		client:        client,
		logCollection: client.Database("cron").Collection("log"),
		// logChan: 日志通道，
		logChan: make(chan *common.JobLog, 1000),
		// autoCommitChan 自动提交的通道，这是干嘛的
		autoCommitChan: make(chan *LogBatch, 1000),
	}

	// 启动一个mongodb处理协程
	go G_logSink.writeLoop()

	return

}

 * @Author: Robby
 * @File name: register.go
 * @Create date: 2021-08-17
 * @Function: 当有worker启动的时候，先将自己的IP写到etcd的/cron/workers/目录下，例如 /cron/workers/192.168.100.111，并且这个key是自动续租的 【完毕】

package worker

import (
	"context"
	"go-task/common"
	clientv3 "go.etcd.io/etcd/client/v3"
	"log"
	"net"
	"time"
)

var G_register *Register

// Register 注册节点到etcd： /cron/workers/IP地址
type Register struct {
	client *clientv3.Client
	kv clientv3.KV
	lease clientv3.Lease
	localIP string // 本机IP
}

// 获取本机网卡IP
func getLocalIP() (ipv4 string, err error) {

	// 获取所有网卡
	addrs, err := net.InterfaceAddrs()
	if err != nil {
		log.Panicf(err.Error())
		return
	}

	// 取第一个非lo的网卡IP
	for _, addr := range addrs {
		// 这个网络地址是IP地址: ipv4, ipv6
		ipNet, isIpNet := addr.(*net.IPNet)

		if isIpNet && !ipNet.IP.IsLoopback() {
			// 跳过IPV6
			if ipNet.IP.To4() != nil {
				ipv4 = ipNet.IP.String()	// 192.168.1.1
				return
			}
		}
	}

	err = common.ERR_NO_LOCAL_IP_FOUND
	return
}

// InitRegister 初始化worker注册
func InitRegister() (err error) {


	// 初始化配置
	config := clientv3.Config{
		Endpoints: G_config.EtcdEndpoints, // 集群地址
		DialTimeout: time.Duration(G_config.EtcdDialTimeout) * time.Millisecond, // 连接超时
	}

	// 建立连接
	client, err := clientv3.New(config)
	if err != nil {
		log.Panicf(err.Error())
		return
	}

	// 本机IP
	localIp, err := getLocalIP()
	if err != nil {
		log.Panicf(err.Error())
		return
	}

	// 得到KV和Lease的API子集
	kv := clientv3.NewKV(client)
	lease := clientv3.NewLease(client)

	G_register = &Register{
		client: client,
		kv: kv,
		lease: lease,
		localIP: localIp,
	}

	// 服务注册
	go G_register.keepOnline()

	return
}

// CancelContext 调用context的cancel函数
func CancelContext(cancelFunc context.CancelFunc)  {
	time.Sleep(1 * time.Second)
	if cancelFunc != nil {
		cancelFunc()
	}
}

// 注册到/cron/workers/IP, 并自动续租
func (r *Register) keepOnline() {

	var cancelFunc context.CancelFunc

	for {

		// 获取etcd的注册路径
		regKey := common.JOB_WORKER_DIR + r.localIP

		// 创建租约
		leaseGrantResp, err := r.lease.Grant(context.TODO(), 10)
		if err != nil {
			CancelContext(cancelFunc) // 取消自动续租
		}

		// 自动续租
		keepAliveChan, err := r.lease.KeepAlive(context.TODO(), leaseGrantResp.ID)
		if err != nil {
			CancelContext(cancelFunc) // 取消自动续租
		}

		cancelCtx, cancelFunc := context.WithCancel(context.TODO())

		// 注册到etcd
		_, err = r.kv.Put(cancelCtx, regKey, "", clientv3.WithLease(leaseGrantResp.ID))
		if err != nil {
			CancelContext(cancelFunc)
		}

		// 处理续租应答, 如果续租应答为nil，那么同样取消自动续租
		for {
			select {
			case keepAliveResp := <- keepAliveChan:
				if keepAliveResp == nil {	// 续租失败
					CancelContext(cancelFunc)  // 取消自动续租
				}
			}
		}

	}

}
