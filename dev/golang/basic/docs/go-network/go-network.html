    func process(conn net.Conn) {
	defer conn.Close() // 关闭连接

	for {
		reader := bufio.NewReader(conn)
		var buf [128]byte
		n, err := reader.Read(buf[:]) // 读取数据
		if err != nil {
			fmt.Println("read from client failed, err:", err)
			break
		}

		recvStr := string(buf[:n])

		fmt.Println("收到client端发来的数据：", recvStr)

		// 发送数据
		sendN, err := conn.Write([]byte(recvStr))
		if err != nil {
			fmt.Println("send to client failed, err:", err)
			break
		}
		fmt.Printf("发送的字节数: %d\n", sendN)
	}
}

func main() {
	// 监听端口
	listen, err := net.Listen("tcp", "127.0.0.1:20000")
	if err != nil {
		fmt.Println("listen failed, err:", err)
		return
	}

	for {
		conn, err := listen.Accept() // 建立连接
		if err != nil {
			fmt.Println("accept failed, err:", err)
			continue
		}
		go process(conn) // 启动一个goroutine处理连接
	}
}

    func main() {
	conn, err := net.Dial("tcp", "127.0.0.1:20000")
	if err != nil {
		fmt.Println("err :", err)
		return
	}
	defer conn.Close() // 关闭连接

	// 这里的reader是标准输入
	inputReader := bufio.NewReader(os.Stdin)

	for {
		input, _ := inputReader.ReadString('\n') // 读取用户输入
		inputInfo := strings.Trim(input, "\r\n")

		if strings.ToUpper(inputInfo) == "Q" { // 如果输入q就退出
			return
		}

		_, err = conn.Write([]byte(inputInfo)) // 发送数据
		if err != nil {
			return
		}

		var buf [1024]byte
		n, err := conn.Read(buf[:])

		if err != nil {
			fmt.Println("recv failed, err:", err)
			return
		}

		fmt.Println(string(buf[:n]))
	}
}

    # (二)：TCP粘包分析

    • 【粘包】可发生在发送端也可发生在接收端
    1：TCP的Nagle算法是一种改善网络传输效率的算法，当客户端提交一段数据给TCP发送到服务器端时，TCP并不立刻发送数据，而是等待一段时间，如果还有数据被提交到TCP中，会一起发送

2：当服务器端从TCP中获取到数据，会先存储到缓冲区中，然后通知应用层取数据，如果应用层取数据不及时，那么也会造成一次性取多个数据片段的情况

    2
    3

    # (三)：TCP粘包解决方案

    • 出现【粘包】的关键在于接收方不确定将要传输的数据包的大小，因此我们可以对数据包进行封包和拆包的操作
    封包：
	封包就是给一段数据加上包头，这样数据包就分为 包头、包体 两部分内容了，如果是要过来非法包，那么就需要加上包尾。根据包头存储的包体长度，就可以正确的拆分出一个完整的数据包，虽然当前大包中可能粘有多个包，但是只要获取到包的实际长度，就不会一次性取多个包的数据

解包：
	解包就是获取一段数据的真实长度信息，获取当前包的真实数据

    // Encode 封包
func Encode(message string) ([]byte, error) {

	// 读取消息的长度，转换成int32类型（int32类型占4个字节）
	var length = int32(len(message))

	var pkg = new(bytes.Buffer)

	// 写入消息头
	// binary.LittleEndian 使用小端字节序
	err := binary.Write(pkg, binary.LittleEndian, length)
	if err != nil {
		return nil, err
	}

	// 写入消息实体
	err = binary.Write(pkg, binary.LittleEndian, []byte(message))
	if err != nil {
		return nil, err
	}

	// 返回缓冲区中所有的字节切片
	return pkg.Bytes(), nil

}

// Decode 解包
func Decode(reader *bufio.Reader) (string, error) {

	// 读取包的长度
	lengthByte, _ := reader.Peek(4) // 读取前4个字节的数据

	lengthBuff := bytes.NewBuffer(lengthByte)

	// 使用4字节变量获取包的长度
	var length int32

	// 获取到包实际的长度
	err := binary.Read(lengthBuff, binary.LittleEndian, &length)
	if err != nil {
		return "", err
	}

	// Buffered返回缓冲中现有的可读取的字节数，如果包的长度+4个存储长度的字节的总长度 大于 当前缓冲区所有字节长度，说明当前缓冲区是无效包
	if int32(reader.Buffered()) < length+4 {
		return "", err
	}

	// 开辟一块数组，长度为包实际长度+4字节
	pack := make([]byte, int(4+length))

	_, err = reader.Read(pack)
	if err != nil {
		return "", err
	}

	// 返回消息内容，前面4个字节是长度
	return string(pack[4:]), nil
}

	defer conn.Close()

	reader := bufio.NewReader(conn)
	for {

		msg, err := proto.Decode(reader)
		// 如果是io.EOF，说明读取完毕
		if err == io.EOF {
			return
		}

    # 三：简单的UDP服务端和客户端实现

    # (一)：服务器端代码

    func main() {

	listen, err := net.ListenUDP("udp", &net.UDPAddr{
		IP:   net.IPv4(127, 0, 0, 1),
		Port: 30000,
	})

	if err != nil {
		fmt.Println("listen failed, err:", err)
		return
	}

	defer listen.Close()


	for {
		var data [1024]byte
		n, addr, err := listen.ReadFromUDP(data[:]) // 接收数据
		if err != nil {
			fmt.Println("read udp failed, err:", err)
			continue
		}

		fmt.Printf("data:%v, addr:%v, count:%v\n", string(data[:n]), addr, n)

		_, err = listen.WriteToUDP(data[:n], addr) // 发送数据

		if err != nil {
			fmt.Println("write to udp failed, err:", err)
			continue
		}
	}

}

    # (二)：客户端端代码

    // UDP 客户端
func main() {

	socket, err := net.DialUDP("udp", nil, &net.UDPAddr{
		IP:   net.IPv4(127, 0, 0, 1),
		Port: 30000,
	})
	if err != nil {
		fmt.Println("连接服务端失败，err:", err)
		return
	}
	defer socket.Close()

	sendData := []byte("你好，我是Robby")

	_, err = socket.Write(sendData) // 发送数据

	if err != nil {
		fmt.Println("发送数据失败，err:", err)
		return
	}

	data := make([]byte, 1024)
	n, remoteAddr, err := socket.ReadFromUDP(data) // 接收数据

	if err != nil {
		fmt.Println("接收数据失败，err:", err)
		return
	}

	fmt.Printf("recv:%v, addr:%v, count:%v\n", string(data[:n]), remoteAddr, n)
}
