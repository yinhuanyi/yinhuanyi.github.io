// Conn 数据连接，把数据库连接对象的创建放在这里是错误的，会导致每次连接数据库，都会创建一个连接请求，
func (p *ProductManager) Conn() (err error) {
	if p.mysqlConn == nil {
		mysql, err := common.NewMysqlConn()
		if err != nil {
			return err
		}
		p.mysqlConn = mysql
	}
	if p.table == "" {
		p.table = "product"
	}
	return
}

//Insert 插入
func (p *ProductManager) Insert(product *datamodels.Product) (productId int64, err error) {

	//1.判断连接是否存在
	if err = p.Conn(); err != nil {
		return
	}

	//2.准备sql
	//sql := "INSERT product SET productName=?,productNum=?,productImage=?,productUrl=?"
	sql := `INSERT INTO product(productName, productNum, productImage, productUrl) values (?,?,?,?)`
	stmt, errSql := p.mysqlConn.Prepare(sql)
	if errSql != nil {
		return 0, errSql
	}

	//3.传入参数
	result, errStmt := stmt.Exec(product.ProductName, product.ProductNum, product.ProductImage, product.ProductUrl)
	if errStmt != nil {
		return 0, errStmt
	}
	// 返回ID
	return result.LastInsertId()

}

	dec := common.NewDecoder(&common.DecoderOptions{TagName: "ipfsmain"}) //
	if err := dec.Decode(p.Ctx.Request().Form, product); err != nil {

	}
	err := p.ProductService.UpdateProduct(product)
	if err != nil {
		p.Ctx.Application().Logger().Debug(err)
	}
	p.Ctx.Redirect("/product/all")
}

    # (二) 将用户id使用AES进行加密，作为cookie信息(这个在生产环境不能这样使用，一般是使用jwt登录，token中可以存储uid、role等信息)

    • AES对称加密，使用CBC模式
    import (
	"bytes"
	"crypto/aes"
	"crypto/cipher"
	"encoding/base64"
	"errors"
	"fmt"
	"log"
)

// PwdKey 16，24，32为字符，分别对应AES-128、AES-192, AES-256加密方法， PwdKey是不能泄露的
var PwdKey = []byte("DIS**#KKKDJJSKDI")

// PKCS7Padding 字节切片填充
func PKCS7Padding(ciphertext []byte, blocksize int) []byte {
	// 计算padding的长度，也就是填充的长度
	padding := blocksize - len(ciphertext)%blocksize
	// byte(padding)是一个字节，[]byte{byte(padding)}是一个字节切片，且byte(padding)是切片的一个元素, padtext是合成的新的字节切片，这个切片里面有padding个字节
	padtext := bytes.Repeat([]byte{byte(padding)}, padding)
	fmt.Printf("padding=%d\n", padding)
	return append(ciphertext, padtext...)
}

// UnPKCS7Padding 删除字节切片的填充
func UnPKCS7Padding(origData []byte) ([]byte, error) {
	length := len(origData)
	if length == 0 {
		return nil, errors.New("加密字符串错误！")
	}else {
		// 获取填充字节的长度, 由于填充的时候，是用padding长度作为byte填充的，因此只需要取出最后一个元素就知道填充的长度了
		unpadding := int(origData[length-1])
		fmt.Printf("padding=%d\n", unpadding)
		// 截取切片，删除填充字节，并且返回明文
		return origData[:(length-unpadding)], nil
	}
}

// AESEncript 加密
func AESEncript(origData, key []byte) ([]byte, error) {
	// 1 NewCipher 创建并返回一个新的 cipher.Block。key 参数应该是 AES 密钥，16、24 或 32 字节可供选择, AES-128、AES-192 或 AES-256。
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, err
	}
	// 返回key的长度
	blockSize := block.BlockSize()
	// 对origdata进行填充，满足长度需求
	origData = PKCS7Padding(origData, blockSize)
	// 采用CAC加密模式, blockmod
	blockMod := cipher.NewCBCEncrypter(block, key[:blockSize])
	// 创建空加密切片
	crypted := make([]byte, len(origData))
	// 执行加密
	blockMod.CryptBlocks(crypted, origData)
	return crypted, nil
}

// AESDecrypt 解密
func AESDecrypt(cripted, key []byte) ([]byte, error) {
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, err
	}
	blockSize := block.BlockSize()
	blockMod := cipher.NewCBCDecrypter(block, key[:blockSize])
	// 创建空的解密字节切片
	origData := make([]byte, len(cripted))
	blockMod.CryptBlocks(origData, cripted)
	// 去除填充字符串
	origData, err = UnPKCS7Padding(origData)
	if err != nil {
		return nil, err
	}
	return origData, nil
}

// EnByteToBase 将字节切片转换为base64字符串
func EnByteToBase(pwd []byte) (string, error) {
	// 先对密码加密
	result, err := AESEncript(pwd, PwdKey)

	fmt.Printf("result=%b\n", result)
	if err != nil {
		return "", err
	}
	// 再转换为base64
	return base64.StdEncoding.EncodeToString(result), nil
}

// DeBaseToByte 将base64字符串转换为字节切片
func DeBaseToByte(pwd string) ([]byte, error)  {
	pwdByte, err := base64.StdEncoding.DecodeString(pwd)
	if err != nil {
		return nil, err
	}
	// 进行AES解密
	return AESDecrypt(pwdByte, PwdKey)
}

func main() {

	pwdRow := []byte("yhy3426356")
	pwdStr, err := EnByteToBase(pwdRow)
	if err != nil {
		log.Fatalf(err.Error())
	}
	fmt.Printf("加密后的字符串为：%s\n", pwdStr)

	pwdRow, err = DeBaseToByte(pwdStr)
	if err != nil {
		log.Fatalf(err.Error())
	}
	fmt.Printf("解密后的字符串为：%s\n", string(pwdRow))

}

    # (三) 对比cookie、session和jwt的用户认证方式

    • 1：cookie认证的方式
    	用户登录浏览器的时候，会获取用户的uid，然后将过期时间和uid一起进行对称加密后进行base64编码，将加密信息写入到浏览器的cookie中，例如【sign、adfkaldkfadfasdfj、yinhuanyi.com、\】
	当用户访问后台的时候，此时获取sign对应的value，也就是adfkaldkfadfasdfj，先base64解码再对称解密，然后获取用户的uid和过期时间，如果cookie过期，直接跳转到登录页面。如果没有过期返回数据给用户
    1
    2
    • 2：session认证的方式

    	用户登录的时候，系统会查询用户是否存在，如果用户存在，基于用户的信息，生成一个session记录到数据库中，然后将sessionid写入到用户的cookie中
	cookie写入到浏览器的Application对应的域名下面，一般会有name、value、domain、path、expires/max-age这几个字段。如果是session会话，那么name=sessionid，value=2ia4ehcsfc4466xkcnt7h7vohlmn946l，当用户访问后台的时候，后台获取sessionid的value值，也就是2ia4ehcsfc4466xkcnt7h7vohlmn946l，在Django的django_session表中查询，获取到session_data和expire_date，如果session没有过期，然后将session_data解密，就可以获取到用户的信息了
    1
    2

    • 3：jwt认证的方式

    	JWT中Header和Payload在服务器端会被Base64编码，编码后的结果加上服务器端secret，再借助HS256算法获得Signature，则JWT = Header + Payload + Signature，这里的Signature=HS256(Header,Payload,服务器的secret)。服务器将JWT返回给客户端，保存到浏览器中，当浏览器再次请求，服务器端只需要再次借助HS256算法基于用户的Header,Payload和服务器的secret获得新的Signature，与客户端JWT的Signature对比，如果一样，说明客户端用户认证合法，再取出Payload中有个用户的非敏感信息，对用户进行数据展示。如果不一样，说明用户非法，直接返回错误信息
    JWT同时也减轻了服务器端session的存储压力和session的同步压力

	jwt的算法实现思路如下：
	①：当用户登录时候，将用户传递的密码md5加密和数据库中加密的md5密码对比，如果相等，说明用户登录正确，再获取用户的uid和username等字段，再结合过期时间、签发人等信息，创建一个claim, 对claim进行sh256加密，获取token，token再结合secret，获取jwt

	②：当用户携带jwt访问服务器某个接口的时候，jwt的token一般会在HTTP的header中，那么header的name可以自己定义，其格式可以是 Authorization: 'Bearer Token'这种格式，或者Token: 'Token'这种格式，因此这个需要与前端定义好。Bearer是Authorization的一种标准schema。当服务器获取Authorization的value后，需要切割字符串，获取Token的值，然后开始解析Token，也是调用解析函数，如果判断jwt的没有超时，且校验正确，就会得到一个claim的结构体，从claim的结构体中可以获取到用户的uid和username，并且可以将这两个字段封装到gin的context上下文中，以便于handler处理函数中可以获取到这个两个字段的值，就知道是哪个用户登录了系统

    # (四) Hash环和一致性Hash原理

    • Memcached的伪分布式
    	我们知道Memcached的分布式其实是一种“伪分布式”，也就是它的服务器结点之间其实是相互无关联的，之间没有网络拓扑关系，由客户端来决定一个key是存放到哪台机器。
	
	具体来讲，假设我有多台memcached服务器，编号分别为m0,m1,m2,…。对于一个key，由客户端来决定存放到哪台机器，那最简单的hash公式就是 key % N，其中N是机器的总数。

	但这有个问题，一旦机器数变少，或者增加机器，N发生变化，那之前存放的数据就全部无效了。因为你按照新的N值取模计算出的机器编号，和当时按旧的N值取模算出的机器编号肯定是不等的，也就意味着绝大部分缓存会失效。

	这个问题的解决办法就是用1种特别的Hash函数，尽可能使得，增加机器/减少机器时，缓存失效的数目降到最低，这就是Hash环，或者叫一致性Hash。

    • Hash环的基本原理

    上面说的Hash函数，只经过了1次hash，即把key hash到对应的机器编号。

Hash环有2次Hash：
（1）把所有机器编号hash到这个环上
（2）把key也hash到这个环上。然后在这个环上进行匹配，看这个key和哪台机器匹配。



具体来讲，如下：
假定有这样一个Hash函数，其值空间为（0到2的32次方-1) ，也就是说，其hash值是个32位无整型数字 ，这些数字组成一个环。

然后，先对机器进行hash(比如根据机器的ip)，算出每台机器在这个环上的位置； 再对key进行hash，算出该key在环上的位置，然后从这个位置往前走，遇到的第一台机器就是该key对应的机器，就把该(key, value) 存储到该机器上。

首先计算出每台Cache服务器在环上的位置（图中的大圆圈）；然后每来一个(key, value)，计算出在环上的位置（图中的小圆圈），然后顺时针走，遇到的第1个机器，就是其要存储的机器。

这里的关键点是：当你增加/减少机器时，其他机器在环上的位置并不会发生改变。这样只有增加的那台机器、或者减少的那台机器附近的数据会失效，其他机器上的数据都还是有效的。

    • 数据倾斜问题
    当你机器不多的时候，很可能出现几台机器在环上面贴的很近，不是在环上均匀分布。这将会导致大部分数据，都会集中在某1台机器上。

为了解决这个问题，可以引入“虚拟机器”的概念，也就是说：1台机器，我在环上面计算出多个位置。怎么弄呢？ 假设用机器的ip来hash，我可以在ip后面加上几个编号, ip_1, ip_2, ip_3, … 把1台物理机器生个多个虚拟机器的编号。

数据首先映射到“虚拟机器上”，再从“虚拟机器”映射到物理机器上。因为虚拟机器可以很多，在环上面均匀分布，从而保证数据均匀分布到物理机器上面。

    package common

import (
	"errors"
	"hash/crc32"
	"sort"
	"strconv"
	"sync"
)

// 自定义切片类型，目的是让自定义类型实现sort.Sort的接口方法
type uints []uint32

// Len 返回切片长度
func (u uints) Len() int {
	return len(u)
}

// Less 比较切片中两个数大小(通过索引位置)
func (u uints) Less(i, j int) bool {
	return u[i] < u[j]
}

// Swap 交换切片中的两个数的位置（通过索引位置）
func (u uints) Swap(i, j int) {
	u[i], u[j] = u[j], u[i]
}

// Consistent 定义一个保存一致性hash信息的结构体
type Consistent struct {
	// hash集合, key是hash值，值存放节点的信息
	circle map[uint32]string
	// 已经排序的节点hash切片，利用切片有序特性，将hash集合逻辑化为环状结构
	sortedHashes uints
	// 虚拟节点个数，用于添加hash的平衡发布, 一个物理节点对应几个虚拟节点
	VituralNode int
	// map读写锁， 这个是一个继承，Consistent结构体继承了sync.RWMutex结构体，可以调用sync.RWMutex的所有方法
	sync.RWMutex
}

// NewConsistent 构造函数
func NewConsistent() *Consistent { 
	return &Consistent{
		circle:      make(map[uint32]string),
		VituralNode: 20,
	}
}

// generateKey 返回ip+i，例如127.0.0.11
func (c *Consistent) generateKey(element string, index int) string { 
	return element + strconv.Itoa(index)
}

// hashKey 计算字符串的hash值
func (c *Consistent) hashKey(key string) uint32 { 
	// 凑一个64位数组
	if len(key) < 64 {

		srcatch := make([]byte, 64)
		copy(srcatch, key)  // 由于key是string类型，string的底层，第一个是字符串指向的底层字节数组，第二个是字节数组的长度
		// 使用crc32计算校验和
		return crc32.ChecksumIEEE(srcatch[:len(key)])
	}

	// 只要这个一句就可以了，上面的代码是多此一举
	return crc32.ChecksumIEEE([]byte(key))
}

// updateSortedHashes 更新切片排序，每次新增物理节点或删除物理节点都需调用这个方法，这个方法是让所有的circle的key切片重新排序
func (c *Consistent) updateSortedHashes() { 
	// 获取一个空的切片，由于切片是指针类型，那么hashes与sortedHashes其实是指向同一个底层数组
	hashes := c.sortedHashes[:0]
	// 判断hashes是否过大，过大则重置
	// 判断hashes是否过大，过大则重置，重置的话，底层数组查询分配内存
	// todo: 这个是啥意思，切片的容量
	if cap(c.sortedHashes) / (c.VituralNode*4) > len(c.circle) {
		hashes = nil
	}
	// 将circle集合中节点key，加入hashed中, 将key进行排序
	for key := range c.circle {
		hashes = append(hashes, key)
	}
	// 对key的切片进行排序
	sort.Sort(hashes)
	// 将排序好的key切片，赋值，这里直接赋值不是就覆盖了吗？这里就是覆盖，因为每次Add或者Delete的时候，那么c.circle这个map中保留的是所有物理节点的hash数据，所以每次都应该让circle的key重新排序
	c.sortedHashes = hashes
}

// Add 像hash环中添加节点，注意map线程不安全，   这里的element是IP， i是虚拟节点的索引，
func (c *Consistent) Add(element string) {  
	c.Lock()
	defer c.Unlock()

	// 一个物理节点对应20个虚拟节点
	for i := 0; i < c.VituralNode; i++ {
		// 根据生成的节点添加到hash环中， c.generateKey(element, i) == 127.0.0.01， c.hashKey(c.generateKey(element, i)) == hash指纹
		// c.circle的结构=={"asldjflla": "127.0.0.1", "asldjfll2": "127.0.0.1", }
		c.circle[c.hashKey(c.generateKey(element, i))] = element
	}
	// 让hash切片排序
	c.updateSortedHashes()
}

// Remove 线程安全的删除节点
func (c *Consistent) Remove(element string) { 
	c.Lock()
	defer c.Unlock()
	for i := 0; i < c.VituralNode; i++ {
		delete(c.circle, c.hashKey(c.generateKey(element, i)))
	}
	// 删完要更新排序
	c.updateSortedHashes()
}

// Get 根据name这个hash值，获取服务器信息，name是uid
func (c *Consistent) Get(name string) (string, error) {  
	c.Lock()
	defer c.Unlock()

	if len(c.circle) == 0 {
		return "", errors.New("Hash 环上没有数据")
	}

	// 传入name计算的hash值，通过hash值，获取大于此hash值的最小值索引
	index := c.search(c.hashKey(name))

	// 根据返回的最小值索引，获取切片中索引对应的hash，再基于hash获取到哈希环上的hash值，也就是服务器IP
	return c.circle[c.sortedHashes[index]], nil
}

//search 这里的key是name计算的hash值，大于这个hash值的最小值的索引
func (c *Consistent) search(key uint32) int {

	f := func(i int) bool {
		return c.sortedHashes[i] >= key
	}

	// 使用"二分查找"算法来搜索指定切片满足条件的最小值
	// sort.Search是二分法查找， 找到[0, n)区间内最小的满足f(i)==true的值i
	i := sort.Search(len(c.sortedHashes), f)

	// 如果超出范围，设置i=0
	if i >= len(c.sortedHashes) {
		i = 0
	}

	return i
}

    • 这里comm.go中，DataToStructByTagSql函数就是应用了reflect反射，将map数据映射到结构体obj中，注意这里的map是map[string]string类型，也就是说map的value类型已经设定好了，因此在TypeConversion函数中，只需要将字符串向其他的类型转换，例如string转int64等

    • 下面是DataToStructByTagSql函数和TypeConversion函数，TypeConversion函数是为DataToStructByTagSql函数服务的。这里的技术点就是reflect反射的应用和string类型转换，这是标签应用的底层原理

    package common

import (
	"errors"
	"reflect"
	"strconv"
	"time"
)

//DataToStructByTagSql 基于sql标签，将map类型映射到结构体
func DataToStructByTagSql(data map[string]string, obj interface{}) {

	// 结构体
	objValue := reflect.ValueOf(obj).Elem()  // reflect.ValueOf(obj)获取接口的类型值，.Elem()是获取指针对应的值
	// objValue.NumField()获取到结构体的成员数量
	for i := 0; i < objValue.NumField(); i++ {
		/*
		这里以结构体Product的第一个属性为例： Id int64 `json:"id" sql:"id"`
		data := map[string]string{"id": "1"}
		*/

		// objValue.Type().Field(i).Tag.Get("sql")为id
		tagKey := objValue.Type().Field(i).Tag.Get("sql")
		// data["id"]就是"1"，类型是string
		mapValue := data[tagKey]
		// 结构体名称是Id
		objFiledName := objValue.Type().Field(i).Name
		// 结构体名称是Id对应的类型: int64
		objFieldType := objValue.Field(i).Type()
		// 获取原始值 "1", 类型是Value
		reflectVal := reflect.ValueOf(mapValue)

		var err error

		// objFieldType:int64， var.Type()为:string
		if objFieldType != reflectVal.Type() { // 判断类型是否相等，
			// 将"1"转换为int64类型
			reflectVal, err = TypeConversion(mapValue, objFieldType.Name())
			if err != nil {
				panic(err)
			}
		}

		// 给objValue结构体对应的字段设置值
		objValue.FieldByName(objFiledName).Set(reflectVal)
	}
}

/*
TypeConversion 类型转换， 将value对应的string类型，转换为ntype类型
满足将string类型的值转换为time.Time、Time、int、int8、int32、int64、float32、float64类型
*/
func TypeConversion(value string, ntype string) (reflect.Value, error) {
	if ntype == "string" {
		return reflect.ValueOf(value), nil
	} else if ntype == "time.Time" {
		t, err := time.ParseInLocation("2006-01-02 15:04:05", value, time.Local)
		return reflect.ValueOf(t), err
	} else if ntype == "Time" {

		return reflect.ValueOf(t), err
	} else if ntype == "int" {
		i, err := strconv.Atoi(value)
		return reflect.ValueOf(i), err
	} else if ntype == "int8" {
		i, err := strconv.ParseInt(value, 10, 64)
		return reflect.ValueOf(int8(i)), err
	} else if ntype == "int32" {
		i, err := strconv.ParseInt(value, 10, 64)
		return reflect.ValueOf(int32(i)), err
	} else if ntype == "int64" {
		i, err := strconv.ParseInt(value, 10, 64)
		return reflect.ValueOf(i), err
	} else if ntype == "float32" {
		i, err := strconv.ParseFloat(value, 64)
		return reflect.ValueOf(float32(i)), err
	} else if ntype == "float64" {
		i, err := strconv.ParseFloat(value, 64)
		return reflect.ValueOf(i), err
	}
	//else if .......增加其他一些类型的转换

	return reflect.ValueOf(value), errors.New("未知的类型：" + ntype)
}

    # (七) golang中HTTP的拦截器，对原有handler函数进行功能增强

    • 这里只是举例一种增强函数的方法，举个例子：在http路由请求调用handler处理之前，对请求进行预处理，这种情况就是对handler进行了增强，就像写一个http拦截器一样。因此这里写了一个HTTP请求的拦截器实现了对原始HTTP handler函数的增强

    • HTTP拦截器代码如下：这里FilterHandle、WebHandle都是函数类型，函数类型和struct类型一样，使用type定义，定义FilterHandle的意义是：map的value是函数类型，定义WebHandle的意义是：真实的handler是函数类型，需要从参数传递进来，在增强数中被调用，因此是用于指定参数的类型

    package common


import (
	"net/http"
)

// FilterHandle 声明一个函数类型，这个函数类型是一个标准的http handler类型函数
type FilterHandle func(rw http.ResponseWriter, r *http.Request) error

// Filter 拦截器结构体
type Filter struct {
	// 用来存储要拦截的url
	filterMap map[string]FilterHandle
}

// NewFilter 构造函数实例化Filter结构体
func NewFilter() *Filter {
	return &Filter{filterMap: make(map[string]FilterHandle)}
}

// RegisterFilterUri 注册拦截器
func (f *Filter) RegisterFilterUri(uri string, handler FilterHandle) {
	f.filterMap[uri] = handler
}

// GetFilterHandle 根据uri获取对应handle
func (f *Filter) GetFilterHandle(uri string) FilterHandle {
	return f.filterMap[uri]
}

// WebHandle 定义一个新的函数类型
type WebHandle func(rw http.ResponseWriter, r *http.Request)

// handle方法，拦截器实例通过调用handle，传入一个函数，返回一个新函数。
// 确切的说，传入的函数被改装了，增加了拦截器实例内的路径和方法执行，如果请求的路径是一致的，则直接使用拦截器内的方法处理 ,如果路径不一致，则使用传入函数的逻辑处理

// Handle 这个有点点像Python的装饰器函数，也是接收一个函数，将函数进行增强，然后返回这个函数
func (f *Filter) Handle(webHandle WebHandle) func(rw http.ResponseWriter, r *http.Request) {
	return func(rw http.ResponseWriter, r *http.Request) {
		for path, handle := range f.filterMap {
			if path == r.RequestURI {   // 判断URL路径是否与request的路径是否相等, 如果相等，那么使用已经存储的函数执行，如果不相等使用WebHandle函数执行
				err := handle(rw, r)
				if err != nil {
					rw.Write([]byte(err.Error()))
					return
				}
				break
			}
		}
		// 执行正常的handler函数，就是请求处理函数
		webHandle(rw, r)
	}
}

    # (八) map类型的线程安全解决方法

    • 使用结构体封装map，且设置读写互斥锁属性
    // 把map属性包一层结构体
type AccessControl struct {
	// 这是缓存用户相关信息，这里才解释了一致性hash算法的意义
	sourceMap map[int]time.Time  // 这个map存储的是{"uid": 当前时间}
	sync.RWMutex
}

// 用户的访问时间信息初始化
var accessControl = &AccessControl{
	sourceMap: make(map[int]time.Time),
}


// 每次读写map中数据的时候，都加个锁，获取数据就加上读锁，
// GetNewRecord 根据uid获取用户的访问时间
func (m *AccessControl) GetNewRecord(uid int) time.Time {
	m.RLock() // 加个读锁
	defer m.RUnlock() // 解锁

	data := m.sourceMap[uid]
	return data
}

// SetNewRecord 根据uid设置访问时间
func (m *AccessControl) SetNewRecord(uid int) {
	m.Lock()
	defer m.Unlock()
	m.sourceMap[uid] = time.Now()  // 这是是设置时间
}

    # (九) 获取当前主机的IP地址

    • 获取当前主机的IP
    // GetIntranceIP 获取本机IP
func GetIntranceIP() (string, error) {
	addrs, err := net.InterfaceAddrs()
	if err != nil {
		return "", err
	}
	for _, address := range addrs {
		if ipnet, ok := address.(*net.IPNet); ok && ipnet.IP.IsGlobalUnicast() {
			if ipnet.IP.To4() != nil {
				return ipnet.IP.String(), nil
			}
		}
	}
	return "", errors.New("获取地址异常")
}
