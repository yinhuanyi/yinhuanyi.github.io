    # 一：对container/list源码进行分析

    # (一) 源码解析

    • 源码解析
    // Element 节点，有四个属性，list属性的目的基于节点可以获取到链表对象
type Element struct {
	next, prev *Element

	list *List

	Value interface{}
}

// Next 获取下一个节点
func (e *Element) Next() *Element {
	if p := e.next; e.list != nil && p != &e.list.root {
		return p
	}
	return nil
}

// Prev 获取前一个节点
func (e *Element) Prev() *Element {
	if p := e.prev; e.list != nil && p != &e.list.root {
		return p
	}
	return nil
}

// List 链表, 两个属性
type List struct {
	root Element // 这是一个虚拟节点
	len  int     // 链表长度
}

// Init 被New函数调用
func (l *List) Init() *List {
	l.root.next = &l.root // 都是nil
	l.root.prev = &l.root
	l.len = 0
	return l
}

// New 链表初始化
func New() *List {
	return new(List).Init()
}

// Len 获取链表的长度
func (l *List) Len() int { return l.len }

// Front 获取链表头部元素的后面一个元素
func (l *List) Front() *Element {
	if l.len == 0 {
		return nil
	}
	return l.root.next
}

// Back 获取链表头部元素的前面一个元素
func (l *List) Back() *Element {
	if l.len == 0 {
		return nil
	}
	return l.root.prev
}

// lazyInit 延时初始化
func (l *List) lazyInit() {
	if l.root.next == nil {
		l.Init()
	}
}

// insert 在at元素后面添加【元素e】
func (l *List) insert(e, at *Element) *Element {
	n := at.next
	at.next = e
	e.prev = at
	e.next = n
	n.prev = e
	e.list = l
	l.len++
	return e
}

// insertValue 在at元素后面添加【数据v】
func (l *List) insertValue(v interface{}, at *Element) *Element {
	return l.insert(&Element{Value: v}, at)
}

// remove 删除【元素e】
func (l *List) remove(e *Element) *Element {
	e.prev.next = e.next
	e.next.prev = e.prev
	e.next = nil
	e.prev = nil
	e.list = nil
	l.len--
	return e
}

// 将元素e移动到元素at的后面
func (l *List) move(e, at *Element) *Element {
	if e == at {
		return e
	}
	e.prev.next = e.next
	e.next.prev = e.prev

	n := at.next
	at.next = e
	e.prev = at
	e.next = n
	n.prev = e

	return e
}

// Remove 删除元素e
func (l *List) Remove(e *Element) interface{} {
	if e.list == l {
		l.remove(e)
	}
	return e.Value
}

// PushFront 在头部元素的【后面添加元素】
func (l *List) PushFront(v interface{}) *Element {
	l.lazyInit()
	return l.insertValue(v, &l.root)
}

// PushBack 在头部元素的【前面添加元素】
func (l *List) PushBack(v interface{}) *Element {
	l.lazyInit()
	return l.insertValue(v, l.root.prev)
}

// InsertBefore 在【mark元素】的【前面添加元素v】
func (l *List) InsertBefore(v interface{}, mark *Element) *Element {
	if mark.list != l {
		return nil
	}

	return l.insertValue(v, mark.prev)
}

// InsertAfter 在【mark元素】的【后面添加元素v】
func (l *List) InsertAfter(v interface{}, mark *Element) *Element {
	if mark.list != l {
		return nil
	}

	return l.insertValue(v, mark)
}

// MoveToFront 将元素e【移动到头部元素的后面】
func (l *List) MoveToFront(e *Element) {
	if e.list != l || l.root.next == e {
		return
	}

	l.move(e, &l.root)
}

// MoveToBack 将元素e【移动到头部元素的前面】
func (l *List) MoveToBack(e *Element) {
	if e.list != l || l.root.prev == e {
		return
	}

	l.move(e, l.root.prev)
}

// MoveBefore 将元素e【移动到mark元素的前面】
func (l *List) MoveBefore(e, mark *Element) {
	if e.list != l || e == mark || mark.list != l {
		return
	}
	l.move(e, mark.prev)
}

// MoveAfter 将元素e【移动到mark元素的后面】
func (l *List) MoveAfter(e, mark *Element) {
	if e.list != l || e == mark || mark.list != l {
		return
	}
	l.move(e, mark)
}

// PushBackList 将other这个链表插入到自己的前面
func (l *List) PushBackList(other *List) {
	l.lazyInit()
	for i, e := other.Len(), other.Front(); i > 0; i, e = i-1, e.Next() {
		l.insertValue(e.Value, l.root.prev)
	}
}

// PushFrontList 将other这个链表插入到自己的后面
func (l *List) PushFrontList(other *List) {
	l.lazyInit()
	for i, e := other.Len(), other.Back(); i > 0; i, e = i-1, e.Prev() {
		l.insertValue(e.Value, &l.root)
	}
}
