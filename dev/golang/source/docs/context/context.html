    # (二) cancelCtx是根Context

    • 有回溯树可知，基于emptyCtx，使用WithCancel()方法派生了根Context(黄色cancelCtx)，如果基于cancelCtx根再调用WithCancel()方法，派生出来的是子Context(红色cancelCtx)，所有的子Context都会被根Context添加到map对象中，也就是说，如果多次调用WithCancel()方法，都会被根Context添加到map对象中。
    func WithCancel(parent Context) (ctx Context, cancel CancelFunc) {
	c := newCancelCtx(parent) // 基于父context，初始化一个子cancelCtx

	// 这里必须保证c实现了cancel和Done方法
	propagateCancel(parent, &c) // 当parent调用了cancel方法，那么让所有的child也调用cancel方法

	// 将child实例对象和cancelFunc返回
	return &c, func() { c.cancel(true, Canceled) }
}


// propagateCancel // 当parent调用了cancel方法，那么让所有的child也调用cancel方法
func propagateCancel(parent Context, child canceler) {

	done := parent.Done() // 获取父Context的只读channel
	if done == nil {
		return
	}

	select {

	case <-done: // 如果父Context的只读channel关闭了，那么子context也必须关闭，并且把父context的err传递到子context

		child.cancel(false, parent.Err())
		return

	default:

	}

	// 判断parent是否是cancelCtx类型
	if p, ok := parentCancelCtx(parent); ok { // 如果parent是cancelCtx类型

		p.mu.Lock()

		if p.err != nil { // 如果parent有err错误，让子cancelCtx关闭channel
			child.cancel(false, p.err)

		} else { // 如果parent没有err错误
			if p.children == nil {  // 判断一下parent的children是不是空
				p.children = make(map[canceler]struct{}) // 如果是空，初始化一下
			}
			p.children[child] = struct{}{} // 将子cancelCtx加入到parent的children map中存储

		}

		p.mu.Unlock()

	} else { // 如果parent不是cancelCtx类型, 可能parent是自己实现的Context，才会启动goroutine,让child调用cancel方法，否则只是将child加入到children的map中

		atomic.AddInt32(&goroutines, +1) // 让goroutines+1

		go func() { // 开启一个goroutine，
			select {
			case <-parent.Done(): // 那么只要parent的channel关闭了，那么所有的child也应该调用cancel方法
				child.cancel(false, parent.Err())

			case <-child.Done(): // 如果是child的channel关闭了，那么什么也不做
			}
		}()
	}

}

    # (三) 当父Context(emptyCtx)调用了cancelFunc()函数会发生什么

    • 当完成本次api或rpc调用后，会执行defer cancelFunc()函数，此时会触发根cancelCtx调用cancel()方法，目的是关闭只读channel，让<-ctx.Done()返回struct{}{}，从而结束goroutine。在cancel()方法里面，根cancelCtx会遍历自己的children属性，获取所有的cancelCtx，所有的cancelCtx都调用一次cancel()方法，关闭自己的只读channel，让所有的子goroutine结束。根cancelCtx同时也会将children属性设置为nil，让GC回收子cancelCtx内存空间
    // 这里的c是根cancelCtx
c.cancel(true, Canceled)

// cancel cancel函数的功能是关闭channel
func (c *cancelCtx) cancel(removeFromParent bool, err error) {

	// 在关闭channel的时候，必须给一个err
	if err == nil {
		panic("context: internal error: missing cancel error")
	}

	c.mu.Lock()
	if c.err != nil {
		c.mu.Unlock()
		return
	}
	// 将错误赋值给context
	c.err = err

	if c.done == nil {  // 如果只读channel是nil，那么初始化一个空channel
		c.done = closedchan
	} else {
		close(c.done) // 这是函数的核心，关闭只读channel
	}

	// 这里是遍历canceler，只获取key
	for child := range c.children {
		// 让所有的canceler都调用自己的cancel方法，但是没有让canceler从父context中移除
		child.cancel(false, err)
	}

	// 所有的子canceler都关闭了channel后，就可以将children释放掉了
	c.children = nil

	c.mu.Unlock() // 再次解锁，避免死锁
	
	
	if removeFromParent {
		removeChild(c.Context, c)
	}
}

// 将child自己从parent中移除掉
func removeChild(parent Context, child canceler) {
	p, ok := parentCancelCtx(parent) // 先判断一下parent是否是cancelCtx类型
	if !ok { // 如果不是那么直接return，不需要移除
		return
	}
	p.mu.Lock()
	if p.children != nil { // 如果parent是cancelCtx类型，且有children这个map属性，那么从children中移除掉对应的child这个key
		delete(p.children, child)
	}
	p.mu.Unlock()
}

    # (四) 当子Context(cancelCtx)调用了cancelFunc()函数会发生什么

    • 由于WithCancel(parent Context) (ctx Context, cancel CancelFunc)方法可以多次调用，生成新的子cancelCtx，那么当前这个cancelCtx就会成为这些子cancelCtx的根cancelCtx，如果当前的cancelCtx调用了其cancel()方法，它只能管理器子cancelCtx，让子cancelCtx调用他们的cancel()方法，当前这个cancelCtx的parent是根cancelCtx，所以会从根cancelCtx的children属性中移除掉，被GC回收，且结束当前cancelCtx的所管理的有子goroutine，也包括自己的goroutine，设置children为nil，让子cancelCtx也被GC回收
    // 让当前cancelCtx从根cancelCtx中移除
func removeChild(parent Context, child canceler) {
	p, ok := parentCancelCtx(parent) // 先判断一下parent是否是cancelCtx类型
	if !ok { // 如果不是那么直接return，不需要移除
		return
	}
	p.mu.Lock()
	if p.children != nil { // 如果parent是cancelCtx类型，且有children这个map属性，那么从children中移除掉对应的child这个key
		delete(p.children, child)
	}
	p.mu.Unlock()
}

    # 二：WithTimeout方法、WithDeadline方法派生的timerCtx介绍

    # (一) 阅读Context源码后，对结构体的继承、方法重写的思考

    • 下面这个例子涵盖的知识点有：
      • 1：结构体基于embeded的形式实现继承
      • 2：子结构体拥有父结构体的所有方法和属性
      • 3：构造函数返回的一般是接口类型，而不是结构体类型，目的是验证结构体实现了接口的方法
      • 4：结构体的方法是指针接受者，那么在构造函数中返回的必须是结构体实例的指针
      • 5：子结构体可以重写父类的方法，包括父类实现接口的方法
    // Adder 接口
type Adder interface {
	add(x, y int) int
}

// addFather 父类结构体
type addFather struct {
	a int
}

// NewAddFather 父类构造函数
func NewAddFather() Adder  {
	return &addFather{}
}

// add 父类实现接口的方法
func (a *addFather) add(x, y int) int {
	return x + y
}

// addSon 子类结构体
type addSon struct {
	addFather
}

// NewAddSon 子类构造函数
func NewAddSon() Adder {
	return &addSon{}
}

// sub 子类的sub方法
func (a *addSon) sub(x, y int) int {
	return x - y
}

// add 子结构体重写父结构体的add方法
func (a *addSon) add(x, y int) int {
	return x + x + y
}



func main() {

	son := NewAddSon()
	fmt.Printf("son.add(1, 2)=%d\n", son.add(1, 2))
	//son.sub(1,2) // Adder接口类型没有这个方法, 报错内容：son.sub undefined (type Adder has no field or method sub)

	// 如果要想调用sub方法，必须实现结构体，这也就是golang中构造函数返回实例，和直接初始化结构体的区别
	trueSon := &addSon{}
	fmt.Printf("trueSon.sub(1,2) = %d\n", trueSon.sub(1,2))
	trueSon.a = 2
	fmt.Printf("trueSon.a = %d\n", trueSon.a)

}

    • 1：在timerCtx结构体中，继承了cancelCtx，在cancelCtx基础之上添加了timer和deadline属性

    • 2：WithDeadline方法也是在WithCancel方法的基础之上，添加了超时逻辑，一旦超时，会自动调用cancelCtx.WithCancel方法，最后返回timerCtx实例和调用了cancel方法的闭包函数对象
    func WithDeadline(parent Context, d time.Time) (Context, CancelFunc) {

	// 当parent为父Context的时候，cur为time.Time类型的0值，ok为false
	if cur, ok := parent.Deadline(); ok && cur.Before(d) {
		return WithCancel(parent)
	}

	// 实例化timerCtx
	c := &timerCtx{
		cancelCtx: newCancelCtx(parent),
		deadline:  d,
	}

	// 主要作用是如果parent是父Context，直接return，什么都不做，如果父parent是cancelCtx，那么将child加入到cancelCtx的children属性中
	propagateCancel(parent, c)

	// 获取当前时间到d时间的时间间隔
	dur := time.Until(d)

	// 如果时间间隔超时了，那么直接调用cancel方法
	if dur <= 0 {
		c.cancel(true, DeadlineExceeded)
		return c, func() { c.cancel(false, Canceled) }
	}

	c.mu.Lock()
	defer c.mu.Unlock()
	if c.err == nil { // 基于超时时间间隔设置一个定时任务，一旦超时，就调用cancel方法
		c.timer = time.AfterFunc(dur, func() {
			c.cancel(true, DeadlineExceeded)
		})
	}
	// 最后返回一个timerCtx对象和调用了cancel方法的闭包函数对象
	return c, func() { c.cancel(true, Canceled) }

}

    • 3：timerCtx重写了父类cancelCtx的cancel方法，在父类的cancel方法之上，添加了从父Context中移除timerCtx功能，以及清空定时器让timer内存被GC回收
    func (c *timerCtx) cancel(removeFromParent bool, err error) {

	// 先调用父类的cancel方法
	c.cancelCtx.cancel(false, err)

	// 将parent的parent从父Context中移除
	if removeFromParent {
		removeChild(c.cancelCtx.Context, c)
	}

	// 停止定时器，并且释放定时器的内存空间
	c.mu.Lock()
	if c.timer != nil {
		c.timer.Stop()
		c.timer = nil
	}
	c.mu.Unlock()
}

    # 四：最后贴出context包的实现代码，带注释

    /**
 * @Author: Robby

package context

import (
	"errors"
	"internal/reflectlite"
	"sync"
	"sync/atomic"
	"time"
)


// Context 接口 【看懂了】
type Context interface {

	// Deadline返回一个到期的time时间点, 以及当前是否以及到期
	Deadline() (deadline time.Time, ok bool)

	// Done()返回一个只读channel，只要可以从里面读取到值，说明context关闭了， 可以控制goroutine的停止
	Done() <-chan struct{}

	// 如果context关闭了，返回一个具体的错误
	Err() error

	// 返回context携带的value值
	Value(key interface{}) interface{}
}

// 全局的error实例 【看懂了】
var Canceled = errors.New("context canceled")

// emptyCtx类型是int的别名 【看懂了】
type emptyCtx int

// background和todo是int类型的指针【看懂了】
var (
	background = new(emptyCtx)

// CancelFunc是一个没有参数和返回值的函数 【看懂了】
type CancelFunc func()

// WithCancel 返回一个子ctx和cancel函数， 主要的派生方法，这个方法非常重要
func WithCancel(parent Context) (ctx Context, cancel CancelFunc) {
	c := newCancelCtx(parent) // 基于父context，初始化一个子cancelCtx

	// 这里必须保证c实现了cancel和Done方法, 如果parent是父Context，直接return，什么都不做，如果父parent是cancelCtx，那么将child加入到cancelCtx的children属性中
	propagateCancel(parent, &c) // 当parent调用了cancel方法，那么让所有的child也调用cancel方法

	// 将child实例对象和cancelFunc返回
	return &c, func() { c.cancel(true, Canceled) }
}

var goroutines int32

// propagateCancel 主要作用是如果parent是父Context，直接return，什么都不做，如果父parent是cancelCtx，那么将child加入到cancelCtx的children属性中
func propagateCancel(parent Context, child canceler) {

	done := parent.Done() // 获取父Context的只读channel
	if done == nil {
		return
	}

	select {

	case <-done: // 如果父Context的只读channel关闭了，那么子context也必须关闭，并且把父context的err传递到子context

		child.cancel(false, parent.Err())
		return

	default:

	}

	// 判断parent是否是cancelCtx类型
	if p, ok := parentCancelCtx(parent); ok { // 如果parent是cancelCtx类型

		p.mu.Lock()

		if p.err != nil { // 如果parent有err错误，让子cancelCtx关闭channel
			child.cancel(false, p.err)

		} else { // 如果parent没有err错误
			if p.children == nil {  // 判断一下parent的children是不是空
				p.children = make(map[canceler]struct{}) // 如果是空，初始化一下
			}
			p.children[child] = struct{}{} // 将子cancelCtx加入到parent的children map中存储

		}

		p.mu.Unlock()

	} else { // 如果parent不是cancelCtx类型, 可能parent是自己实现的Context，才会启动goroutine,让child调用cancel方法，否则只是将child加入到children的map中

		atomic.AddInt32(&goroutines, +1) // 让goroutines+1

		go func() { // 开启一个goroutine，
			select {
			case <-parent.Done(): // 那么只要parent的channel关闭了，那么所有的child也应该调用cancel方法
				child.cancel(false, parent.Err())

			case <-child.Done(): // 如果是child的channel关闭了，那么什么也不做
			}
		}()
	}

}

var cancelCtxKey int

// parentCancelCtx 判断parent是否是cancelCtx 【看懂了】
func parentCancelCtx(parent Context) (*cancelCtx, bool) {

	// 获取父context的只读channel
	done := parent.Done()
	// 基于父context的只读channel判断，
	if done == closedchan || done == nil { // 如果channel为全局的closedchan，或者为nil，那么parent不是cancelCtx
		return nil, false
	}

	// 判断一下，父context的value是不是cancelCtx类型，如果不是，那么说明父context不是cancelCtx
	// parent.Value(&cancelCtxKey)返回的是parent的类型本身
	p, ok := parent.Value(&cancelCtxKey).(*cancelCtx)
	if !ok {
		return nil, false
	}

	p.mu.Lock()
	ok = p.done == done  // 判断done属性 和 Done()的返回值是否相等，如果不是那么父context不是cancelCtx
	p.mu.Unlock()

	if !ok {
		return nil, false
	}

	return p, true

}

// 将child自己从parent中移除掉
func removeChild(parent Context, child canceler) {
	p, ok := parentCancelCtx(parent) // 先判断一下parent是否是cancelCtx类型
	if !ok { // 如果不是那么直接return，不需要移除
		return
	}
	p.mu.Lock()
	if p.children != nil { // 如果parent是cancelCtx类型，且有children这个map属性，那么从children中移除掉对应的child这个key
		delete(p.children, child)
	}
	p.mu.Unlock()
}

// canceler child实例必须实现两个方法cancel和Done， 接口的实例是cancelCtx
type canceler interface {
	cancel(removeFromParent bool, err error)
	Done() <-chan struct{}
}

var closedchan = make(chan struct{})

// 刚刚导入包，会先将全局的只读channel关闭
func init() {
	close(closedchan)
}

// 这个是主要的结构体 【看懂了】
type cancelCtx struct {
	Context // 这里切入到一个Context接口，其实就是父context

	mu       sync.Mutex // 互斥锁
	done     chan struct{} // 只读channel
	// children是一个map，key是cancelCtx类型，value是struct{}{}空值
	children map[canceler]struct{}
	err      error // 错误信息
}

// 初始化方法，初始化一个子context实例
func newCancelCtx(parent Context) cancelCtx {
	return cancelCtx{Context: parent}
}

// 返回一个只读的channel，一旦这个channel关闭，就会立即读出struct{}的零值 【】
func (c *cancelCtx) Done() <-chan struct{} {
	c.mu.Lock()

	if c.done == nil {
		c.done = make(chan struct{})
	}
	d := c.done

	c.mu.Unlock()

	return d
}

// Err 返回err错误， 错误从cancelCtx获取到
func (c *cancelCtx) Err() error {
	c.mu.Lock()
	err := c.err
	c.mu.Unlock()
	return err
}

// 实现了string方法
type stringer interface {
	String() string
}

// 返回Context的名称 【看懂了】
func contextName(c Context) string {
	if s, ok := c.(stringer); ok {
		return s.String()
	}
	return reflectlite.TypeOf(c).String()
}

// String 对cancelCtx结构体打印字符串 【看懂了】
func (c *cancelCtx) String() string {
	return contextName(c.Context) + ".WithCancel"
}

// cancel cancel函数的功能是关闭channel 【看懂了】
func (c *cancelCtx) cancel(removeFromParent bool, err error) {

	// 在关闭channel的时候，必须给一个err
	if err == nil {
		panic("context: internal error: missing cancel error")
	}

	c.mu.Lock()
	if c.err != nil {
		c.mu.Unlock()
		return
	}
	// 将错误赋值给context
	c.err = err

	if c.done == nil {  // 如果只读channel是nil，那么初始化一个空channel
		c.done = closedchan
	} else {
		close(c.done) // 这是函数的核心，关闭只读channel
	}

	// 这里是遍历canceler，只获取key
	for child := range c.children {
		// 让所有的canceler都调用自己的cancel方法，但是没有让canceler从父context中移除
		child.cancel(false, err)
	}

	// 所有的子canceler都关闭了channel后，就可以将children释放掉了
	c.children = nil

	c.mu.Unlock() // 再次解锁，避免死锁

	if removeFromParent {
		removeChild(c.Context, c)
	}
}


// Value 重写了父Context的Value方法
func (c *cancelCtx) Value(key interface{}) interface{} {
	// 如果key是nil，那么直接返回自己
	if key == &cancelCtxKey {
		return c
	}
	// 否则返回
	return c.Context.Value(key)
}

type deadlineExceededError struct{} // deadline结构体
var DeadlineExceeded error = deadlineExceededError{} // 构造函数
func (deadlineExceededError) Error() string   { return "context deadline exceeded" } // 方法
func (deadlineExceededError) Timeout() bool   { return true }
func (deadlineExceededError) Temporary() bool { return true }

// timerCtx 继承cancelCtx类型，在cancelCtx基础之上添加了timer和deadline属性
type timerCtx struct {
	cancelCtx
	timer *time.Timer

	deadline time.Time
}

// WithDeadline 接收一个父Context和Time对象
func WithDeadline(parent Context, d time.Time) (Context, CancelFunc) {

	// 当parent为父Context的时候，cur为time.Time类型的0值，ok为false
	if cur, ok := parent.Deadline(); ok && cur.Before(d) {
		return WithCancel(parent)
	}

	// 实例化timerCtx
	c := &timerCtx{
		cancelCtx: newCancelCtx(parent),
		deadline:  d,
	}

	// 主要作用是如果parent是父Context，直接return，什么都不做，如果父parent是cancelCtx，那么将child加入到cancelCtx的children属性中
	propagateCancel(parent, c)

	// 获取当前时间到d时间的时间间隔
	dur := time.Until(d)

	// 如果时间间隔超时了，那么直接调用cancel方法
	if dur <= 0 {
		c.cancel(true, DeadlineExceeded)
		return c, func() { c.cancel(false, Canceled) }
	}

	c.mu.Lock()
	defer c.mu.Unlock()
	if c.err == nil { // 基于超时时间间隔设置一个定时任务，一旦超时，就调用cancel方法
		c.timer = time.AfterFunc(dur, func() {
			c.cancel(true, DeadlineExceeded)
		})
	}
	// 最后返回一个timerCtx对象和调用了cancel方法的闭包函数对象
	return c, func() { c.cancel(true, Canceled) }

}



// Deadline 重写父类的Deadline方法
func (c *timerCtx) Deadline() (deadline time.Time, ok bool) {
	return c.deadline, true
}

// String 重写父类的String方法
func (c *timerCtx) String() string {
	return contextName(c.cancelCtx.Context) + ".WithDeadline(" +
		c.deadline.String() + " [" +
		time.Until(c.deadline).String() + "])"
}

// cancel 重写父类的cancel方法
func (c *timerCtx) cancel(removeFromParent bool, err error) {

	// 先调用父类的cancel方法
	c.cancelCtx.cancel(false, err)

	// 将parent的parent从父Context中移除
	if removeFromParent {
		removeChild(c.cancelCtx.Context, c)
	}

	// 停止定时器，并且释放定时器的内存空间
	c.mu.Lock()
	if c.timer != nil {
		c.timer.Stop()
		c.timer = nil
	}
	c.mu.Unlock()
}

// WithTimeout 其实就是在WithDeadline方法上计算了时间间隔，核心逻辑与timerCtx一样
func WithTimeout(parent Context, timeout time.Duration) (Context, CancelFunc) {
	return WithDeadline(parent, time.Now().Add(timeout))
}

// WithValue 返回一个valueCtx实例，但是没有cancelFunc函数
func WithValue(parent Context, key, val interface{}) Context {
	if key == nil {
		panic("nil key")
	}
	// 判断类型是否是可比较的
	if !reflectlite.TypeOf(key).Comparable() {
		panic("key is not comparable")
	}
	// 判断 返回一个ctx对象
	return &valueCtx{parent, key, val}
}

// 返回value的字符串类型
func stringify(v interface{}) string {
	switch s := v.(type) {
	case stringer:
		return s.String()
	case string:
		return s
	}
	return "<not Stringer>"
}

// 打印valueCtx的key和value
func (c *valueCtx) String() string {
	return contextName(c.Context) + ".WithValue(type " +
		reflectlite.TypeOf(c.key).String() +
		", val " + stringify(c.val) + ")"
}

// Value 重写父类的value方法，基于key返回valueCtx的value值
func (c *valueCtx) Value(key interface{}) interface{} {
	if c.key == key {
		return c.val
	}
	return c.Context.Value(key)
}
