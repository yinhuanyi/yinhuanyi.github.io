    # (三)：连接Redis单节点

    • V8之前的版本，操作Redis不需要传递context上下文参数
    // 声明一个全局的rdb变量
var client *redis.Client

// 初始化连接
func Init(cfg *settings.RedisConfig) (err error) {

	client = redis.NewClient(&redis.Options{
		Addr:     fmt.Sprintf("%s:%d", cfg.Host, cfg.Port),
		Password: cfg.Password,
		DB:       cfg.Db,
		PoolSize: cfg.PoolSize,
	})

	_, err = client.Ping().Result()
	if err != nil {
		log.Fatalf("Redis连接失败: %v\n", err)
	}

	return nil
}

func Close() {
	_ = client.Close()
}

    package main

import (
	"context"
	"fmt"
	"time"

	"github.com/go-redis/redis/v8" // 注意导入的是新版本
)

var (
	rdb *redis.Client
)

// 初始化连接
func initClient() (err error) {
	rdb = redis.NewClient(&redis.Options{
		Addr:     "localhost:16379",
		Password: "",  // no password set
		DB:       0,   // use default DB
		PoolSize: 100, // 连接池大小
	})

	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()
	
	// 对Redis的操作都需要传递ctx参数
	_, err = rdb.Ping(ctx).Result()
	return err
}



// 包括对Redis的数据类型的CRUD操作都需要传递ctx参数
ctx := context.Background()
if err := initClient(); err != nil {
	return
}

err := rdb.Set(ctx, "key", "value", 0).Err()
if err != nil {
	panic(err)
}

val, err := rdb.Get(ctx, "key").Result()
if err != nil {
	panic(err)
}
fmt.Println("key", val)

val2, err := rdb.Get(ctx, "key2").Result()
if err == redis.Nil {
	fmt.Println("key2 does not exist")
} else if err != nil {
	panic(err)
} else {
	fmt.Println("key2", val2)
}

		MasterName:    "master",

	})
	_, err = rdb.Ping().Result()
	if err != nil {
		return err
	}
	return nil
}

    # 二：Redis的命令列表 参考文档 (opens new window)

    # (一)：键(key) 命令

    • 在key的命令主要是在命令行对key的临时查看，修改等操作
    1：Type：返回 key 所储存的值的类型

2：Pexpireat：设置 key 的过期时间以毫秒计

3：Pexpireat：设置 key 过期时间的时间戳(unix timestamp) 以毫秒计

4：Rename：修改 key 的名称

5：Persist：移除 key 的过期时间，key 将持久保持

6：Move：将当前数据库的 key 移动到给定的数据库 db 当中

7：Randomkey：从当前数据库中随机返回一个 key 

8：Dump：序列化给定 key ，并返回被序列化的值

9：Ttl：以秒为单位，返回给定 key 的剩余生存时间(TTL, time to live)

10：Expire：seconds 为给定 key 设置过期时间

11：DEL：删除 key

12：Pttl：以毫秒为单位返回 key 的剩余的过期时间

13：Renamenx：仅当 newkey 不存在时，将 key 改名为 newkey 

14：Exists：检查给定 key 是否存在

15：Expireat：EXPIREAT 的作用和 EXPIRE 类似，都用于为 key 设置过期时间 不同在于 EXPIREAT：接受的时间参数是 UNIX 时间戳(unix timestamp)

16：Keys：查找所有符合给定模式( pattern)的 key

    # (二)：字符串(String) 命令

    • 下面是String类型的命令
    1：Setnx：只有在 key 不存在时设置 key 的值

2：Getrange：返回 key 中字符串值的子字符

3：Mset：同时设置一个或多个 key-value 对。

4：Setex：将值 value 关联到 key ，并将 key 的过期时间设为 seconds (以秒为单位)。

5：SET：设置指定 key 的值

6：Get：获取指定 key 的值

7：Getbit：对 key 所储存的字符串值，获取指定偏移量上的位(bit)

8：Setbit：对 key 所储存的字符串值，设置或清除指定偏移量上的位(bit)

9：Decr：将 key 中储存的数字值减一

10：Decrby：key 所储存的值减去给定的减量值（decrement） 

11：Strlen：返回 key 所储存的字符串值的长度

12：Msetnx：同时设置一个或多个 key-value 对，当且仅当所有给定 key 都不存在

13：Incrby：将 key 所储存的值加上给定的增量值（increment）

14：Incrbyfloat：将 key 所储存的值加上给定的浮点增量值（increment） 

15：Setrange：用 value 参数覆写给定 key 所储存的字符串值，从偏移量 offset 开始

16：Psetex：这个命令和 SETEX 命令相似，但它以毫秒为单位设置 key 的生存时间，而不是像 SETEX 命令那样，以秒为单位

17：Append：如果 key 已经存在并且是一个字符串， APPEND 命令将 value 追加到 key 原来的值的末尾

18：Getset：将给定 key 的值设为 value ，并返回 key 的旧值(old value)

19：Mget：获取所有(一个或多个)给定 key 的值

20：Incr：将 key 中储存的数字值增一

    # (三)：哈希表(Hash) 命令

    • 下面是Hash类型的命令
    1：Hmset：同时将多个 field-value (域-值)对设置到哈希表 key 中

2：Hmget：获取所有给定字段的值

3：Hset：将哈希表 key 中的字段 field 的值设为 value 

4：Hgetall：获取在哈希表中指定 key 的所有字段和值

5：Hget：获取存储在哈希表中指定字段的值

6：Hexists：查看哈希表 key 中，指定的字段是否存在

7：Hincrby：为哈希表 key 中的指定字段的整数值加上增量 increment 

8：Hlen：获取哈希表中字段的数量

9：Hdel：删除一个或多个哈希表字段

10：Hvals：获取哈希表中所有值

11：Hincrbyfloat：为哈希表 key 中的指定字段的浮点数值加上增量 increment 

12：Hkeys：获取所有哈希表中的字段

13：Hsetnx：只有在字段 field 不存在时，设置哈希表字段的值。

    # (四)：列表(List) 命令

    • 下面是List类型的命令
    1：Lindex：通过索引获取列表中的元素

2：Rpush：在列表中添加一个或多个值

3：Lrange：获取列表指定范围内的元素

4：Rpoplpush：移除列表的最后一个元素，并将该元素添加到另一个列表并返回

5：Blpop：移出并获取列表的第一个元素， 如果列表没有元素会阻塞列表直到等待超时或发现可弹出元素为止

6：Brpop：移出并获取列表的最后一个元素， 如果列表没有元素会阻塞列表直到等待超时或发现可弹出元素为止

7：Brpoplpush：从列表中弹出一个值，将弹出的元素插入到另外一个列表中并返回它； 如果列表没有元素会阻塞列表直到等待超时或发现可弹出元素为止

8：Lrem：移除列表元素

9：Llen：获取列表长度

10：Ltrim：对一个列表进行修剪(trim)，就是说，让列表只保留指定区间内的元素，不在指定区间之内的元素都将被删除

11：Lpop：移出并获取列表的第一个元素

12：Lpushx：将一个或多个值插入到已存在的列表头部

13：Linsert：在列表的元素前或者后插入元素

14：Rpop：移除并获取列表最后一个元素

15：Lset：通过索引设置列表元素的值

16：Lpush：将一个或多个值插入到列表头部

17：Rpushx：为已存在的列表添加值

    # (六)：有序集合(sorted set) 命令

    • 下面是ZSet类型的命令
    1：Zrevrank：返回有序集合中指定成员的排名，有序集成员按分数值递减(从大到小)排序

4：Zremrangebyrank：移除有序集合中给定的排名区间的所有成员

5：Zcard：获取有序集合的成员数

6：Zrem：移除有序集合中的一个或多个成员

7：Zinterstore：计算给定的一个或多个有序集的交集并将结果集存储在新的有序集合 key 中

8：Zrank：返回有序集合中指定成员的索引

9：Zincrby：有序集合中对指定成员的分数加上增量 increment

10：Zrangebyscore：通过分数返回有序集合指定区间内的成员

11：Zrangebylex：通过字典区间返回有序集合的成员

12：Zscore：返回有序集中，成员的分数值

13：Zremrangebyscore：移除有序集合中给定的分数区间的所有成员

14：Zscan：迭代有序集合中的元素（包括元素成员和元素分值）

15：Zrevrangebyscore：返回有序集中指定分数区间内的成员，分数从高到低排序

16：Zremrangebylex：移除有序集合中给定的字典区间的所有成员

17：Zrevrange：返回有序集中指定区间内的成员，通过索引，分数从高到底

18：Zrange：通过索引区间返回有序集合成指定区间内的成员

19：Zcount：计算在有序集合中指定区间分数的成员数

20：Zadd：向有序集合添加一个或多个成员，或者更新已存在成员的分数

    # 四：Redis的Pipeline、事务、Watch相关操作

    # (一)：Redis的Pipeline操作：当我们有多条命令要执行时，就可以考虑使用pipeline来优化

    • Pipeline 主要是一种网络优化，它本质上意味着客户端缓冲一堆命令并一次性将它们发送到服务器，这些命令不能保证在事务中执行。这样做的好处是节省了每个命令的网络往返时间（RTT）
    // 创建一个pipeline
pipe := rdb.Pipeline()


incr := pipe.Incr("pipeline_counter") // 第一条Redis命令
pipe.Expire("pipeline_counter", time.Hour) // 第二条Redis命令

_, err := pipe.Exec() // 两条命令同时发送给服务器

fmt.Println(incr.Val(), err)

    # (二)：Redis的事务操作

    • Redis是单线程的，因此单个命令始终是原子的，但是来自不同客户端的两个给定命令可以依次执行，例如在它们之间交替执行，因此在并发情况下，Redis需要TxPipeline来保证多个命令的原子性。TxPipeline与Pipeline使用方法类似，但是TxPipeline保证了多个命令的原子性

    # (三)：Redis的Watch操作

    • 在某些场景下，我们除了要使用TxPipeline()命令外，还需要配合使用Watch命令。在用户使用Watch命令监视某个键之后，直到该用户执行事务的这段时间里，如果有其他用户抢先对被监视的键进行了替换、更新、删除等操作，那么当用户尝试执行事务的时候，事务将失败并返回一个错误，用户可以根据这个错误选择重试事务或者放弃事务
    // 定义一个事务函数
txf := func(tx *redis.Tx) error {
	
	// 先获取一下key的值，如果key的值有变化，那么err会有值
	n, err := tx.Get(key).Int()
	if err != nil && err != redis.Nil {
		return err
	}
	
	// 在err没有值的情况下，执行下面的事务操作
	_, err = tx.Pipelined(func(pipe redis.Pipeliner) error {
		pipe.Set(key, n+1, 0)
		return nil
	})
	return err
}

// 监控key的变化的过程中，执行txf事务函数，如果key有变化，事务函数立即返回err
err := rdb.Watch(txf, key)

    • 下面的代码是，模拟多个goroutine，对Redis中的某个key的值进行修改，如果在修改的过程中，key的值发生了变化，会返回err，然后再次重试
    func transactionDemo() {
	var (
		maxRetries   = 1000
		routineCount = 10
	)

	// Increment 使用GET和SET命令以事务方式递增Key的值
	increment := func(key string) error {
		// 事务函数
		txf := func(tx *redis.Tx) error {
			// 获得key的当前值或零值
			n, err := tx.Get(key).Int()
			if err != nil && err != redis.Nil {
				return err
			}

			// 实际的操作代码（乐观锁定中的本地操作）
			n++

			// 操作仅在 Watch 的 Key 没发生变化的情况下提交
			_, err = tx.TxPipelined(func(pipe redis.Pipeliner) error {
				pipe.Set(key, n, 0)
				return nil
			})

			return err
		}

		// 最多重试 maxRetries 次
		for i := 0; i < maxRetries; i++ {

			// 监控key的同事，执行txf函数
			err := rdb.Watch(txf, key)
			if err == nil {
				// 成功
				return nil
			}

			// 如果事务执行失败，也就是说，再这个过程中key发生变化了，直接continue，重新执行事务操作
			if err == redis.TxFailedErr {
				// 乐观锁丢失 重试
				continue
			}
			// 返回其他的错误
			return err
		}

		return errors.New("increment reached maximum number of retries")
	}

	// 模拟 routineCount 个并发同时去修改 counter3 的值
	var wg sync.WaitGroup
	wg.Add(routineCount)

	for i := 0; i < routineCount; i++ {
		go func() {
			defer wg.Done()
			if err := increment("counter3"); err != nil {
				fmt.Println("increment error:", err)
			}
		}()
	}

	wg.Wait()

	n, err := rdb.Get("counter3").Int()
	fmt.Println("ended with", n, err)
}

    # 五：执行lua脚本操作Redis

    # (一)：Redis中执行lua脚本

    • Redis允许执行lua脚本来操作Redis中的数据，使用lua脚本操作Redis有如下知识点
    1：Lua脚本可以实现原子性操作，避免并发情况下造成临界资源争抢，造成数据不一致问题

2：如果Redis需要原子性操作，将多个操作捆绑为一个事务操作，可以使用txPipeline，也可以使用lua脚本，如果操作细节较多推荐使用lua脚本

3：为了保证安全性，在 Lua 脚本中不要定义自己的全局变量

4：Lua 脚本不要进行高耗时操作

5：通过lua脚本生成的sha值，可以在Redis中进行缓存，多次执行，减少了通信成本

    # (二)：在高并发情况下，4个用户发送6个请求抢购商品

    • 这是一个扣减库存的操作，扣减库存必须是事务性的，因此使用lua脚本操作redis，下面代码保证了一个用户只能购买一次商品
    func createScript() *redis.Script {
	script := redis.NewScript(`
		
			-- 声明局部变量
			local goodsSurplus
			local flag
			
			-- existUserIds：hadBuyUids
			-- goodsSurplusKey：goodsSurplus
			local existUserIds    = tostring(KEYS[1])
			local goodsSurplusKey = tostring(KEYS[2])
			
			-- memberUid：5824742984
			local memberUid       = tonumber(ARGV[1])
			
			-- 执行Redis命令：查看existUserIds集合中是否存在memberUid
			local hasBuy = redis.call("sIsMember", existUserIds, memberUid)
			
			-- 判断用户是否买了商品，如果为0，说明没有买，如果不为0，说明买了，那么直接结束脚本
			if hasBuy ~= 0 then
			  return 0
			end
			
			-- 如果用户买有买，获取goodsSurplusKey商品的库存，如果库存不存在，那么直接结束脚本
			goodsSurplus =  redis.call("GET", goodsSurplusKey)
			if goodsSurplus == false then
			  return 0
			end
			
			-- 如果库存存在，且库存为0，那么直接结束脚本
			goodsSurplus = tonumber(goodsSurplus)
			if goodsSurplus <= 0 then
			  return 0
			end
			
			-- 扣减库存：先将用户的id加入到已买集合，再让库存减一
			flag = redis.call("SADD", existUserIds, memberUid)
			flag = redis.call("DECR", goodsSurplusKey)
			
			-- 最后返回1，表示扣减库存成功
			return 1
	`)
	return script
}

// evalScript 执行lua脚本
func evalScript(client *redis.Client, userId string, wg *sync.WaitGroup) {
	defer wg.Done()

	// 获取脚本对象
	script := createScript()

	// 创建sha值与脚本的对应关系
	sha, err := script.Load(client).Result()
	if err != nil {
		log.Fatalln(err)
	}

	// 执行sha值对应的脚本，key是"hadBuyUids", "goodsSurplus"，args是userId，例如5824742984，keys和args会传递到Redis脚本中
	ret := client.EvalSha(sha, []string{"hadBuyUids", "goodsSurplus"}, userId)

	// 获取Redis的返回结果
	if result, err := ret.Result(); err != nil {
		log.Fatalf("Execute Redis fail: %v\n", err.Error())
	} else {
		if result.(int64) == 1 {
			fmt.Printf("userId：%s，第一次抢购商品，扣减库存成功\n", userId)
		} else {
			fmt.Printf("userId：%s，已经抢购过商品，扣减库存失败\n", userId)
		}
	}

}

func main() {
	var wg sync.WaitGroup
	client := redis.NewClient(&redis.Options{
		Addr:     "localhost:6379",
		Password: "",
		DB:       0,
	})

	// 这里假设4个用户发送了6个请求去扣减库存
	for _, v := range []string{"5824742984", "5824742984", "5824742983", "5824742983", "5824742982", "5824742980"} {
		wg.Add(1)
		go evalScript(client, v, &wg)
	}
	wg.Wait()

}
