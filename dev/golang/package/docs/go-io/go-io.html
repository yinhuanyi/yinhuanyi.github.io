    # 一：io.io包

    # (一)：概念

    • Golang的输入和输出操作是使用原语实现的，这些原语将数据模拟成可读的或可写的字节流，Go 的 io 包提供了 io.Reader 和 io.Writer 接口，分别用于数据的输入和输出，IO包支持对内存结构，文件，网络连接等资源进行操作，如图

    # (二)：io.Reader 接口表示一个读取器，它将数据从某个资源读取到传输缓冲区。在缓冲区中，数据可以被流式传输和使用

    • 1：io.Reader是一个接口，任何类型只要实现了Read(p []byte) (n int, err error)都可以认为实现了Reader接口，下面的功能是将reader中的数据写入数据到缓冲区，最后从缓冲区读取数据的过程
    func main() {

	// 这里是通过strings.NewReader方法创建了一个Reader实例，这个Reader类型实现了Read方法，那么就可以认为reader实例是一个读取器
	reader := strings.NewReader("Clear is better than clever")

	// 定义读取写进写入的缓冲区
	p := make([]byte, 4)

	// 定义所有的字节写入的缓冲区
	var results []byte
	
	// 循环将reader中的数据，拷贝到缓冲区p中，每次循环缓冲区的数据都会被覆盖掉
	for {
		n, err := reader.Read(p)
		if err != nil{
			if err == io.EOF { // 如果读取完毕，返回0，EOF
				fmt.Println("EOF:", n)
				break
			}
			fmt.Println(err)
			os.Exit(1)
		}
		// 遍历缓冲区字节，添加到results里面
		for _, v := range p[:n] {
			results = append(results, v)
		}
		fmt.Println(n, string(p[:n]))
	}

	fmt.Printf("results=%s\n", results)

}

    • 2：自己实现一个MyReader结构体，且结构体实现Read方法，满足io.Reader接口
    type MyReader struct {
	src string  // 读取的数据
	cur int     // 当前读取的位置
}

func NewMyReader(src string) *MyReader {
	return &MyReader{src: src}
}

// alpha 过滤函数
func alpha(r byte) byte {
	if (r >= 'A' && r <= 'Z') || (r >= 'a' && r <= 'z') {
		return r
	}
	return 0
}

// Read 实现io.Reader的方法, p是缓存字节数组
func (m *MyReader) Read (p []byte) (int, error) {

	// 如果当前读取的位置已经到了末尾，那么直接返回io.EOF
	if m.cur >= len(m.src) {
		return 0, io.EOF
	}


	x := len(m.src) - m.cur

	n, bound := 0, 0

	if x >= len(p) {
		// 剩余长度超过缓冲区大小，说明本次可完全填满缓冲区
		bound = len(p)

	} else if x < len(p) {
		// 剩余长度小于缓冲区大小，使用剩余长度输出，缓冲区不补满
		bound = x

	}

	// 内部缓冲区
	buf := make([]byte, bound)

	for n < bound {
		// 每次读取一个字节，执行过滤函数
		if char := alpha(m.src[m.cur]); char != 0 {
			buf[n] = char
		}
		n++
		m.cur++
	}

	// 将缓冲区数据拷贝一份到缓冲区p中
	copy(p, buf)

	return n, nil

}

func main() {
	reader := NewMyReader("Hello! 110, my name is Robby")

	p := make([]byte, 4)

	var results []byte

	for {
		n, err := reader.Read(p)

		if err != nil{
			if err == io.EOF {
				fmt.Println("EOF:", n)
				break
			}
			fmt.Println(err)
			os.Exit(1)
		}

		for _, v := range p[:n] {
			results = append(results, v)
		}

		fmt.Println(n, string(p[:n]))
	}

	fmt.Printf("results=%s\n", results)
}

    • 3：通过继承的方式，在io.Reader的基础上继续扩展其他功能，这里使用strings.NewReader作为接口的实现，这种方式在开发过程中很常用
    type MyReader struct {
	reader io.Reader
}

func NewMyReader(reader io.Reader) *MyReader {
	return &MyReader{reader: reader}
}

func alpha(r byte) byte {
	if (r >= 'A' && r <= 'Z') || (r >= 'a' && r <= 'z') {
		return r
	}
	return 0
}

func (m *MyReader) Read(p []byte) (int, error) {
	// 调用的父类的Read方法，由于子类的方法与父类同名，那么调用父类方法的时候，需要使用父类的实例调用
	n, err := m.reader.Read(p)

	if err != nil {
		return n, err
	}

	buf := make([]byte, n)

	for i := 0; i < n; i++ {
		if char := alpha(p[i]); char != 0 {
			buf[i] = char
		}
	}

	copy(p, buf)

	return n, nil
}

    # (三)：io.Writer 表示一个编写器，它从缓冲区读取数据，并将数据写入目标资源

    • 1：io.Writer是一个接口，任何类型只要实现了Write(p []byte) (n int, err error)方法，都可以认为实现了Writer接口，下面的功能是将缓冲区的数据写入到writer中，然后writer再打印出来
    func main() {

	proverbs := []string{
		"I am Robby",
		"I am Chinese",
		"I love China",
	}

	// 初始化writer
	var writer bytes.Buffer

	for _, p := range proverbs {
		// 从缓冲区p中读取数据，写入到writer中
		n, err := writer.Write([]byte(p))
		if err != nil {
			fmt.Println(err)
			os.Exit(1)
		}

		// 如果缓冲区数据与写入数据不等，说明写入丢失数据了
		if n != len(p) {
			fmt.Println("failed to write data")
			os.Exit(1)
		}
	}

	fmt.Println(writer.String())
}

    • 2：自己实现一个MyWriter
    type MyWriter struct {
	ch chan byte
}

func NewMyWriter() *MyWriter {
	return &MyWriter{make(chan byte, 1024)}
}

func (m *MyWriter) Chan() <-chan byte {
	return m.ch
}

func (m *MyWriter) Write(p []byte) (int, error) {
	n := 0
	// 遍历输入数据，按字节写入目标资源
	for _, b := range p {
		m.ch <- b
		n++
	}
	return n, nil
}

func (m *MyWriter) Close() error {
	close(m.ch)
	return nil
}

func main() {
	writer := NewMyWriter()

	go func() {
		defer writer.Close()
		writer.Write([]byte("Stream "))
		writer.Write([]byte("me!"))
	}()

	for c := range writer.Chan() {
		fmt.Printf("%c", c)
	}
	fmt.Println()
}

    # 六：小结

    # (一)： io.io包、bufio.bufio包、io.ioutil.ioutil包、bytes.buffer包、os.file包的异同和使用场景

    • 1：io库属于底层接口定义库，其作用是是定义一些基本接口和一些基本常量，并对这些接口的作用给出说明，常见的接口有io.Reader、io.Writer等，任何实现了io.Reader接口或者io.Writer接口的实例，都可以调用Read或者Write方法。一般用这个库只是为了调用它的一些常量，比如io.EOF

    • 2：ioutil库包含在io目录下，它的主要作用是作为一个工具包，里面有一些比较实用的函数，比如 ReadAll(从某个源读取数据)、ReadFile（读取文件内容）、WriteFile（将数据写入文件）、ReadDir（获取目录)，ioutil包进一步封装了IO的方法，让调用者使用更加简单

    • 3：os库主要是跟操作系统打交道，所以文件操作基本都会跟os库挂钩，比如创建文件、打开一个文件等，这个库往往会和ioutil库、bufio库等配合使用

    • 4：bufio库可以理解为在io库基础上加了缓存功能，实现了内存读写，提升了IO速度

    • 5：bufio VS ioutil库：两者都提供了对文件的读写功能，唯一的不同就是bufio多了一层缓存的功能，这个优势主要体现读取大文件的时候（ioutil.ReadFile是一次性将内容加载到内存，如果内容过大，很容易爆内存），bufio可以设置缓冲区大小，从而很好的控制了内存的利用率

    • 6：bufio VS bytes.Buffer：两者都提供一层缓存功能，它们的不同在于 bufio 主要针对的是文件到内存的缓存，而 bytes.Buffer 的针对的是内存到内存的缓存
