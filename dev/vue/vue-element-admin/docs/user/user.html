    • 编辑views/user-manage/components/Roles.vue组件，获取当前用户的角色数据，让某些角色在el-checkbox被选中
    <script setup>
import { defineProps, defineEmits, ref, watch } from 'vue'
import { getUserRoles } from '@/api/user-manager'
// ...

const props = defineProps({
  // ...
  userId: {
    type: String,
    required: true
  }
})

// ...

// 当前用户角色
const userRoleTitleList = ref([])
// 获取当前用户角色
const getUserRolesData = async () => {
  const res = (await getUserRoles(props.userId)).data
  // 获取角色名称列表，赋值给userRoleTitleList数组
  userRoleTitleList.value = res.role.map((item) => item.title)
}

// 由于props.userId不是响应式数据，因此需要使用一个函数对象来提供响应式能力
watch(
  () => props.userId,
  // 这个val是变化的userId
  (val) => {
    if (val) getUserRolesData()
  }
)

    # （六）基于RBAC的权限控制体现原理与实现

    • RBAC权限控制最基础部分是权限部分，其主要功能如下
    1：我们可以先为 员工角色 指定 空权限

5：然后我们重新登录 超级管理员 账号

6：为 员工角色 指定 员工管理 && 分配角色 权限

7：然后为我们的 测试用户 指定指定 员工角色

8：此时我们重新登录 测试用户

9：可以发现左侧菜单中多出 员工管理 页面，并且页面中仅存在指定的 分配角色 功能

    import { publicRoutes, privateRoutes } from '@/router'

const state = {
  // 初始路由表
  routes: publicRoutes
}

const mutations = {
  // 动态增加路由，将newRoutes与publicRoutes合并
  setRoutes: (state, newRoutes) => {
    state.routes = [...publicRoutes, ...newRoutes]
  }
}

const actions = {
  // 根据权限数据，过滤路由. menus是权限数据
  filterRoutes: ({ commit, state }, menus) => {
    // routes 是过滤之后，匹配权限的路由表，后面会通过addRoute合并到路由表数组中
    const routes = []

    // 过滤privateRoutes中的路由配置
    menus.forEach((key) => {
      const matchedRoutes = privateRoutes.filter((item) => item.name === key)
      routes.push(...matchedRoutes)
    })

    // 将所有不匹配的路由，全部进入到404的路由配置
    // 注意：该配置需要在所有路由指定之后
    // Todo: 为什么要加一个这样的路由呢？
    routes.push({
      path: '/:catchAll(.*)',
      redirect: '404'
    })

    // 调用mutations，传递动态routes
    commit('setRoutes', routes)

    return routes
  }
}

export default {
  namespaced: true,
  state,
  mutations,
  actions
}

    import store from '@/store'
function checkPermission(el, binding) {
  // 获取指令绑定的数据. 例如v-pin="200"， 那么value就是200
  const { value } = binding
  // 获取用户的所有功能权限
  const points = store.getters.userInfo.permission.points
  // 保证value必须是一个数组
  if (value && value instanceof Array) {
    // 判断用户权限是否在指令指定的权限内，如果有一个权限包含，返回ture，some方法用于检查point是否满足对应的条件
    const hasPermissoin = points.some((point) => {
      // 如果value数组中有对应的point，那么返回true
      return value.includes(point)
    })
    // 如果没有权限, 那么删除这个el节点，相当于隐藏
    if (!hasPermissoin) {
      // 获取当前节点的父节点，让父节点将子节点删除
      el.parentNode && el.parentNode.removeChild(el)
    }
  } else {
    throw new Error('v-permission value must be ["admin", "editor"]')
  }
}

export default {
  // 在绑定元素在父组件中被挂载之后调用
  // el是dom元素，binding是传递的数据
  mounted(el, binding) {
    checkPermission(el, binding)
  },
  // 在包含组件的VNodey以及其子组件的VNode更新后被调用
  update(el, binding) {
    checkPermission(el, binding)
  }
}
