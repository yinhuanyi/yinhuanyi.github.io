    # 2：jump - 切换分页的回调

    • 这个属性用于与服务器端数据交互
    <script>
  layui.use("laypage", function () {
      var laypage = layui.laypage;

      //执行一个laydate实例
      laypage.render({
        // 这不不要加#
        elem: 'page',
        // 数据总数
        count: 5000,
        // limit: 10,
        // 指定每页的数据条数
        limits: [10, 20, 30, 40, 50],
        // 自定义分页组件
        layout: ['prev', 'page', 'next', 'limit'],
        // 分页的回调
        jump: function(obj, first){
            //obj包含了当前分页的所有参数，比如：
            // 得到当前页，以便向服务端请求对应页的数据。
            console.log(obj.curr);
            // 得到每页显示的条数, 告诉服务器返回多少条数据
            console.log(obj.limit);

            //首次不执行
            if(!first){
                console.log("每次点击分页都会调用, 在这里可以向服务器发送请求");
            }
        }
      });
  });
</script>

    # 2：方法渲染

    • 表格方法渲染中，数据请求会使用到的参数为
    table.render({ 
  // get请求的时候，默认携带的参数为：page和limit，这两个参数是告诉服务器端前端请求是第几页的数据，且需要多少条
  url: '/api/data/'
  // 这里是自定义的参数
  where: {token: 'sasasas', id: 123} 
  // 如果是post请求，需要加入这个参数
  method: 'post' 
  // 如果后台需要自定义查询参数，不使用page和limit，那么需要在这个request参数中转换
  request: {} 
  // 如果后台返回的字段不是标准的code、msg、count、data，那么就需要在这个response里面将后台的字段转换标准字段
  response: {} 
});

<!--使用表格模块-->
<script>
  // 加载表格模块
  layui.use('table', function () {
      // 获取表格对象
      var table = layui.table;

      // 渲染表格
      table.render({
        // 选择表格元素
        elem: '#demo',
        // 设置高度
        // height: 312,
        limit: 15,
        // 请求json数据
        url: "{% static 'layuiadmin/json/table/simple.js' %}",
        // 开启分页功能
        page: true,
        //   开启工具条
        toolbar: 'true',
        // 使用字段过滤和导出功能
        defaultToolbar: ['filter', 'exports'],
        // 指定排序字段
        initSort: {
          //排序字段，对应 cols 设定的各字段名,
          field: 'id',
          //排序方式  asc: 升序、desc: 降序、null: 默认排序
          type: 'asc'
        },
        // 设置表格容器到浏览器底部距离为20px
        height: 'full-20',
        // 设置字段
        cols: [[
            // fixed: 'left'表示第一个字段固定在表左边
            {field: 'id', title: 'ID', sort: true, fixed: 'left',align: 'center'},
            // 表格左侧添加复选框
            {type:'checkbox'},
            // edit: 'text'表示字段可以编辑
            {field: 'username', title: '用户名', edit: 'text', align: 'center'},
            // align: 'center' 文字单元格居中
            {field: 'email', title: '邮件', align: 'center'},
            {field: 'sex', title: '性别', align: 'center'},
            {field: 'city', title: '城市', align: 'center'},
            {field: 'sign', title: '签名', align: 'center'},
            {field: 'experience', title: '积分', sort: true, align: 'center'},
            {field: 'ip', title: 'IP', minWidth: 100, align: 'center'},
            {field: 'logins', title: '登录次数', sort: true, align: 'center'},
            {field: 'joinTime', title: '创建时间', align: 'center'},
            // 添加工具栏
            {fixed: 'right', width:150, align:'center', toolbar: '#barDemo'}

        ]]
      });

      // tool 是工具条事件名，simple-table是table原始容器的属性 lay-filter="对应的值"
      // 这个就是监听表格的工具条的触发的动作
      table.on('tool(simple-table)', function (obj) {
        //获得当前行数据, {id: "10003", username: "王勃", email: "xianxin@layui.com", sex: "男", city: "浙江杭州", …}city: "浙江杭州"email: "xianxin@layui.com"experience: "65"id: "10003"ip: "192.168.0.8"joinTime: "2016-10-14"logins: "106"sex: "男"sign: "人生恰似一场修行"username: "王勃"__proto__: Object
        var data = obj.data;

        // 获得lay-event对应的值 , edit
        var layEvent = obj.event;

        if(layEvent === 'del'){
          // 首先前端删除
          layer.confirm('真的删除行么', function(index){
          obj.del(); //删除对应行（tr）的DOM结构，并更新缓存
          layer.close(index);

          // 再向服务端发送删除指令

          });

        } else if(layEvent === 'edit'){
          //编辑的时候，先获取到编辑的数据，


          //同步更新缓存对应的值, 哪个字段的值修改了就更新哪个字段即可
          obj.update({
            username: '123'
            ,title: 'xxx'
          });

        } else if(layEvent === 'LAYTABLE_TIPS'){
          layer.alert('Hi，头部工具栏扩展的右侧图标。');
        }

      });
  });
</script>

    # 4：table模块的主要方法

    //设定全局默认参数。options即各项基础参数
table.set(options); 
//事件监听。event为内置事件名（详见下文），filter为容器lay-filter设定的值
table.on('event(filter)', callback); 
//filter为容器lay-filter设定的值，options即各项基础参数。例子见：转换静态表格
table.init(filter, options); 
//获取表格选中行（下文会有详细介绍）。id 即为 id 参数对应的值
table.checkStatus(id);
//用于表格方法级渲染，核心方法。应该不用再过多解释了，详见：方法级渲染 
table.render(options); 
//表格重载
table.reload(id, options); 
//重置表格尺寸
table.resize(id); 
//导出数据
table.exportFile(id, data, type);

    • 自定义验证方式
    form.verify({
  //value：表单的值、item：表单的DOM对象
  username: function(value, item){ 
    if(!new RegExp("^[a-zA-Z0-9_\u4e00-\u9fa5\\s·]+$").test(value)){
      return '用户名不能有特殊字符';
    }
    if(/(^\_)|(\__)|(\_+$)/.test(value)){
      return '用户名首尾不能出现下划线\'_\'';
    }
    if(/^\d+\d+\d$/.test(value)){
      return '用户名不能全为数字';
    }
  }
});
