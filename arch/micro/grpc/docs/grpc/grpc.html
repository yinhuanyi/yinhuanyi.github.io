func (s *Server) SayHello(ctx context.Context, request *proto.HelloRequest) (*proto.HelloReply, error) {
	return &proto.HelloReply{
		Message: "hello " + request.Name,
	}, nil
}

func main() {

	// 1：实例化server
	g := grpc.NewServer()

	// 2：注册业务逻辑，这里是把Server这个结构体注册进来了，结构体里面有SayHello这个方法
	proto.RegisterGreeterServer(g, &Server{})

	// 3：设置server监听端口
	lis, err := net.Listen("tcp", "0.0.0.0:8080")
	if err != nil {
		panic(err)
	}

	// 4：启动服务, 这里使用了协程处理连接请求
	err = g.Serve(lis)
	if err != nil {
		panic(err)
	}

}

    # 三：gRPC的四种数据流(golang实现)

    • 数据流模式分类
    1：一元调用（之前演示的都是一元调用）

2：服务端流式调用

3：客户端流式调用

4：双向流式调用

    # (一)：一元调用

    • 普通RPC调用，客户端带一个请求对象进行调用，服务端返回一个响应对象，这个就是上面演示的内容

    # (二)：服务端流式调用

    • 客户端带一个请求对象进行调用，服务端返回多个响应对象，或者说是一段连续的数据流
    典型案例：客户端向服务器端发送一个币对，服务器端将币对的实时数据响应给客户端，这个有点像websocket长连接
    1

    # (三)：客户端流式调用

    • 客户端带有多个对象进行请求，服务端返回一个响应对象
    典型案例：物联网终端向服务器端发送数据，这个有点像监控的数据上报服务
    1

// 定义一个暴露的service
service Greeter {

  // 服务器端流模式，这里的StreamResData使用了stream关键字修饰
  rpc GetStream (StreamReqData) returns (stream StreamResData);

  // 客户端流模式，
  rpc PutStream (stream StreamReqData) returns (StreamResData);

  // 双向流模式
  rpc AllStream(stream StreamReqData) returns (stream StreamResData);

}

message StreamReqData {
  string data = 1;
}

message StreamResData {
  string data = 1;
}

    const PORT = ":28080"

type Server struct{}

// GetStream 服务器端流模式
func (s *Server) GetStream(req *proto.StreamReqData, res proto.Greeter_GetStreamServer) error {

	i := 0

	for {
		i++

		// 将数据发送给客户端
		err := res.Send(&proto.StreamResData{
			Data: fmt.Sprintf("%v", time.Now().Unix()),
		})

		if err != nil {
			panic(err)
		}

		time.Sleep(time.Second)

		if i > 10 {
			break
		}

	}

	return nil

}

// PutStream 客户端流模式
func (s *Server) PutStream(res proto.Greeter_PutStreamServer) error {

	for {
		// 不断获取客户端发送的数据
		a, err := res.Recv()
		if err != nil {
			// 这里如果出错，一定是break，不能panic
			fmt.Println(err)
			break
		}
		fmt.Println(a)
	}
	return nil

}

// AllStream 双向流模式
func (s *Server) AllStream(allStr proto.Greeter_AllStreamServer) error {
	wg := sync.WaitGroup{}
	wg.Add(2)
	// 启动两个协程，一个协程负责接收数据，一个协程负责发送数据
	go func() {
		defer wg.Done()
		for {
			data, _ := allStr.Recv()
			fmt.Printf("服务器端收到:%v\n", data.Data)
		}
	}()

	go func() {
		defer wg.Done()
		for {
			_ = allStr.Send(&proto.StreamResData{
				Data: "我是服务器端",
			})
			time.Sleep(time.Second)
		}
	}()
	wg.Wait()
	return nil
}

func main() {

	// 1：实例化server
	g := grpc.NewServer()

	// 2：注册业务逻辑
	proto.RegisterGreeterServer(g, &Server{})

	// 3：设置server监听端口
	lis, err := net.Listen("tcp", PORT)
	if err != nil {
		panic(err)
	}

	// 4：启动服务, 这里使用了协程处理连接请求
	err = g.Serve(lis)
	if err != nil {
		panic(err)
	}

}

const PORT = "127.0.0.1:28080"

func main() {

	// 1：连接服务器
	conn, err := grpc.Dial(PORT, grpc.WithInsecure())
	if err != nil {
		panic(err)
	}
	defer conn.Close()

	// 2：创建client
	c := proto.NewGreeterClient(conn)

	// （一）服务器端流模式
	res, err := c.GetStream(context.Background(), &proto.StreamReqData{Data: "Robby"})
	if err != nil {
		panic(err)
	}
	// 死循环不断获取服务器端数据
	for {
		// 如果服务器端发送停止，return nil，err就是EOF
		a, err := res.Recv()
		if err != nil {
			// 这里一定是break
			fmt.Println(err)
			break
		}
		fmt.Printf("a = %v\n", a.Data)
	}

	// （二）客户端流模式
	putS, err := c.PutStream(context.Background())
	if err != nil {
		panic(err)
	}
	i := 0
	for {
		i++
		_ = putS.Send(&proto.StreamReqData{Data: fmt.Sprintf("%d", time.Now().Unix())})

		time.Sleep(time.Second)

		if i > 10 {
			break
		}
	}

	// （三）双向流
	allS, err := c.AllStream(context.Background())
	if err != nil {
		panic(err)
	}
	wg := sync.WaitGroup{}
	wg.Add(2)

	go func() {
		defer wg.Done()
		for {
			a, _ := allS.Recv()
			fmt.Printf("客户端收到: %v\n", a.Data)
		}
	}()

	go func() {
		defer wg.Done()
		for {
			_ = allS.Send(&proto.StreamReqData{
				Data: "我是客户端",
			})
			time.Sleep(time.Second)
		}
	}()

	wg.Wait()
}
