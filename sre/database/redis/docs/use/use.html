    # (四)：库存超卖问题的解决方案

    • 扣减库存的操作，在高并发下请求过多，那么会导致超卖的情况出现。解决的方法是将扣减库存的逻辑放在Redis中来实现
    1：声明一个Redis的key，这个key可以是前缀+活动Id

2：基于抢购活动的字段数据，构建一个map，写入到Redis中

3：在Redis中，将map的amount字段进行自减，那么这里就是将原本在MySQL记录中做的扣减库存操作，改为在Redis中来做

4：那在Redis中扣除库存的时候，必须保证amount不能变的情况下，才能进行amount--操作，因此在Golang中需要使用watch+txPipeline的事务方式来做，保证amount的查看和修改必须在一个线程中完成，且是原子性操作。当然，也可以在Golang中调用Redis的lua脚本，将这个操作封装在lua脚本中完成

5：那么MySQL中的的活动表就转移到了Redis中了，MySQL的活动表就不用了

    # (五)：分布式锁(多进程锁)解决方案

    • Golang自带的sync.Mutex互斥锁只能解决单一进程中不同的goroutine之间的共享。但是在生产业务中，一般都是分布式的多进程业务场景，在这种情况下，要保证多个进程都共享锁，那么就需要使用分布式锁，分布式锁满足下面三个条件
    1：多进程可见

2：互斥

3：可重入：同一线程或进程可以在同一时刻对临界资源重复加锁

    • 在Redis中，有一个setnx指令，只有当key不存在的时候，才能插入成功，否则插入失败。那么setnx指令基本用法如下

127.0.0.1:6379> del account

127.0.0.1:6379> setnx account 1

    • 但是这里还有一个问题，例如有三个进程A,B,C。A进程使用setnx指令设置了一把锁，但是由于运行时间过长，出现timeout了，那么锁自动被删除了，但是此时A进程还没有执行del指令去释放锁。此时B进程就可以访问临界资源了(因为没有锁)，B进程使用setnx指令设置了一把锁，然后A进程运行到释放锁的指令，将B进程的锁删除了。此时C进程就可以访问临界资源了(因为B进程的锁被删除了)，那么此时C再加上一把锁，对临界资源进行访问。在这个过程中，B、C进程之间无法保证同一时刻只能有一个进程访问临界资源，因此临界资源的操作又会有问题。解决问题的方法也很简单，让不同进程只能释放自己加的锁
    那么在加锁的时候，也就是执行setnx指令的时候，key对应的值可以是requestId、cookieId、goroutineid等有唯一标识的值，那么在执行del指令的时候，先判断，是否是自己设置的值，如果是就删除，如果说明是别人设置的，就不删除

    # (六)：可重入锁

    • 在goroutine在并发访问的时候，可能会出现死锁的情况。同一个goroutine两次申请锁，就会出现阻塞panic，因为Mutex是不可重入锁，因为第一次申请的锁还没有释放，第二次申请锁的前提提交是第一次申请的锁被释放了，第一次申请的锁被释放的前提条件是第二次申请锁再释放锁，例如下面的代码
    var x int64
var wg sync.WaitGroup
var lock sync.Mutex

func Call() {
	defer wg.Done()

	lock.Lock() // 第一次申请锁
	incr()
	lock.Unlock() // 第一次释放锁，必须等incr()函数执行完毕，但是incr()函数里面需要申请锁
}

func incr() {
	for i := 0; i < 1000; i++ {
		lock.Lock() // 第二次申请锁，必须要第一次申请的锁是否掉
		x++
		lock.Unlock()
	}
}

func main() {

	wg.Add(1)
	go Call()
	fmt.Println("goroutine1启动")
	go Call()
	fmt.Println("goroutine2启动")
	wg.Wait()
	fmt.Println(x)

}

    • 解决这个不可重入锁问题也很简单，就是设计可重入锁，在Golang中标准库中并没有可重入锁，因为Golang认为，如果你需要可重入锁，说明你的代码有问题了。下面是Golang中可重入锁的设计思想，下面的代码中，GoroutineId通过github.com/cch123/goroutineid第三方库获取
    type ReentrantLock struct {
	lock      *sync.Mutex
	cond      *sync.Cond
	recursion int32 // 计数器
	host      int64
}

func NewReentrantLock() sync.Locker {
	res := &ReentrantLock{
		lock:      new(sync.Mutex),
		recursion: 0,
		host:      0,
	}
	res.cond = sync.NewCond(res.lock)
	return res
}

// Lock 可重入锁加锁
func (rt *ReentrantLock) Lock() {
	id := goroutineid.GetGoID()
	fmt.Println(id)

	rt.lock.Lock()

	defer rt.lock.Unlock()

	// 如果当前goroutine拿到这把锁，那么直接recursion+1就好
	if rt.host == id {
		rt.recursion++
		return
	}

	// 如果是其他的goroutine拿到这把锁，那么需要阻塞等待
	for rt.recursion != 0 {
		fmt.Println("阻塞等待")
		rt.cond.Wait()
	}

	// 让可重入锁设置为其他的goroutine的id
	rt.host = id
	rt.recursion = 1
}

// Unlock 可重入锁解锁
func (rt *ReentrantLock) Unlock() {
	rt.lock.Lock()
	defer rt.lock.Unlock()

	// goroutine解锁的时候，如果发现recursion为0，或者不是自己的锁，直接return，什么都不做，或者panic也可以
	if rt.recursion == 0 || rt.host != goroutineid.GetGoID() {
		panic(fmt.Sprintf("the wrong call host: (%d); current_id: %d; recursion: %d", rt.host, goroutineid.GetGoID(), rt.recursion))
		//fmt.Printf("the wrong call host: (%d); current_id: %d; recursion: %d\n", rt.host, goroutineid.GetGoID(), rt.recursion)
		//return
	}

	rt.recursion--

	// 如果发现recursion为0了，说明锁已经可以被释放了
	if rt.recursion == 0 {
		rt.cond.Signal() // 通知
	}

}

func main() {
	lock := NewReentrantLock()

	for i := 0; i < 500; i++ {
		go func() {
			lock.Lock()
		}()

	}

}

    # (七)：Redis中的可重入锁的解决方案

    • 设计思路如下
    1：假设锁的key为lock，hashKey是当前goroutineId，锁超时自动释放的时间为20s

2：获取锁的步骤：
	①：判断lock这个key是否存在：【EXISTS lock】
	②：lock不存在，获取锁，记录重入次数，第一次为1：【 Hset lock goroutineId 1】，且设置超时时间
	③：lock存在，说明有人已经获取锁了，判断一下是不是自己的锁，通过判断lock中，是否存在goroutineId这个hashKey：【HEXISTS lock goroutineId】
		i：如果goroutineId这个hashKey不存在，那么说明不是自己的锁，那么此时不会再次创建hashKey了必须等别人的hashKey删除掉才行，因此获取锁失败
		ii：如果goroutineId这个hashKey存在，说明是自己的锁，让重入次数+1，且更新hashKey过期时间：【HINCRBY lock goroutineId 1】【EXPIRE lock 20】


3：释放锁的步骤：
	①：先判断goroutineId这个hashKey是否存在：【HEXISTS lock goroutineId】
		i：如果存在，说明自己的锁还在，让重入次数-1：【HINCRBY lock goroutineId -1】，再判断重入次数是否为0，如果为0，说明锁全部释放，需要删除锁：【del lock】
		ii：如果不存在，说明锁已经超时自动被释放了，不需要任何操作

    • 使用lua脚本操作Redis(原子操作)，获取锁代码如下
    -- key：lock
local key = KEYS[1]
local goroutineId = ARGV[1]
-- timeout: 20秒
local timeout = ARGV[2]

-- 判断lock这个key不存在
if (redis.call('exists', key) == 0) then
	-- 获取锁，记录重入次数，第一次为1
	redis.call('hset', key, goroutineId, '1')
	-- 设置超时时间
	redis.call('expire', key, timeout)
	return 1
end

-- 判断是不是自己的锁，如果是自己的锁
if (redis.call('hexists', key, goroutineId) == 1) then
	-- 直接加1
	redis.call('hincrby', key, goroutineId, '1')
	-- 设置超时时间
	redis.call('expire', key, timeout)
	return 1
end

-- 如果是别人的锁，说明获取锁失败，返回0
return 0

var client *redis.Client

type ReentrantLock struct{}

func NewReentrantLock() sync.Locker {

// lockScript 获取锁脚本
func lockScript() *redis.Script {
	script := redis.NewScript(`
		-- key：lock
		local key = KEYS[1]
		local goroutineId = ARGV[1]
		-- timeout: 20秒
		local timeout = ARGV[2]
		
		-- 判断lock这个key不存在

			-- 获取锁，记录重入次数，第一次为1
			redis.call('hset', key, goroutineId, '1')
			-- 设置超时时间
			redis.call('expire', key, timeout)
			return 1
		end
		
		-- 判断是不是自己的锁，如果是自己的锁
		if (redis.call('hexists', key, goroutineId) == 1) then
			-- 直接加1
			redis.call('hincrby', key, goroutineId, '1')
			-- 设置超时时间
			redis.call('expire', key, timeout)
			return 1
		end
		
		-- 如果是别人的锁，说明获取锁失败，返回0
		return 0

	`)
	return script
}

// unlockScript 释放锁脚本
func unlockScript() *redis.Script {
	script := redis.NewScript(`
		-- key：lock
		local key = KEYS[1]
		local goroutineId = ARGV[1]
		
		-- 如果不存在，直接返回0
		if (redis.call('HEXISTS', key, goroutineId) == 0) then
			return 1
		end
		
		-- 判断重入次数是否为0，如果为0，说明锁全部释放，需要删除锁
		local count = redis.call('HINCRBY', key, goroutineId, -1)
		
		if (count == 0) then
			redis.call('DEL', key)
			return 1
		end

		return 1

	`)
	return script
}

// Lock 加锁
func (r *ReentrantLock) Lock() {

	lock := lockScript()
	sha, err := lock.Load(client).Result()
	if err != nil {
		panic(err)
	}
	// 执行脚本
	goroutineId := goroutineid.GetGoID()

	ret := client.EvalSha(sha, []string{"lock"}, goroutineId, 20)

	if result, err := ret.Result(); err != nil {
		log.Panicf("获取锁Execute lua fail: %v\n", err.Error())
	} else {

		if result.(int64) == 1 {
			fmt.Printf("goroutineId：%d，获取锁成功\n", goroutineId)
		} else {
			fmt.Printf("goroutineId：%d，获取锁失败\n", goroutineId)
		}
	}
}

// Unlock 解锁
func (r *ReentrantLock) Unlock() {

	unlock := unlockScript()
	sha, err := unlock.Load(client).Result()
	if err != nil {
		panic(err)
	}
	// 执行脚本
	goroutineId := goroutineid.GetGoID()
	ret := client.EvalSha(sha, []string{"lock"}, goroutineId)

	if _, err = ret.Result(); err != nil {
		log.Panicf("释放锁Execute lua fail: %v\n", err.Error())
	}

	fmt.Printf("goroutineId：%d，释放锁成功\n", goroutineId)
}

    # (八)：Redis中实现一人一单的限购

    • 示例代码
    func createScript() *redis.Script {
	script := redis.NewScript(`
		
			-- 声明局部变量
			local goodsSurplus
			local flag
			
			-- existUserIds：hadBuyUids
			-- goodsSurplusKey：goodsSurplus
			local existUserIds    = tostring(KEYS[1])
			local goodsSurplusKey = tostring(KEYS[2])
			
			-- memberUid：5824742984
			local memberUid       = tonumber(ARGV[1])
			
			-- 执行Redis命令：查看existUserIds集合中是否存在memberUid
			local hasBuy = redis.call("sIsMember", existUserIds, memberUid)
			
			-- 判断用户是否买了商品，如果为0，说明没有买，如果不为0，说明买了，那么直接结束脚本
			if hasBuy ~= 0 then
			  return 0
			end
			
			-- 如果用户没有买，获取goodsSurplusKey商品的库存，如果库存不存在，那么直接结束脚本
			goodsSurplus =  redis.call("GET", goodsSurplusKey)
			if goodsSurplus == false then
			  return 0
			end
			
			-- 如果库存存在，且库存为0，那么直接结束脚本
			goodsSurplus = tonumber(goodsSurplus)
			if goodsSurplus <= 0 then
			  return 0
			end
			
			-- 扣减库存：先将用户的id加入到已买集合，再让库存减一
			flag = redis.call("SADD", existUserIds, memberUid)
			flag = redis.call("DECR", goodsSurplusKey)
			
			-- 最后返回1，表示扣减库存成功
			return 1
	`)
	return script
}

// evalScript 执行lua脚本
func evalScript(client *redis.Client, userId string, wg *sync.WaitGroup) {
	defer wg.Done()

	// 获取脚本对象
	script := createScript()

	// 创建sha值与脚本的对应关系
	sha, err := script.Load(client).Result()
	if err != nil {
		log.Fatalln(err)
	}

	// 执行sha值对应的脚本，key是"hadBuyUids", "goodsSurplus"，args是userId，例如5824742984，keys和args会传递到Redis脚本中
	ret := client.EvalSha(sha, []string{"hadBuyUids", "goodsSurplus"}, userId)

	// 获取Redis的返回结果
	if result, err := ret.Result(); err != nil {
		log.Fatalf("Execute Redis fail: %v\n", err.Error())
	} else {
		if result.(int64) == 1 {
			fmt.Printf("userId：%s，第一次抢购商品，扣减库存成功\n", userId)
		} else {
			fmt.Printf("userId：%s，已经抢购过商品，扣减库存失败\n", userId)
		}
	}
}

	1：string类型：结构比较简单，例如用户信息(用户名，昵称，头像，年龄等)，写入时需要序列化，读取时需要反序列化，如果数据量比较小，可以忽略这种序列化的开销，适合固定数据结构，且数据量较小的数据缓存

	2：hash类型：hash在写入和读取都没有序列化的开销，适合存储数据量较大，且经常变换的数据类型
