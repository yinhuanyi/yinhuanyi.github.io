    # 一：NGINX对静态资源的访问

    # (一) NGINX静态资源访问用到的模块

    • 1：sendfile：让静态资源的请求响应过程，不需要经过用户空间，直接再内核空间完成数据的交互
    # 可以作用在http、server、local、if in location中
sendfile on | off
    1
    2
    • 2：tcp_nopush：在sendfile开启情况下，提高网络包的传输效率。意义是 当有多个数据包需要发送出去，那么将会作为一次向发送
    # 可以作用在http、server、local中
tcp_nopush on | off;
    1
    2
    • 3：tcp_nodelay：在用户要求数据发送实时性比较高的情况下使用，只要有数据包就发送，在keepalive连接情况下，提高网络包传输的实时性
    # 可以作用在http、server、local、if in location中
tcp_nodelay on | off;
    1
    2
    • 4：gzip：在传输的过程中使用压缩
    # 可以作用在http、server、local、if in location中
gzip on | off;
gzip_comp_level level;  # 压缩比率
gzip_http_version 1.0 | 1.1; # http的版本
    1
    2
    3
    4
    5
    • 5：示例：
    	# 创建/app/web/static/目录，在目录下存放静态文件
    location ~ ^/static {
        root   /app/web;
        gzip on;
        gzip_comp_level 2;
        gzip_http_version 1.1;
        gzip_types *;
    }

    # (二) NGINX跨域模块已经解决(需要后台支持跨域，且NGINX支持跨越)

    • 如果一个html页面中，有ajax请求发起对另外一个域名的访问，那么这个时候，就会出现跨域的问题，默认的情况下，浏览器是不允许跨域的，例如下面这个admin.html页面，页面请求的百度，因此会有跨域的问题

    <meta name="viewport"
          content="width=device-width, user-scalable=no, initial-scale=1.0, maximum-scale=1.0, minimum-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <script src="http://libs.baidu.com/jquery/2.0.0/jquery.min.js"></script>
    <title>admin</title>

          success: function () {
              alert("跨域成功");

    # 三：NGINX 作为缓存服务器

    # (一) NGINX缓存

    • 配置示例
    # proxy_cache_path表示开启一个缓存空间
# /opt/cache缓存路径
# levels是缓存的层级
# keys_zone是缓存空间的名称，10m是缓存key的内存大小，1m一般可以存放8000个key，10m可以存放80000个key
# max_size指存储value最大值
# inactive表示如果10分钟未使用到这个缓存，那么将会被清理
# use_temp_path表示关闭临时文件存放目录

proxy_cache_path /opt/cache levels=1:2 keys_zone=nginx_cache:10m max_size=10g inactive=10m use_temp_path=off;

# 设置组
upstream backend {
	server 127.0.0.1:8001;
	server 127.0.0.1:8002;
	server 127.0.0.1:8003;
}

# 当请求以url3、login、password/reset开头，设置cookie_nocache这个自定义变量为1
if ($request_uri ~ ^/(url3|login|passwrod\/reset)){
	set $cookie_nocache 1;
}

location / {
	# 将请求反向代理到后台组
	proxy_pass http://backend;
	# 应用缓存
	proxy_cache nginx_cache;
	# 200和304状态码是1h过期
	proxy_cache_valid 200 304 1h;
	# 其他状态码过期时间是10分钟
	proxy_cache_valid any 10m;
	# 设置缓存key的命名规则
	proxy_cache_key $host$uri$is_args$args;
	# 添加头部信息, 这个头部会出现在response响应头里面，如果Nginx-Cache是HIT，表明命中缓存，如果Nginx-Cache是MISS表明没有命中缓存
	add_header Nginx-Cache "$upstream_cache_status";
	
	# 当$cookie_nocache为1时，不使用缓存
	proxy_no_cache $cookie_nocache;
	
}

    # (二) rewrite重写规则

    • 基本语法
    # 可以作用在location或server中
rewrite regex replacement [flag];
    1
    2
    • flag的区别
    关键字 功能
    last 停止rewrite检测, 重新发起一个请求
    break 停止rewrite检测，不会重新发起一个请求，直接在root指定的目录中寻找
    redirect 返回302临时重定向，地址栏会显示跳转后的地址
    permanent 返回301永久重定向，地址栏会显示跳转后的地址

    # (三) last与break

    • 配置示例
    server {
    listen       80;
    server_name  docs.yhyblog.cn;
    root /app/web;
    index index.html;
    
    location ~ ^/break {
        rewrite ^/break /test/ break;
    }

    location ~ ^/last {
        rewrite ^/last /test/ last;
    }

    location /test/ {
            default_type application/json;
            return 200 '{"status": "success"}';
    }

    error_page   500 502 503 504  /50x.html;
    location = /50x.html {
        root   /usr/share/nginx/html;
    }
}

	# 临时重定向，每次请求都会发起两次请求
	rewrite ^/test http://www.baidu.com redirect;
	# 永久重定向，第一次发起两次请求，第二次只发起一次请求
	rewrite ^/test http://www.baidu.com permanent;
}

    # (一) 基本思想

    • 基本思想
    请求调度到NGINX，会进入到Lua脚本中进行判断，在Lua脚本中获取用户的IP地址或Token，然后在Redis/Memcache中查找，如果能够查找到说明是测试用户，将请求调度到新系统。反之将请求调度到老系统
    1
    • 基本原理
        ngx_lua将Lua嵌入Nginx，可以让Nginx执行Lua脚本，并且高并发、非阻塞的处理各种请求。Lua内建协程，这样就可以很好的将异步回 调转换成顺序调用的形式。ngx_lua在Lua中进行的IO操作都会委托给Nginx的事件模型，从而实现非阻塞调用。开发者可以采用串行的方式编写程 序，ngx_lua会自动的在进行阻塞的IO操作时中断，保存上下文；然后将IO操作委托给Nginx事件处理机制，在IO操作完成后，ngx_lua会 恢复上下文，程序继续执行，这些操作都是对用户程序透明的。
